8;; DESCRIPTION:
9;;
10;; Calculates the magnitude for the 32-bit floating point real FFT.
11;;

20;; typedef RFFT_F32_STRUCT* RFFT_F32_STRUCT_Handle;
21;;
22;; and where RFFT_F32_STRUCT is a structure defined as:
23;;
24;; typedef struct {
25;; float *InBuf;
26;; float *OutBuf;
27;; float *CosSinBuf;
28;; float *MagBuf;
29;; float *PhaseBuf;
30;; uint16_t FFTSize;
31;; uint16_t FFTStages;
32;; } RFFT_F32_STRUCT;
33;;
34;; ASSUMPTIONS:
35;;
36;; * FFTSize must be a power of 2 (32, 64, 128, etc)
37;; * FFTSize must be greater or equal to 32
38;; * FFTStages must be ln(FFTSize)/ln(2)
39;; * InBuf, OutBuf, CosSinBuf are FFTSize in length
40;; * MagBuf is (FFTSize/2 + 1) in length
41;; * PhaseBuf is FFTSize/2 in length
42;; * RFFT_f32() or RFFT_f32u() has been called prior to this function
43;;
44;;
45;; ALGORITHM:
46;;
47;; Calculate the real FFT magnitude. The data is organized as follows in
48;; the output buffer after calculating FFT:
49;;
50;; +----------------------------+
51;; | R(0) = DC | 0
52;; | R(1) |
53;; | R(2) | Real Component
54;; | R(3) |
55;; . .
56;; . .
57;; . .
58;; . .
59;; | R(FFTSize/2) = fs/2 |
60;; | I(FFTSize/2 - 1) |
61;; . .
62;; . .
63;; . .
64;; . .
65;; | I(4) | Imaginary Component
66;; | I(3) |
67;; | I(2) |
68;; | I(1) | FFTSize - 1
69;; +----------------------------+
70;;
71;; CODE OPTIMIZATION:
72;;
73;; The loop was temporally unrolled, and then spatially unrolled once to
74;; eliminate FPU delay slots.
75;;
76;; BENCHMARK: 18 cycles/complex value + overhead
77;;
78;; On Entry: XAR4 = &RFFT_F32_STRUCT

145_RFFT_f32_mag:

147 MOV32 *SP++, R4H ; Save R4H (save-on-entry register)
148 MOV32 *SP++, R5H ; Save R5H (save-on-entry register)
149 MOV32 *SP++, R6H ; Save R6H (save-on-entry register)
150 MOV32 *SP++, R7H ; Save R7H (save-on-entry register)
151
152 MOVL XAR6, *+XAR4[2] ; XAR6 = Rptr = &OutBuf[0]
153 MOVL XAR5, *+XAR4[6] ; XAR5 = &MagBuf
154
155 ADDB XAR4, #10 ; Move the structure pointer up
156
157;--- Calculate pointer to imaginary part results
158 MOV ACC, *+XAR4[0] << 1 ; ACC = 2*FFTSize
159 ADDL ACC, XAR6 ; Add OutBuff address
160 MOVL XAR7, ACC ; XAR7 = Iptr = &OutBuf[last value]
161
162;--- Calculate main loop count
163 MOV AL, *+XAR4[0] ; AL = FFTSize
164 LSR AL, #2 ; AL = FFTSize/4
165 SUBB AL, #2 ; mag[0:1] handled seperately, and "N-1"
166 ; loop count so subtract 2
167;----------------------------------------------------------------------
168; Begin magnitude calculations
169;----------------------------------------------------------------------
170
171;--- Mag[0] is handled special since no Im part
172 MOV32 R0H, *XAR6++ ; R0H = Re[0]
173 ABSF32 R0H, R0H ; R0H = abs(Re[0])
174 MOV32 *XAR5++, R0H ; Store mag[0]
175
176;--- Must also compute Mag[1] seperately to keep an even number of mag
177; computations in the main loop
178
179 ;--- Compute magnitude squared
180 MOV32 R0H, *XAR6++ ; R0H = Re[1]
181 MPYF32 R0H, R0H, R0H ; R0H = Re[1]*Re[1]
182 || MOV32 R1H, *--XAR7 ; R1H = Im[1]
183 MPYF32 R1H, R1H, R1H ; R1H = Im[1]*Im[1]
184 NOP
185 ADDF32 R0H, R0H, R1H ; R0H = (Re[1]*Re[1]) + (Im[1]*Im[1])
186 NOP
187
188 ;--- Do the sqrt
189 EISQRTF32 R1H, R0H ; R1H = Ye[1] = Estimate(1/sqrt(X[1]))
190 MPYF32 R2H, R0H, #0.5 ; R2H = X[1]*0.5
191 MPYF32 R3H, R1H, R1H ; R3H = Ye[1]*Ye[1]
192 NOP
193 MPYF32 R3H, R3H, R2H ; R3H = Ye[1]*Ye[1]*X[1]*0.5
194 NOP
195 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[1]*Ye[1]*X[1]*0.5
196 NOP
197 MPYF32 R1H, R1H, R3H ; R2H = Ye[1] = Ye[1]*(1.5 - Ye[1]*Ye[1]*X[1]*0.5)
198 NOP
199 MPYF32 R3H, R1H, R2H ; R3H = Ye[1]*X[1]*0.5
200 NOP
201 MPYF32 R3H, R1H, R3H ; R3H = Ye[1]*Ye[1]*X[1]*0.5
202 NOP
203 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[1]*Ye[1]*X[1]*0.5
204 CMPF32 R0H, #0.0 ; Check if X[1] == 0
205 MPYF32 R1H, R1H, R3H ; R2H = Ye[1] = Ye[1]*(1.5 - Ye[1]*Ye[1]*X[1]*0.5)
206 NOP
207 MOV32 R1H, R0H, EQ ; If X is zero, change the Ye[1] estimate to 0
208 MPYF32 R0H, R0H, R1H ; R0H = Y[1] = X[1]*Ye[1] = sqrt(X[1])
209 NOP
210
211 ;--- Store the result
212 MOV32 *XAR5++,R0H ; Store mag[1]
213
214;--- Main loop
215
216_rfft_f32_calc_mag:
217
218 RPTB _rfft_f32_mag_end, @AL
219
220 ;--- Compute magnitude squared
221
222 MOV32 R0H, *XAR6++ ; R0H = Re[i]
223 MPYF32 R0H, R0H, R0H ; R0H = Re[i]*Re[i]
224 || MOV32 R1H, *--XAR7 ; R1H = Im[i]
225
226 MOV32 R4H, *XAR6++ ; R4H = Re[i+1]
227 MPYF32 R4H, R4H, R4H ; R4H = Re[i+1]*Re[i+1]
228 || MOV32 R5H, *--XAR7 ; R5H = Im[i+1]
229
230 MPYF32 R1H, R1H, R1H ; R1H = Im[i]*Im[i]
231 MPYF32 R5H, R5H, R5H ; R5H = Im[i+1]*Im[i+1]
232
233 ADDF32 R0H, R1H, R0H ; R0H = (Re[i]*Re[i]) + (Im[i]*Im[i])
234 ADDF32 R4H, R5H, R4H ; R4H = (Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])
235
236 ;--- Do the sqrt
237
238 EISQRTF32 R1H, R0H ; R1H = Ye[i] = Estimate(1/sqrt(X[i]))
239 EISQRTF32 R5H, R4H ; R5H = Ye[i+1] = Estimate(1/sqrt(X[i+1]))
240
241 MPYF32 R2H, R0H, #0.5 ; R2H = X[i]*0.5
242 MPYF32 R6H, R4H, #0.5 ; R6H = X[i+1]*0.5
243
244 MPYF32 R3H, R1H, R1H ; R3H = Ye[i]*Ye[i]
245 MPYF32 R7H, R5H, R5H ; R7H = Ye[i+1]*Ye[i+1]
246
247 MPYF32 R3H, R3H, R2H ; R3H = Ye[i]*Ye[i]*X*0.5
248 MPYF32 R7H, R7H, R6H ; R7H = Ye[i+1]*Ye[i+1]*X[i+1]*0.5
249
250 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[i]*Ye[i]*X[i]*0.5
251 SUBF32 R7H, #1.5, R7H ; R7H = 1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5
252
253 MPYF32 R1H, R1H, R3H ; R1H = Ye[i] = Ye[i]*(1.5 - Ye[i]*Ye[i]*X[i]*0.5)
254 MPYF32 R5H, R5H, R7H ; R6H = Ye[i+1] = Ye[i+1]*(1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5)
255
256 MPYF32 R3H, R1H, R2H ; R3H = Ye[i]*X[i]*0.5
257 MPYF32 R7H, R5H, R6H ; R7H = Ye[i+1]*X[i+1]*0.5
258
259 MPYF32 R3H, R1H, R3H ; R3H = Ye[i]*Ye[i]*X[i]*0.5
260 MPYF32 R7H, R5H, R7H ; R7H = Ye[i+1]*Ye[i+1]*X[i+1]*0.5
261
262 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[i]*Ye[i]*X*0.5
263 CMPF32 R0H, #0.0 ; Check if X[i] == 0
264 MPYF32 R1H, R1H, R3H ; R1H = Ye[i] = Ye[i]*(1.5 - Ye[i]*Ye[i]*X[i]*0.5)
265
266 SUBF32 R7H, #1.5, R7H ; R7H = 1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5
267 MOV32 R1H, R0H, EQ ; If X[i] is zero, change the Ye[i] estimate to 0
268 CMPF32 R4H, #0.0 ; Check if X[i+1] == 0
269 MPYF32 R5H, R5H, R7H ; R5H = Ye[i+1] = Ye[i+1]*(1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5)
270
271 MPYF32 R0H, R0H, R1H ; R0H = Y[i] = X[i]*Ye[i] = sqrt(X[i])
272 MOV32 R5H, R4H, EQ ; If X[i+1] is zero, change the Ye[i+1] estimate to 0
273 MPYF32 R4H, R4H, R5H ; R4H = Y[i+1] = X[i+1]*Ye[i+1] = sqrt(X[i+1])
274
275 ;--- Store the result
276 MOV32 *XAR5++,R0H ; Store mag[i]
277 MOV32 *XAR5++,R4H ; Store mag[i+1]
278
279_rfft_f32_mag_end:
280
281;--- Mag[N/2+1] is handled special since no Im part
282 MOV32 R0H, *XAR6++ ; R0H = Re[N/2+1]
283 ABSF32 R0H, R0H ; R0H = abs(Re[N/2+1])
284 MOV32 *XAR5++, R0H ; Store mag[N/2+1]
285
286;--- Finish up
287 MOV32 R7H, *--SP ; Restore R7H
288 MOV32 R6H, *--SP ; Restore R6H
289 MOV32 R5H, *--SP ; Restore R5H
290 MOV32 R4H, *--SP ; Restore R4H
291 LRETR
