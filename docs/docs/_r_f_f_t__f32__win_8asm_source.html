70;;! \fn RFFT_f32_win
71;;! \brief Real FFT Windowing Function
72;;! \param pBuffer (XAR4), pointer to the buffer that needs to be windowed
73;;! \param pWindow (XAR5), pointer to the windowing table
74;;! \param size (AL), size of the buffer (size of the RFFT)
75;;
76;; This function windows a 2N-point real valued data sequence stored as an
77;; N point complex sequence in the location pointed to by "pBuffer". It is to
78;; be used with the real FFT module
79;;
80;; A 2N-pt real FFT is carried out by first doing an N point complex FFT
81;; followed by post processing (unwrapping) to get the correct answer. The
82;; windows are of size N (they are symmetrical).
83;;
84;; This routine will apply the window in two steps
85;;
86;; Step 1: Forward walk through win
87;; Loop for i = 0 to N - 1 {
88;; p -> pBuffer, q -> win
89;; a = *p (float) , b = *q (float)
90;; c = *(p+1) (float), d = *(q+1)(float)
91;; temp1 = A x B (float)
92;; temp2 = C x D (float)
93;; *p = temp1 , *(p+1) = temp2
94;; p = p + 1
95;; q = q + 1
96;; }
97;;
98;; Step 2: Backward walk through win
99;; Loop for i = N to 2N - 1 {
100;; p -> pBuffer(from the last step), q -> win(N/2)
101;; a = *p (float) , b = *q (float)
102;; c = *(p+1) (float), d = *(q+1)(float)
103;; temp1 = A x B (float)
104;; temp2 = C x D (float)
105;; *p = temp1 , *(p+1) = temp2
106;; p = p + 1
107;; q = q - 1
108;; }
109;;
110;; Register Usage
111;; XAR4 -> pBuffer
112;; XAR5 -> pWindow
113;; AL = size
114;; AR7 = loop counter 1/2
115;; R0H = window element
116;; R1H = buffer element
117;; R2H = window element
118;; R3H = buffer element
119;; Registers Save on Entry = 0
120;;=============================================================================
121;;
122 .text
123_RFFT_f32_win:
124 ; size -> size of RFFT i.e. 2N-point
125 LSR AL, #2 ; AL = (size/4)
126 MOVZ AR7, AL ; AR7 = (size/2)
127 SUBB XAR7,#1 ; AR7 = (size/2)-1
128
129 .align 2
130 RPTB _RFFT_f32_win_loop1, AR7
131 MOV32 R0H, *XAR5++ ; R0H = win[i++]
132 MOV32 R1H, *XAR4 ; R1H = d[i].real
133 MPYF32 R0H, R0H, R1H ; |R0H = (win[i]*d[i].real)
134 || MOV32 R2H, *XAR5++ ; |R2H = win[i] | i++
135 MOV32 R3H, *+XAR4[2] ; +1|R3H = d[i].imag
136 MPYF32 R2H, R2H, R3H ; |R2H = (win[i]*d[i].imag)
137 MOV32 *XAR4++, R0H ; +1|*XAR4.real = R0H | i++
138 MOV32 *XAR4++, R2H ; *XAR4.imag = R2H | ARP = 4
139_RFFT_f32_win_loop1:
140 ; At the end of this loop, XAR5 points to the
141 ; N/2 th element of the real input
142 .align 2
143 RPTB _RFFT_f32_win_loop2, AR7
144 MOV32 R0H, *--XAR5 ; R0H = win[i] | i--
145 MOV32 R1H, *XAR4 ; R1H = data point d[i].real
146 MPYF32 R0H, R0H, R1H ; |R0H = (win[i]*d[i].real)
147 || MOV32 R2H, *--XAR5 ; |R2H = win[i]
148 MOV32 R3H, *+XAR4[2] ; +1|R3H = d[i].imag
149 MPYF32 R2H, R2H, R3H ; |R2H = (win[i]*d[i].imag)
150 MOV32 *XAR4++, R0H ; +1|*XAR4.real = R0H | i++
151 MOV32 *XAR4++, R2H ; *XAR4.imag = R2H | ARP = 4
152_RFFT_f32_win_loop2:
153 LRETR
