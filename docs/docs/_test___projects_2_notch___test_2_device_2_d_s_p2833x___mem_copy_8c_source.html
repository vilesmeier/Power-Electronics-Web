69MemCopy(Uint16 *SourceAddr, Uint16* SourceEndAddr, Uint16* DestAddr)
70{
71 while(SourceAddr < SourceEndAddr)
72 {
73 *DestAddr++ = *SourceAddr++;
74 }
75 return;
76}
