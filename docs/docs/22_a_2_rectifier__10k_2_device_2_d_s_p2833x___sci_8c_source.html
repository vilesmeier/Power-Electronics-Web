71// InitSciGpio - This function initializes GPIO to function as SCI-A, SCI-B, or
72// SCI-C
73//
74// Each GPIO pin can be configured as a GPIO pin or up to 3 different
75// peripheral functional pins. By default all pins come up as GPIO
76// inputs after reset.
77//
78// Caution:
79// Only one GPIO pin should be enabled for SCITXDA/B operation.
80// Only one GPIO pin shoudl be enabled for SCIRXDA/B operation.
81// Comment out other unwanted lines.
82//
83void
84InitSciGpio()
85{
86 InitSciaGpio();
87#if DSP28_SCIB
88 InitScibGpio();
89#endif // if DSP28_SCIB
90
91#if DSP28_SCIC
92 InitScicGpio();
93#endif // if DSP28_SCIC
94}

96//
97// InitSciaGpio - This function initializes GPIO pins to function as SCI-A pins
98//
99void
100InitSciaGpio()
101{
102 EALLOW;
103
104 //
105 // Enable internal pull-up for the selected pins
106 // Pull-ups can be enabled or disabled disabled by the user.
107 // This will enable the pullups for the specified pins.
108 //
109 GpioCtrlRegs.GPAPUD.bit.GPIO28 = 0; // Enable pull-up for GPIO28 (SCIRXDA)
110 GpioCtrlRegs.GPAPUD.bit.GPIO29 = 0; // Enable pull-up for GPIO29 (SCITXDA)
111
112 //
113 // Set qualification for selected pins to asynch only
114 // Inputs are synchronized to SYSCLKOUT by default.
115 // This will select asynch (no qualification) for the selected pins.
116 //
117 GpioCtrlRegs.GPAQSEL2.bit.GPIO28 = 3; // Asynch input GPIO28 (SCIRXDA)
118
119 //
120 // Configure SCI-A pins using GPIO regs
121 // This specifies which of the possible GPIO pins will be SCI functional
122 // pins.
123 //
124 GpioCtrlRegs.GPAMUX2.bit.GPIO28 = 1; // Configure GPIO28 to SCIRXDA
125 GpioCtrlRegs.GPAMUX2.bit.GPIO29 = 1; // Configure GPIO29 to SCITXDA
126
127 EDIS;
128}
