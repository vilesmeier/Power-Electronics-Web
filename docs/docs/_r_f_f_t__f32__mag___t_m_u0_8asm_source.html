8;; DESCRIPTION:
9;;
10;; Calculates the magnitude for the 32-bit floating point real FFT.
11;; Requires a C28x device with TMU0 or higher.
12;; On non-TMU devices, use the RFFT_f32_mag() function.
13;;
14;; HISTORY:
15;; 11/12/14 - optimized v1.40.00.00 function. (D. Alter)
16;;
17;; FUNCTIONS:
18;;
19;; void RFFT_f32_mag_TMU0 (RFFT_F32_STRUCT_Handle)
20;;
21;; where RFFT_F32_STRUCT_Handle is defined as:
22;; typedef RFFT_F32_STRUCT* RFFT_F32_STRUCT_Handle;
23;;
24;; and where RFFT_F32_STRUCT is a structure defined as:
25;;
26;; typedef struct {
27;; float *InBuf;
28;; float *OutBuf;
29;; float *CosSinBuf;
30;; float *MagBuf;
31;; float *PhaseBuf;
32;; uint16_t FFTSize;
33;; uint16_t FFTStages;
34;; } RFFT_F32_STRUCT;
35;;
36;; ASSUMPTIONS:
37;;
38;; * FFTSize must be a power of 2 (32, 64, 128, etc)
39;; * FFTSize must be greater or equal to 32
40;; * FFTStages must be ln(FFTSize)/ln(2)
41;; * InBuf, OutBuf, CosSinBuf are FFTSize in length
42;; * MagBuf is (FFTSize/2 + 1) in length
43;; * PhaseBuf is FFTSize/2 in length
44;; * RFFT_f32() or RFFT_f32u() has been called prior to this function
45;;
46;;
47;; ALGORITHM:
48;;
49;; Calculate the real FFT magnitude. The data is organized as follows in
50;; the output buffer after calculating FFT:
51;;
52;; +----------------------------+
53;; | R(0) = DC | 0
54;; | R(1) |
55;; | R(2) | Real Component
56;; | R(3) |
57;; . .
58;; . .
59;; . .
60;; . .
61;; | R(FFTSize/2) = fs/2 |
62;; | I(FFTSize/2 - 1) |
63;; . .
64;; . .
65;; . .
66;; . .
67;; | I(4) | Imaginary Component
68;; | I(3) |
69;; | I(2) |
70;; | I(1) | FFTSize - 1
71;; +----------------------------+
72;;
73;; CODE OPTIMIZATION:
74;;
75;; The loop was temporally unrolled, and then spatially unrolled once to
76;; eliminate FPU delay slots.
77;;
78;; BENCHMARK: 4 cycles/complex value + overhead
79;;
80;; On Entry: XAR4 = &RFFT_F32_STRUCT

130; RFFT_F32_STRUCT structure offsets from XAR4
131;----------------------------------------------------------------------
132;RFFT_InBuf .set (0*2)
133;RFFT_OutBuf .set (1*2)
134;RFFT_CosSinBuf .set (2*2)
135;RFFT_MagBuf .set (3*2)
136;RFFT_PhaseBuf .set (4*2)
137;RFFT_FFTSize .set (5*2)
138;RFFT_FFTStages .set (RFFT_FFTSize+1)

147_RFFT_f32_mag_TMU0:
148 MOV32 *SP++, R4H ; Save R4H (save-on-entry register)
149
150 MOVL XAR6, *+XAR4[2] ; XAR6 = Rptr = &OutBuf[0]
151 MOVL XAR5, *+XAR4[6] ; XAR5 = &MagBuf
152
153 ADDB XAR4, #10 ; Move the structure pointer up
154
155;--- Calculate pointer to imaginary part results
156 MOV ACC, *+XAR4[0] << 1 ; ACC = 2*FFTSize
157 ADDL ACC, XAR6 ; Add OutBuff address
158 MOVL XAR7, ACC ; XAR7 = Iptr = &OutBuf[last value]
159
160;--- Calculate main loop count
161 MOV AL, *+XAR4[0] ; AL = FFTSize
162 LSR AL, #2 ; AL = FFTSize/4
163 SUBB AL, #3 ; Init loop counter to (N-1) minus 2
164 ; more for prologue/epilogue
165
166;----------------------------------------------------------------------
167; Begin magnitude calculations
168;----------------------------------------------------------------------
169
170;--- Loop prologue
171; Note: there is no Im[0], so mag[0] is handled differently.
172
173 MOV32 R0H, *XAR6++ ; R0H = Re[i]
174
175 MPYF32 R4H, R0H, R0H ; R0H = Re[i]*Re[i]
176 || MOV32 R2H, *XAR6++ ; R2H = Re[i+1]
177 MOV32 R3H, *--XAR7 ; R3H = Im[i+1]
178
179 MPYF32 R2H, R2H, R2H ; R2H = Re[i+1]*Re[i+1]
180 || MOV32 R0H, *XAR6++ ; R0H = Re[i]
181 MPYF32 R3H, R3H, R3H ; R3H = Im[i+1]*Im[i+1]
182 || MOV32 R1H, *--XAR7 ; R1H = Im[i]
183 SQRTF32 R4H, R4H ; R4H = sqrt[(Re[i]*Re[i]) + (Im[i]*Im[i])]
184 ADDF32 R6H, R3H, R2H ; R6H = (Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])
185 || MOV32 *SP++, R6H ; Save R6H (save-on-entry register)
186
187;--- Loop body
188_rfft_f32_calc_mag:
189
190 RPTB _rfft_f32_mag_end, @AL
191
192 MPYF32 R0H, R0H, R0H ; R0H = Re[i]*Re[i]
193 || MOV32 R2H, *XAR6++ ; R2H = Re[i+1]
194 MPYF32 R1H, R1H, R1H ; R1H = Im[i]*Im[i]
195 || MOV32 R3H, *--XAR7 ; R3H = Im[i+1]
196 SQRTF32 R6H, R6H ; R6H = sqrt[(Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])]
197 ADDF32 R4H, R1H, R0H ; R0H = (Re[i]*Re[i]) + (Im[i]*Im[i])
198 || MOV32 *XAR5++,R4H ; Store mag[i]
199
200 MPYF32 R2H, R2H, R2H ; R2H = Re[i+1]*Re[i+1]
201 || MOV32 R0H, *XAR6++ ; R0H = Re[i]
202 MPYF32 R3H, R3H, R3H ; R3H = Im[i+1]*Im[i+1]
203 || MOV32 R1H, *--XAR7 ; R1H = Im[i]
204 SQRTF32 R4H, R4H ; R4H = sqrt[(Re[i]*Re[i]) + (Im[i]*Im[i])]
205 ADDF32 R6H, R3H, R2H ; R6H = (Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])
206 || MOV32 *XAR5++,R6H ; Store mag[i+1]
207
208_rfft_f32_mag_end:
209
210;--- Loop epilogue
211
212; mag[N/2+1] is handled separately, and is not part of the epilogue. But,
213; it is interleaved at the end to eliminate some delay slots. Also, note
214; that it has no Im part so it is handled differently.
215
216 MPYF32 R0H, R0H, R0H ; R0H = Re[i]*Re[i]
217 || MOV32 R2H, *XAR6++ ; R2H = Re[i+1]
218 MPYF32 R1H, R1H, R1H ; R1H = Im[i]*Im[i]
219 || MOV32 R3H, *--XAR7 ; R3H = Im[i+1]
220 SQRTF32 R6H, R6H ; R6H = sqrt[(Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])]
221 ADDF32 R4H, R1H, R0H ; R0H = (Re[i]*Re[i]) + (Im[i]*Im[i])
222 || MOV32 *XAR5++,R4H ; Store mag[i]
223
224 MPYF32 R2H, R2H, R2H ; R2H = Re[i+1]*Re[i+1]
225 MPYF32 R3H, R3H, R3H ; R3H = Im[i+1]*Im[i+1]
226 SQRTF32 R4H, R4H ; R4H = sqrt[(Re[i]*Re[i]) + (Im[i]*Im[i])]
227 ADDF32 R6H, R3H, R2H ; R6H = (Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])
228 || MOV32 *XAR5++,R6H ; Store mag[i+1]
229
230 MOV32 R0H, *XAR6++ ; R0H = Re[N/2+1]
231 ABSF32 R0H, R0H ; R0H = abs(Re[N/2+1])
232 SQRTF32 R2H, R6H ; R2H = sqrt[(Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])]
233 MOV32 *XAR5++,R4H ; Store mag[i]
234
235 NOP
236 MOV32 R6H,*--SP ; Restore R6H
237 MOV32 R4H,*--SP ; Restore R4H
238 MOV32 *XAR5++,R2H ; Store mag[i+1]
239
240 MOV32 *XAR5++, R0H ; Store mag[N/2+1]
241
242;--- Finish up
243 LRETR
