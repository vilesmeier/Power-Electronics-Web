155; This function saturates an input data value to defined upper & lower limits.
156; If the input data lies definitely within limits the function returns 0,
157; otherwise it returns 1.
158; i.e. return 0 if (lowerLim < data < upperLim), otherwise return 1.
159
160; C prototype: uint16_t DCL_runClamp_C1(float *data, float Umax, float Umin)
161; argument 1 = *data: address of input data [XAR4]
162; argument 2 = Umax : upper limit [R0H]
163; argument 3 = Umin : lower limit [R1H]
164; return = integer : 1 = clamped, 0 = not clamped [AL]
165
166 .align 2
167
168_DCL_Clamp_Run:
169 .asmfunc
170 ZAPA ; AL = 0
171 MOV32 R2H, *+XAR4[0] ; R2H = data
172 CMPF32 R2H, R0H ; compare: data, Umax
173 MOVST0 ZF, NF ; copy flags: Z = ZF, N = NF
174 MOVB AL, #1, GEQ ; data >= Umax ---> AL = 1
175 MOV32 R0H, R2H, LT ; data < Umax ---> R0H = data
176 CMPF32 R2H, R1H ; comapare: data, Umin
177 MOVST0 ZF, NF ; copy flags: Z = ZF, N = NF
178 MOVB AL, #1, LEQ ; data <= Umin ---> AL = 1
179 MOV32 R0H, R1H, LT ; data < Umin ---> R0H = Umin
180 MOV32 *+XAR4[0], R0H ; &data = R0H
181 LRETR
182 .endasmfunc
