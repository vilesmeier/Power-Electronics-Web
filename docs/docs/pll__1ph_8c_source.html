42void SOGI_1PH_coeff_calc(SOGI_1PH *spll_obj)
43{
44 float32_t osgx,osgy,temp, wn;
45 wn= spll_obj->fn *(float32_t) 2.0f * (float32_t) 3.14159265f;
46 spll_obj->osg_coeff.osg_k=(float32_t)(0.5);
47 osgx = (float32_t)(2.0f*0.5f*wn*spll_obj->delta_t);
48 spll_obj->osg_coeff.osg_x=(float32_t)(osgx);
49 osgy = (float32_t)(wn*spll_obj->delta_t*wn*spll_obj->delta_t);
50 spll_obj->osg_coeff.osg_y=(float32_t)(osgy);
51 temp = (float32_t)1.0/(osgx+osgy+4.0f);
52 spll_obj->osg_coeff.osg_b0=((float32_t)osgx*temp);
53 spll_obj->osg_coeff.osg_b2=((float32_t)(-1.0f)*spll_obj->osg_coeff.osg_b0);
54 spll_obj->osg_coeff.osg_a1=((float32_t)(2.0*(4.0f-osgy))*temp);
55 spll_obj->osg_coeff.osg_a2=((float32_t)(osgx-osgy-4)*temp);
56 spll_obj->osg_coeff.osg_qb0=((float32_t)(0.5f*osgy)*temp);
57 spll_obj->osg_coeff.osg_qb1=(spll_obj->osg_coeff.osg_qb0*(float32_t)(2.0));
58 spll_obj->osg_coeff.osg_qb2=spll_obj->osg_coeff.osg_qb0;
59}

76void SOGI_1PH_Run(SOGI_1PH *spll_obj,float32_t acValue)
77{
78 // Update the spll_obj->u[0] with the grid value
79 spll_obj->u[0]=acValue;
80
81 //
82 // Orthogonal Signal Generator
83 //
84 spll_obj->osg_u[0] = (spll_obj->osg_coeff.osg_b0*
85 (spll_obj->u[0]-spll_obj->u[2])) +
86 (spll_obj->osg_coeff.osg_a1*spll_obj->osg_u[1]) +
87 (spll_obj->osg_coeff.osg_a2*spll_obj->osg_u[2]);
88
89 spll_obj->osg_u[2]=spll_obj->osg_u[1];
90 spll_obj->osg_u[1]=spll_obj->osg_u[0];
91
92 spll_obj->osg_qu[0]=(spll_obj->osg_coeff.osg_qb0*spll_obj->u[0]) +
93 (spll_obj->osg_coeff.osg_qb1*spll_obj->u[1]) +
94 (spll_obj->osg_coeff.osg_qb2*spll_obj->u[2]) +
95 (spll_obj->osg_coeff.osg_a1*spll_obj->osg_qu[1]) +
96 (spll_obj->osg_coeff.osg_a2*spll_obj->osg_qu[2]);
97
98 spll_obj->osg_qu[2]=spll_obj->osg_qu[1];
99 spll_obj->osg_qu[1]=spll_obj->osg_qu[0];
100
101 spll_obj->u[2]=spll_obj->u[1];
102 spll_obj->u[1]=spll_obj->u[0];
103
104 //
105 // Park Transform from alpha beta to d-q axis
106 //
107 spll_obj->u_Q[0]=(spll_obj->cosine*spll_obj->osg_u[0]) + (spll_obj->sine*spll_obj->osg_qu[0]);
108 spll_obj->u_D[0]=(spll_obj->cosine*spll_obj->osg_qu[0]) - (spll_obj->sine*spll_obj->osg_u[0]);
109
110 //
111 // Loop Filter
112 //
113 spll_obj->ylf[0]=spll_obj->ylf[1] +
114 (spll_obj->lpf_coeff.b0*spll_obj->u_Q[0]) +
115 (spll_obj->lpf_coeff.b1*spll_obj->u_Q[1]);
116 spll_obj->ylf[1]=spll_obj->ylf[0];
117
118 spll_obj->u_Q[1]=spll_obj->u_Q[0];
119
120 //
121 // VCO
122 //
123 spll_obj->fo=spll_obj->fn+spll_obj->ylf[0];
124
125 spll_obj->theta=spll_obj->theta + (spll_obj->fo*spll_obj->delta_t)*(float32_t)(2.0*3.1415926f);
126
127 if(spll_obj->theta>(float32_t)(2.0*3.1415926f))
128 {
129 spll_obj->theta=spll_obj->theta - (float32_t)(2.0*3.1415926f);
130 //spll_obj->theta=0;
131 }
132
133 spll_obj->sine=(float32_t)sinf(spll_obj->theta);
134 spll_obj->cosine=(float32_t)cosf(spll_obj->theta);
135}
