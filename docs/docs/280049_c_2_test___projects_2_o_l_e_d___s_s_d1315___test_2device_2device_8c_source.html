61//*****************************************************************************
62//
63// Function to initialize the device. Primarily initializes system control to a
64// known state by disabling the watchdog, setting up the SYSCLKOUT frequency,
65// and enabling the clocks to the peripherals.
66// The function also configures the GPIO pins 22 and 23 in digital mode.
67// To configure these pins as analog pins, use the function GPIO_setAnalogMode
68//
69// Note : In case XTAL is used as the PLL source, it is recommended to invoke
70// the Device_verifyXTAL() before configuring PLL
71//
72//*****************************************************************************
73void Device_init(void)
74{
75 //
76 // Disable the watchdog
77 //
78 SysCtl_disableWatchdog();
79#ifdef CMDTOOL
80 CMD_init();
81#endif
82
83#ifdef _FLASH
84#ifndef CMDTOOL
85 //
86 // Copy time critical code and flash setup code to RAM. This includes the
87 // following functions: InitFlash();
88 //
89 // The RamfuncsLoadStart, RamfuncsLoadSize, and RamfuncsRunStart symbols
90 // are created by the linker. Refer to the device .cmd file.
91 //
92 memcpy(&RamfuncsRunStart, &RamfuncsLoadStart, (size_t)&RamfuncsLoadSize);
93#endif
94 //
95 // Call Flash Initialization to setup flash waitstates. This function must
96 // reside in RAM.
97 //
98 Flash_initModule(FLASH0CTRL_BASE, FLASH0ECC_BASE, DEVICE_FLASH_WAITSTATES);
99#endif
100
101 //
102 // Set up PLL control and clock dividers
103 //
104 SysCtl_setClock(DEVICE_SETCLOCK_CFG);
105
106 //
107 // Make sure the LSPCLK divider is set to the default (divide by 4)
108 //
109 SysCtl_setLowSpeedClock(SYSCTL_LSPCLK_PRESCALE_4);
110
111 //
112 // These asserts will check that the #defines for the clock rates in
113 // device.h match the actual rates that have been configured. If they do
114 // not match, check that the calculations of DEVICE_SYSCLK_FREQ and
115 // DEVICE_LSPCLK_FREQ are accurate. Some examples will not perform as
116 // expected if these are not correct.
117 //
118 ASSERT(SysCtl_getClock(DEVICE_OSCSRC_FREQ) == DEVICE_SYSCLK_FREQ);
119 ASSERT(SysCtl_getLowSpeedClock(DEVICE_OSCSRC_FREQ) == DEVICE_LSPCLK_FREQ);
120
121#ifndef _FLASH
122 //
123 // Call Device_cal function when run using debugger
124 // This function is called as part of the Boot code. The function is called
125 // in the Device_init function since during debug time resets, the boot code
126 // will not be executed and the gel script will reinitialize all the
127 // registers and the calibrated values will be lost.
128 // Sysctl_deviceCal is a wrapper function for Device_Cal
129 //
130 SysCtl_deviceCal();
131#endif
132
133 //
134 // Turn on all peripherals
135 //
136 Device_enableAllPeripherals();
137
138 //
139 //Disable DC DC in Analog block
140 //
141 ASysCtl_disableDCDC();
142
143 //
144 //Configure GPIO in Push Pull,Output Mode
145 //
146 GPIO_setPadConfig(22U, GPIO_PIN_TYPE_STD);
147 GPIO_setPadConfig(23U, GPIO_PIN_TYPE_STD);
148 GPIO_setDirectionMode(22U, GPIO_DIR_MODE_OUT);
149 GPIO_setDirectionMode(23U, GPIO_DIR_MODE_OUT);
150
151 //
152 // Configure GPIO22 and GPIO23 as digital pins
153 //
154 GPIO_setAnalogMode(22U, GPIO_ANALOG_DISABLED);
155 GPIO_setAnalogMode(23U, GPIO_ANALOG_DISABLED);
156}

261//*****************************************************************************
262//
263// Function to verify the XTAL frequency
264// freq is the XTAL frequency in MHz
265// The function return true if the the actual XTAL frequency matches with the
266// input value
267//
268// Note that this function assumes that the PLL is not already configured and
269// hence uses SysClk freq = 10MHz for DCC calculation
270//
271//*****************************************************************************
272bool Device_verifyXTAL(float freq)
273{
274 //
275 // Use DCC to verify the XTAL frequency using INTOSC2 as reference clock
276 //
277
278 //
279 // Turn on XTAL and wait for it to power up using X1CNT
280 //
281 SysCtl_turnOnOsc(SYSCTL_OSCSRC_XTAL);
282 SysCtl_clearExternalOscCounterValue();
283 while(SysCtl_getExternalOscCounterValue() != SYSCTL_X1CNT_X1CNT_M);
284
285 //
286 // Enable DCC0 clock
287 //
288 SysCtl_enablePeripheral(SYSCTL_PERIPH_CLK_DCC0);
289
290 //
291 // Insert atleast 5 cycles delay after enabling the peripheral clock
292 //
293 asm(" RPT #5 || NOP");
294
295 //
296 // Configures XTAL as CLKSRC0 and INTOSC2 as CLKSRC1
297 // Fclk0 = XTAL frequency (input parameter)
298 // Fclk1 = INTOSC2 frequency = 10MHz
299 //
300 // Configuring DCC error tolerance of +/-1%
301 // INTOSC2 can have a variance in frequency of +/-10%
302 //
303 // Assuming PLL is not already configured, SysClk freq = 10MHz
304 //
305 // Note : Update the tolerance and INTOSC2 frequency variance as necessary.
306 //
307 return (DCC_verifyClockFrequency(DCC0_BASE,
308 DCC_COUNT1SRC_INTOSC2, 10.0F,
309 DCC_COUNT0SRC_XTAL, freq,
310 1.0F, 10.0F, 10.0F));
311
312}
