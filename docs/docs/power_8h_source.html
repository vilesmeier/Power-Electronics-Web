90static inline void Power_Filter_Set(DCL_DF22 *fil, float32_t isr_freq, float32_t wc)
91{
92 float32_t dt = 1.0f / isr_freq;
93 float32_t A0;
94 A0 = 4.0f + 4.0f * wc * dt * 0.9239f + wc * wc * dt * dt;
95 fil->a1 = (2 * wc * wc * dt * dt - 8.0f) / A0;
96 fil->a2 = (4.0f - 4.0f * wc * dt * 0.9239f + wc * wc * dt * dt) / A0;
97 fil->b0 = (wc * wc * dt * dt) / A0;
98 fil->b1 = 2.0f * fil->b0;
99 fil->b2 = fil->b0;
100
101 A0 = 4.0f + 4.0f * wc * dt * 0.3827f + wc * wc * dt * dt;
102 (fil+1U)->a1 = (2 * wc * wc * dt * dt - 8.0f) / A0;
103 (fil+1U)->a2 = (4.0f - 4.0f * wc * dt * 0.3827f + wc * wc * dt * dt) / A0;
104 (fil+1U)->b0 = (wc * wc * dt * dt) / A0;
105 (fil+1U)->b1 = 2.0f * (fil+1U)->b0;
106 (fil+1U)->b2 = (fil+1U)->b0;
107}

149static inline void Power_1ph_Run(POWER_1PH *power, float32_t v, float32_t i)
150{
151 // rms 滤波器, 更新结果
152 power->v_rms = Power_Filter_Run(power->v_fil, v*v);
153 power->i_rms = Power_Filter_Run(power->i_fil, i*i);
154 power->v_rms = sqrtf(power->v_rms);
155 power->i_rms = sqrtf(power->i_rms);
156
157 // 有功 无功滤波器, 更新结果
158 power->p = Power_Filter_Run(power->p_fil, v*i);
159
160 // 其他变量计算
161 power->s = power->v_rms * power->i_rms;
162 power->q = sqrtf(power->s*power->s-power->p*power->p);
163 power->pf = power->p / power->s;
164}

195static inline void Power_3ph_Run(POWER_3PH *power, float32_t va, float32_t vb, float32_t vc, float32_t ia, float32_t ib, float32_t ic)
196{
197 // rms 滤波器, 更新结果
198 power->va_rms = Power_Filter_Run(power->va_fil, va*va);
199 power->vb_rms = Power_Filter_Run(power->vb_fil, vb*vb);
200 power->vc_rms = Power_Filter_Run(power->vc_fil, vc*vc);
201
202 power->ia_rms = Power_Filter_Run(power->ia_fil, ia*ia);
203 power->ib_rms = Power_Filter_Run(power->ib_fil, ib*ib);
204 power->ic_rms = Power_Filter_Run(power->ic_fil, ic*ic);
205
206 power->va_rms = sqrtf(power->va_rms);
207 power->vb_rms = sqrtf(power->vb_rms);
208 power->vc_rms = sqrtf(power->vc_rms);
209
210 power->ia_rms = sqrtf(power->ia_rms);
211 power->ib_rms = sqrtf(power->ib_rms);
212 power->ic_rms = sqrtf(power->ic_rms);
213
214 // 有功 无功 视在 功率因数
215 power->p = Power_Filter_Run(power->p_fil, va*ia+ vb*ib + vc*ic);
216 power->s = power->va_rms*power->ia_rms+power->vb_rms*power->ib_rms+power->vc_rms*power->ic_rms;
217 power->q = sqrtf(power->s*power->s-power->p*power->p);
218 power->pf = power->p / power->s;
219}
