62//
63// InitEQepGpio - This function initializes GPIO pins to function as eQEP pins
64//
65// Each GPIO pin can be configured as a GPIO pin or up to 3 different
66// peripheral functional pins. By default all pins come up as GPIO
67// inputs after reset.
68//
69// Caution:
70// For each eQEP peripheral
71// Only one GPIO pin should be enabled for EQEPxA operation.
72// Only one GPIO pin should be enabled for EQEPxB operation.
73// Only one GPIO pin should be enabled for EQEPxS operation.
74// Only one GPIO pin should be enabled for EQEPxI operation.
75// Comment out other unwanted lines.
76//
77void
78InitEQepGpio()
79{
80#if DSP28_EQEP1
81 InitEQep1Gpio();
82#endif // endif DSP28_EQEP1
83#if DSP28_EQEP2
84 InitEQep2Gpio();
85#endif // endif DSP28_EQEP2
86}
