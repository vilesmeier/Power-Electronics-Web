147static inline void SPLL_1PH_SOGI_coeff_calc(SPLL_1PH_SOGI *spll_obj)
148{
149 float32_t osgx,osgy,temp, wn;
150 wn= spll_obj->fn *(float32_t) 2.0f * (float32_t) 3.14159265f;
151 spll_obj->osg_coeff.osg_k=(float32_t)(0.5);
152 osgx = (float32_t)(2.0f*0.5f*wn*spll_obj->delta_t);
153 spll_obj->osg_coeff.osg_x=(float32_t)(osgx);
154 osgy = (float32_t)(wn*spll_obj->delta_t*wn*spll_obj->delta_t);
155 spll_obj->osg_coeff.osg_y=(float32_t)(osgy);
156 temp = (float32_t)1.0/(osgx+osgy+4.0f);
157 spll_obj->osg_coeff.osg_b0=((float32_t)osgx*temp);
158 spll_obj->osg_coeff.osg_b2=((float32_t)(-1.0f)*spll_obj->osg_coeff.osg_b0);
159 spll_obj->osg_coeff.osg_a1=((float32_t)(2.0*(4.0f-osgy))*temp);
160 spll_obj->osg_coeff.osg_a2=((float32_t)(osgx-osgy-4)*temp);
161 spll_obj->osg_coeff.osg_qb0=((float32_t)(0.5f*osgy)*temp);
162 spll_obj->osg_coeff.osg_qb1=(spll_obj->osg_coeff.osg_qb0*(float32_t)(2.0));
163 spll_obj->osg_coeff.osg_qb2=spll_obj->osg_coeff.osg_qb0;
164}

194static inline void SPLL_1PH_SOGI_run(SPLL_1PH_SOGI *spll_obj,
195 float32_t acValue)
196{
197 // Update the spll_obj->u[0] with the grid value
198 spll_obj->u[0]=acValue;
199
200 //
201 // Orthogonal Signal Generator
202 //
203 spll_obj->osg_u[0]=(spll_obj->osg_coeff.osg_b0*
204 (spll_obj->u[0]-spll_obj->u[2])) +
205 (spll_obj->osg_coeff.osg_a1*spll_obj->osg_u[1]) +
206 (spll_obj->osg_coeff.osg_a2*spll_obj->osg_u[2]);
207
208 spll_obj->osg_u[2]=spll_obj->osg_u[1];
209 spll_obj->osg_u[1]=spll_obj->osg_u[0];
210
211 spll_obj->osg_qu[0]=(spll_obj->osg_coeff.osg_qb0*spll_obj->u[0]) +
212 (spll_obj->osg_coeff.osg_qb1*spll_obj->u[1]) +
213 (spll_obj->osg_coeff.osg_qb2*spll_obj->u[2]) +
214 (spll_obj->osg_coeff.osg_a1*spll_obj->osg_qu[1]) +
215 (spll_obj->osg_coeff.osg_a2*spll_obj->osg_qu[2]);
216
217 spll_obj->osg_qu[2]=spll_obj->osg_qu[1];
218 spll_obj->osg_qu[1]=spll_obj->osg_qu[0];
219
220 spll_obj->u[2]=spll_obj->u[1];
221 spll_obj->u[1]=spll_obj->u[0];
222
223 //
224 // Park Transform from alpha beta to d-q axis
225 //
226 spll_obj->u_Q[0]=(spll_obj->cosine*spll_obj->osg_u[0]) +
227 (spll_obj->sine*spll_obj->osg_qu[0]);
228 spll_obj->u_D[0]=(spll_obj->cosine*spll_obj->osg_qu[0]) -
229 (spll_obj->sine*spll_obj->osg_u[0]);
230
231 //
232 // Loop Filter
233 //
234 spll_obj->ylf[0]=spll_obj->ylf[1] +
235 (spll_obj->lpf_coeff.b0*spll_obj->u_Q[0]) +
236 (spll_obj->lpf_coeff.b1*spll_obj->u_Q[1]);
237 spll_obj->ylf[1]=spll_obj->ylf[0];
238
239 spll_obj->u_Q[1]=spll_obj->u_Q[0];
240
241 //
242 // VCO
243 //
244 spll_obj->fo=spll_obj->fn+spll_obj->ylf[0];
245
246 spll_obj->theta=spll_obj->theta + (spll_obj->fo*spll_obj->delta_t)*
247 (float32_t)(2.0*3.1415926f);
248
249 if(spll_obj->theta>(float32_t)(2.0*3.1415926f))
250 {
251 spll_obj->theta=spll_obj->theta - (float32_t)(2.0*3.1415926f);
252 }
253
254 spll_obj->sine=(float32_t)sinf(spll_obj->theta);
255 spll_obj->cosine=(float32_t)cosf(spll_obj->theta);
256}
