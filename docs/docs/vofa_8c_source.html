56void VOFA_Set_Channel_Input(uint16_t channel, float32_t *input_ptr)
57{
58 // 小于最大值且 channel 尚未越界时,可以继续关联通道数量
59 if ((message_array.used_channel_max < VOFA_CHANNEL_NUM_MAX) && (channel < VOFA_CHANNEL_NUM_MAX))
60 {
61 // 该通道尚未关联
62 if (message_array.input_pointer[channel] == NULL)
63 {
64 // 关联指针
65 message_array.input_pointer[channel] = input_ptr;
66
67 // 记录使用通道数
68 if (channel >= message_array.used_channel_max)
69 {
70 message_array.used_channel_max = channel + 1U;
71 }
72 }
73 }
74 // used_channel_max >= VOFA_CHANNEL_NUM_MAX 下, 代表指针数组无空余位置, 不再接受新的通道关联
75}

77void VOFA_Run(void)
78{
79 // 在使能本次发送后, 发送数据包
80 if (message_array.flag_send == VOFA_Flag_Send_Enable)
81 {
82 // 设置标志位
83 message_array.state = VOFA_Flag_Sending;
84
85 uint16_t i = 0;
86
87 // 遍历每个通道
88 for (i = 0; i < message_array.used_channel_max; i++)
89 {
90 // 该通道关联变量
91 if (message_array.input_pointer[i] != NULL)
92 {
93 // 获取变量值, 数据类型转换
94 message_array.data_transfer.float_num = *(message_array.input_pointer[i]);
95
96 // 形成 SCI 发送的数据包
97 message_array.package_data[0] = message_array.data_transfer.dat[0];
98 message_array.package_data[1] = message_array.data_transfer.dat[0] >> 8U;
99 message_array.package_data[2] = message_array.data_transfer.dat[1];
100 message_array.package_data[3] = message_array.data_transfer.dat[1] >> 8U;
101 }
102
103 // 该通道未关联变量
104 else
105 {
106 // 发送空数据包, 防止波形错位
107 message_array.package_data[0] = 0x00;
108 message_array.package_data[1] = 0x00;
109 message_array.package_data[2] = 0x00;
110 message_array.package_data[3] = 0x00;
111 }
112
113 // SCI 发送数据包
114 SCI_writeCharArray(VOFA_UART_BASE, message_array.package_data, VOFA_PACK_LENGTH);
115 }
116
117 // 完成遍历后, 发送数据包尾
118 SCI_writeCharArray(VOFA_UART_BASE, message_array.package_tail, VOFA_PACK_LENGTH);
119
120 // 发送后清空标志位
121 message_array.flag_send = VOFA_Flag_Send_Disable;
122 message_array.state = VOFA_Flag_Idle;
123 }
124}
