2;;! \file source/fft/RFFT_adc_f32u.asm
3;;!
4;;! \brief Real FFT with 12-bit ADC input.
5;;! \author c2000
6;;
7;; DESCRIPTION:
8;;
9;; This function calcuates a real FFT with 12-bit fixed-point ADC input.
10;; The input buffer does not have to be aligned. If alignment can be
11;; done, then use the RFFT_adc_f32 function to improve performance.
12;;
13;; FUNCTIONS:
14;;
15;; void RFFT_adc_f32u (RFFT_ADC_F32_STRUCT_Handle)
16;;
17;; where RFFT_ADC_F32_STRUCT_Handle is defined as:
18;; typedef RFFT_ADC_F32_STRUCT* RFFT_ADC_F32_STRUCT_Handle;
19;;
20;; and where RFFT_ADC_F32_STRUCT is a structure defined as:
21;;
22;; typedef struct {
23;; uint16_t *InBuf;
24;; void *Tail;
25;; } RFFT_ADC_F32_STRUCT;
26;;
27;; ASSUMPTIONS:
28;;
29;; * FFTSize must be a power of 2 (32, 64, 128, etc)
30;; * FFTSize must be greater or equal to 32
31;; * FFTStages must be log2(FFTSize)
32;; * InBuf, OutBuf, CosSinBuf are FFTSize in length
33;; * MagBuf and PhaseBuf are FFTSize/2 in length
34;; * MagBuf and PhaseBuf are not used by this function.
35;; They are only used by the magitude and phase calculation functions.
36;;
37;; ALGORITHM:
38;;
39;; 1) Bit reverse input data and calculate stages 1, 2 & 3:
40;;
41;; In Buf (read in bit reverse order) Out Buf
42;; +----+ +----+
43;; | I1 | (((I1 + I2) + (I3 + I4)) + ((I5 + I6) + (I7 + I8)))/8 -> | I1'|
44;; | I2 | ((I1 - I2) + COS*((I5 - I6) + (I8 - I7)) )/8 -> | I2'|
45;; | I3 | ((I1 + I2) - (I3 + I4) )/8 -> | I3'|
46;; | I4 | ((I1 - I2) - COS*((I5 - I6) + (I8 - I7)) )/8 -> | I4'|
47;; | I5 | (((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)))/8 -> | I5'|
48;; | I6 | (COS*((I8 - I7) - (I5 - I6)) - (I4 - I3) )/8 -> | I6'|
49;; | I7 | ((I7 + I8) - (I5 + I6) )/8 -> | I7'|
50;; | I8 | (COS*((I8 - I7) - (I5 - I6)) + (I4 - I3) )/8 -> | I8'|
51;; .
52;; .
53;; \|/
54;; Repeat above FFTSize/8 (i.e. if FFTSize = 1024, Repeat = 128 times)
55;;
56;; Note: COS = COS( 1*2*PI/8) = SIN( 1*2*PI/8)
57;;
58;; 2) Calculate stages 4 and up:
59;;
60;;
61;; Out Buf 4 5 6 7 8 9 10 <- Stages
62;; +- +---------+ - - - - - - --
63;; | | Y1 | 0 0 0 0 0 0 0 <- Y1 + Y3
64;; | |---------|
65;; | | X(I1) | 1 1 1 1 1 1 1 <- X(I1) + [X(I3)*COS + X(I4)*SIN]
66;; | |---------|
67;; | | . |
68;; | | . | 3 7 15 31 63 127 255 <- Inner Loop Repeat Times
69;; | | \./ |
70;; | I |---------|
71;; | N | Y2 | 4 8 16 32 64 128 256 <- Y2
72;; | N |---------|
73;; | E | /.\ |
74;; | R | . |
75;; | | . |
76;; | L |---------|
77;; | O | X(I2) | 7 15 31 63 127 255 511 <- X(I1) - [X(I3)*COS + X(I4)*SIN]
78;; | O |---------|
79;; | P | Y3 | 8 16 32 64 128 256 512 <- Y1 - Y3
80;; | |---------|
81;; | | X(I3) | 9 17 33 65 129 257 513 <- [X(I4)*COS - X(I3)*SIN] - X(I2)
82;; | |---------|
83;; | | . |
84;; | | . |
85;; | | \./ |
86;; | |---------|
87;; | | Y4 |12 24 48 96 192 384 768 <- -Y4
88;; | |---------|
89;; | | /.\ |
90;; | | . |
91;; | | . |
92;; | |---------|
93;; | | X(I4) |15 31 63 127 255 511 1023 <- [X(I4)*COS - X(I3)*SIN] + X(I2)
94;; +- |---------|
95;; | |16 32 64 128 256 512 1024
96;; -- --- --- --- --- --- ----
97;; . 1 <- Outer Loop Repeat Times (16 FFT)
98;; . 2 1 <- Outer Loop Repeat Times (32 FFT)
99;; . 4 2 1 <- Outer Loop Repeat Times (64 FFT)
100;; . 8 4 2 1 <- Outer Loop Repeat Times (128 FFT)
101;; . 16 8 4 2 1 <- Outer Loop Repeat Times (256 FFT)
102;; . 32 16 8 4 2 1 <- Outer Loop Repeat Times (512 FFT)
103;; . 64 32 16 8 4 2 1 <- Outer Loop Repeat Times (1024 FFT)
104;; .
105;; .
106;; \|/

175; =============================================================================
176; Function: void rfft_adc_f32u_Stages1and2and3andBitReverse(
177; RFFT_ADC_F32_STRUCT_Handle)
178; =============================================================================
179;---------------------------------------------------------------------------
180;
181; DESCRIPTION:
182;
183; This function bit reverses the input and computes stages 1, 2 and 3
184; This function does not have any alignment requirements for the input
185; buffer.
186;
187; ON ENTRY:
188;
189; XAR4 = Starting address of the RFFT_F32_STRUCT structure
190;
191; REGISTER USAGE:
192;
193; AR0, XAR1, XAR2, XAR4, XAR5, AR3, AR6, XAR7, ACC,
194; R0H, R1H, R2H, R3H, R4H, R5H, R6H, R7H
195;
196; On Exit:
197;
198; RFFT_F32_STRUCT OutBuf contains the computed result
199;
200;---------------------------------------------------------------------------
201;offsets to the stack which holds the bit reversed elements
202;---------------------------------------------------------------------------
203
204I1 .set 06H
205I2 .set 08H
206I3 .set 0AH
207I4 .set 0CH
208I5 .set 0EH
209I6 .set 10H
210I7 .set 12H
211I8 .set 14H
212cossinBuf .set 04H
213;------------------------------------------------------------------------------
214;Offset to the stack which holds intermediate results of FFT computation
215;------------------------------------------------------------------------------
216 .text

218_rfft_adc_f32u_Stages1and2and3andBitReverse:
219;----------------------------------------------------------------------
220; Save all save-on-entry registers used
221;----------------------------------------------------------------------
222 PUSH XAR1
223 PUSH XAR2
224 PUSH XAR3
225 MOV32 *SP++,R4H
226 MOV32 *SP++,R5H
227 MOV32 *SP++,R6H
228 MOV32 *SP++,R7H
229 ADDB SP,#14h
230
231 MOVL XAR2,*+XAR4[0] ; &Inbuf
232 MOVL XAR5,*+XAR4[2] ; &Outbuf
233 SUBB XAR5, #2
234 MOVL XAR4, XAR5
235 MOVL XAR5,*+XAR4[2]
236 MOVL XAR3,*+XAR4[2]
237 ADDB XAR3,#8 ; &Outbuf[4]
238 MOVL XAR7,*+XAR4[4] ; &CosSinbuf
239
240 MOV AR0,#0Ah
241 MOV AH,*+XAR4[AR0] ; FFT SIZE
242 LSR AH,1 ; FFT SIZE/2 - for 16-bit input data
243 MOV AR0,AH
244; LSR AH,3
245 LSR AH,2 ; for 16-bit input data
246 SUBB AH,#1 ; (Size / 8) - 1 3f
247 MOVL XAR1,#0000h ; index if memory is not aligned
248
249 RPTB _rfft_32_Last, AH
250;------------------------------------------------------------------------------
251; Code assumes no alignment
252; Assumes ADC integer data is the input
253; Convert and scale while bit reversing
254;------------------------------------------------------------------------------
255 MOVI32 R2H, 0x39800801
256 NOP *,ARP1
257 UI16TOF32 R0H, *+XAR2[AR1] ;I1 load
258 NOP *,ARP1
259 NOP *BR0++
260 MPYF32 R0H, R0H, R2H ;I1 scale
261 UI16TOF32 R1H, *+XAR2[AR1] ;I2 load
262 NOP *,ARP1
263 MPYF32 R1H, R1H, R2H ;I2 scale
264 || MOV32 *-SP[I1], R0H ;I1 save
265 NOP *BR0++
266 UI16TOF32 R0H, *+XAR2[AR1] ;I3 load
267 NOP *,ARP1
268 MPYF32 R0H, R0H, R2H ;I3 scale
269 || MOV32 *-SP[I2], R1H ;I2 save
270 NOP *BR0++
271 UI16TOF32 R1H, *+XAR2[AR1] ;I4 load
272 NOP *,ARP1
273 MPYF32 R1H, R1H, R2H ;I4 scale
274 || MOV32 *-SP[I3], R0H ;I5 save
275 NOP *BR0++
276 UI16TOF32 R0H, *+XAR2[AR1] ;I5 load
277 NOP *,ARP1
278 MPYF32 R0H, R0H, R2H ;I5 scale
279 || MOV32 *-SP[I4], R1H ;I4 save
280 NOP *BR0++
281 UI16TOF32 R1H, *+XAR2[AR1] ;I6 load
282 NOP *,ARP1
283 MPYF32 R1H, R1H, R2H ;I6 scale
284 || MOV32 *-SP[I5], R0H ;I5 save
285 NOP *BR0++
286 UI16TOF32 R0H, *+XAR2[AR1] ;I7 load
287 NOP *,ARP1
288 MPYF32 R0H, R0H, R2H ;I7 scale
289 || MOV32 *-SP[I6], R1H ;I6 store
290 NOP *BR0++
291 UI16TOF32 R1H, *+XAR2[AR1] ;I8 load
292 NOP *,ARP1
293 MPYF32 R1H, R1H, R2H ;I8 scale
294 || MOV32 *-SP[I7], R0H ;I7 store
295 NOP *BR0++
296 MOV32 *-SP[I8], R1H ;I8 store
297
298;-------------------------------------------------------------------------------
299; Computations for stages 1 2, and 3
300; OutBufIndex++ = (I1 + I2) + (I3 + I4) + (I5 + I6) + (I7 + I8); (A) <- XAR5
301; OutBufIndex++ = (I1 - I2) + COS x ((I5 - I6) + (I8 - I7)); (B)
302; OutBufIndex++ = (I1 + I2) - (I3 + I4); (C)
303; OutBufIndex++ = (I1 - I2) - COS x ((I5 - I6) + (I8 - I7)); (D)
304; OutBufIndex++ = ((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)); (E) <- XAR3
305; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) - (I4 - I3); (F)
306; OutBufIndex++ = (I7 + I8) - (I5 + I6); (G)
307; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) + (I4 - I3); (H)
308;-------------------------------------------------------------------------------
309 MOV32 R0H, *-SP[I8] ; R0H = I8
310 MOV32 R1H, *-SP[I7] ; R1H = I7
311 SUBF32 R2H, R0H, R1H ; R2H = I8-I7
312 || MOV32 R3H, *-SP[I5] ; R3H = I5
313 ADDF32 R0H, R1H, R0H ; R0H = I7+I8
314 || MOV32 R4H, *-SP[I6] ; R4H = I6
315 SUBF32 R1H, R3H, R4H ; R1H = I5-I6
316 || MOV32 R5H, *-SP[I1] ; R5H = I1
317 ADDF32 R3H, R3H, R4H ; R3H = I5+I6
318 ADDF32 R4H, R1H, R2H ; R4H = (I5-I6) + (I8-I7)
319 || MOV32 R6H,*XAR7 ; R6H = COS
320 SUBF32 R7H, R0H, R3H ; R7H = (I7+I8) - (I5+I6) (G)
321 MPYF32 R4H, R6H, R4H ; R4H = COS x ((I5-I6) + (I8-I7))
322 || SUBF32 R2H, R2H, R1H ; R2H = (I8-I7) - (I5-I6)
323 ADDF32 R0H, R3H, R0H ; R0H = (I5+I6) + (I7+I8)
324 || MOV32 *+XAR3[4], R7H ; store G
325 MPYF32 R2H, R6H, R2H ; R2H = COS x ((I8-I7) - (I5-I6))
326 || MOV32 R7H, *-SP[I2] ; R7H = I2
327 SUBF32 R6H, R5H, R7H ; R6H = I1-I2
328 ADDF32 R5H, R5H, R7H ; R5H = I1+I2
329 || MOV32 R1H, *-SP[I4] ; R1H = I4
330 ADDF32 R3H, R6H, R4H ; R3H = (I1-I2) + COS x ((I5-I6) + (I8-I7)) (B)
331 || MOV32 R7H, *-SP[I3] ; R7H = I3
332 SUBF32 R6H, R6H, R4H ; R6H = (I1-I2) - COS x ((I5-I6) + (I8-I7)) (D)
333 SUBF32 R3H, R1H, R7H ; R3H = I4-I3
334 || MOV32 *+XAR5[2], R3H ; store B
335 ADDF32 R4H, R7H, R1H ; R4H = I3+I4
336 || MOV32 *+XAR5[6], R6H ; store D
337 SUBF32 R1H, R2H, R3H ; R1H = COS x ((I8-I7) - (I5-I6)) - (I4-I3) (F)
338 ADDF32 R2H, R2H, R3H ; R2H = COS x ((I8-I7) - (I5-I6)) + (I4-I3) (H)
339 ADDF32 R6H, R5H, R4H ; R6H = (I1+I2) + (I3+I4)
340 || MOV32 *+XAR3[2], R1H ; save F
341 SUBF32 R1H, R5H, R4H ; R1H = (I1+I2) - (I3+I4) (C)
342 || MOV32 *+XAR3[6], R2H ; save H
343 SUBF32 R3H, R6H, R0H ; R3H = (I1+I2) + (I3+I4) - ((I5+I6) + (I8+I7)) (E)
344 ADDF32 R0H, R6H, R0H ; R0H = (I1+I2) + (I3+I4) + (I5+I6) + (I8+I7) (A)
345 || MOV32 *+XAR5[4], R1H ; store C
346 MOV32 *+XAR3[0], R3H ; store E
347 MOV32 *+XAR5[0], R0H ; store A
348 ADDB XAR5, #16
349 ADDB XAR3, #16

354 MOV32 R7H,*--SP
355 MOV32 R6H,*--SP
356 MOV32 R5H,*--SP
357 MOV32 R4H,*--SP
358 POP XAR3
359 POP XAR2
360 POP XAR1
361 LRETR
