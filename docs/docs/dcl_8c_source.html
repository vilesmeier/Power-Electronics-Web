57void DCL_PR_Set(DCL_PR *p, float32_t Kp, float32_t Kr, float32_t wc, float32_t w0, float32_t isr_freq)
58{
59 p->Kp = Kp;
60 p->Kr = Kr;
61
62 float32_t T_isr = (1.0f) / isr_freq;
63 float32_t a0 = 4.0f + 4.0f * T_isr * wc + w0 * w0 * T_isr * T_isr;
64 p->pr.a1 = (2.0f * w0 * w0 * T_isr * T_isr - 8.0f) / a0;
65 p->pr.a2 = (4.0f - 4.0f * T_isr * wc + w0 * w0 * T_isr * T_isr) / a0;
66 p->pr.b0 = (4.0f * Kp + 4.0f * T_isr * (Kp + Kr) * wc + Kp * w0 * w0 * T_isr * T_isr) / a0;
67 p->pr.b1 = (Kp * (2.0f * w0 * w0 * T_isr * T_isr - 8.0f)) / a0;
68 p->pr.b2 = (4.0f * Kp - 4.0f * T_isr * (Kp + Kr) *wc + Kp * w0 *w0 * T_isr * T_isr) / a0;
