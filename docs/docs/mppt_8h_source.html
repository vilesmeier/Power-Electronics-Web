126static inline void MPPT_INCC_Run(MPPT_INCC *v, float32_t Vpv, float32_t Ipv)
127{
128 v->Vpv = Vpv;
129 v->Ipv = Ipv;
130 if (v->mppt_enable==1)
131 { /*MPPT mode enable*/
132 if (v->mppt_first == 1)
133 {
134 v->VmppOut= v->Vpv - v->StepFirst;
135 v->VpvOld=v->Vpv;
136 v->IpvOld=v->Ipv;
137 v->mppt_first=0;
138 }
139 else
140 {
141 v->DeltaV = v->Vpv-v->VpvOld ; /* PV voltage change */
142 v->DeltaI = v->Ipv-v->IpvOld ; /* PV current change */
143 if(v->DeltaV == 0) /* DeltaV=0 */
144 {
145 if(v->DeltaI !=0 ) /* dI=0 operating at MPP */
146 {
147 if ( v->DeltaI > 0 ) /* climbing towards MPP */
148 {
149 v->VmppOut=v->Vpv+v->Stepsize; /* Increment Vref */
150 }
151 else
152 {
153 v->VmppOut=v->Vpv-v->Stepsize; /* Decrement Vref */
154 }
155 }
156 }
157 else
158 {
159 v->Cond = v->Ipv*__einvf32(v->Vpv); /*Cond =-(v->Ipv<<10)/v->Vpv;*/
160 v->IncCond = v->DeltaI*__einvf32(v->DeltaV); /*IncCond =(DeltaI<<10)/DeltaV;*/
161 if (v->IncCond != v->Cond) /* Operating at MPP */
162 {
163 if (v->IncCond > (-v->Cond)) /* left of MPP */
164 {
165 v->VmppOut=v->Vpv+v->Stepsize; /* change Vref */
166 }
167 else /* Right of MPP */
168 {
169 v->VmppOut=v->Vpv-v->Stepsize; /* change Vref */
170 }
171 }
172 }
173 if(v->VmppOut < v->MinVolt)
174 {
175 v->VmppOut = v->MinVolt;
176 }
177 else if(v->VmppOut > v->MaxVolt)
178 {
179 v->VmppOut = v->MaxVolt;
180 }
181 else
182 {
183 v->VmppOut = v->VmppOut;
184 }
185 v->VpvOld = v->Vpv;
186 v->IpvOld = v->Ipv;
187 }
188 }
189}

300static inline void MPPT_PNO_Run(MPPT_PNO *v, float32_t Vpv, float32_t Ipv)
301{
302 v->Vpv = Vpv;
303 v->Ipv = Ipv;
304 if (v->mppt_enable==1)
305 {
306 if (v->mppt_first == 1)
307 {
308 v->VmppOut= v->Vpv - (0.02);
309 v->mppt_first=0;
310 v->PanelPower_Prev=v->PanelPower;
311 }
312 else
313 {
314 v->PanelPower=(v->Vpv*v->Ipv);
315 v->DeltaP=v->PanelPower-v->PanelPower_Prev;
316 if (v->DeltaP > v->DeltaPmin)
317 {
318 v->VmppOut=v->Vpv+v->Stepsize;
319 }
320 else
321 {
322 if (v->DeltaP < -v->DeltaPmin)
323 {
324 v->Stepsize=-v->Stepsize;
325 v->VmppOut=v->Vpv+v->Stepsize;
326 }
327 }
328 v->PanelPower_Prev = v->PanelPower;
329 }
330 if(v->VmppOut < v->MinVolt)
331 {
332 v->VmppOut = v->MinVolt;
333 }
334 else if(v->VmppOut > v->MaxVolt)
335 {
336 v->VmppOut = v->MaxVolt;
337 }
338 else
339 {
340 v->VmppOut = v->VmppOut;
341 }
342 }
