213//
214// System Control Register File
215//
216struct SYS_CTRL_REGS {
217 Uint16 rsvd1; // 0
218 union PLLSTS_REG PLLSTS; // 1
219 Uint16 rsvd2[8]; // 2-9
220
221 //
222 // 10: High-speed peripheral clock pre-scaler
223 //
224 union HISPCP_REG HISPCP;
225
226 union LOSPCP_REG LOSPCP; // 11: Low-speed peripheral clock pre-scaler
227 union PCLKCR0_REG PCLKCR0; // 12: Peripheral clock control register
228 union PCLKCR1_REG PCLKCR1; // 13: Peripheral clock control register
229 union LPMCR0_REG LPMCR0; // 14: Low-power mode control register 0
230 Uint16 rsvd3; // 15: reserved
231 union PCLKCR3_REG PCLKCR3; // 16: Peripheral clock control register
232 union PLLCR_REG PLLCR; // 17: PLL control register
233
234 //
235 // No bit definitions are defined for SCSR because
236 // a read-modify-write instruction can clear the WDOVERRIDE bit
237 //
238 Uint16 SCSR; // 18: System control and status register
239
240 Uint16 WDCNTR; // 19: WD counter register
241 Uint16 rsvd4; // 20
242 Uint16 WDKEY; // 21: WD reset key register
243 Uint16 rsvd5[3]; // 22-24
244
245 //
246 // No bit definitions are defined for WDCR because
247 // the proper value must be written to the WDCHK field
248 // whenever writing to this register.
249 //
250 Uint16 WDCR; // 25: WD timer control register
251
252 Uint16 rsvd6[4]; // 26-29
253 union MAPCNF_REG MAPCNF; // 30: Dual-mapping configuration register
254 Uint16 rsvd7[1]; // 31
255};
