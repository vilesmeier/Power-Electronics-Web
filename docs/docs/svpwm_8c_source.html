78void SVPWM_Run(SVPWM* svpwm, float32_t Ualpha, float32_t Ubeta)
79{
80 //get parameter and sector
81 SVPWM_Sector_Cal(svpwm, Ualpha, Ubeta);
82
83 //SVPWM modulate
84 switch(svpwm->Sector)
85 {
86 //Sector Tx Ty
87 case 1: //Sector 2 110 010
88 svpwm->t6 = -svpwm->u2 * svpwm->k;
89 svpwm->t2 = -svpwm->u3 * svpwm->k;
90 if((svpwm->t6 + svpwm->t2) > svpwm->Ts)
91 {
92 svpwm->Mod_Rate = svpwm->Ts/(svpwm->t2 + svpwm->t6);
93 svpwm->t2 = svpwm->Mod_Rate * svpwm->t2;
94 svpwm->t6 = svpwm->Mod_Rate * svpwm->t6;
95 }
96 svpwm->duty2 = 0.25f * (svpwm->Ts - svpwm->t2 - svpwm->t6);
97 svpwm->duty1 = svpwm->duty2 + 0.5f * svpwm->t6;
98 svpwm->duty3 = svpwm->duty1 + 0.5f * svpwm->t2;
99 break;
100
101 case 2: //Sector 6 101 100
102 svpwm->t5 = -svpwm->u3 * svpwm->k;
103 svpwm->t4 = -svpwm->u1 * svpwm->k;
104 if((svpwm->t5 + svpwm-> t4) > svpwm->Ts)
105 {
106 svpwm->Mod_Rate = svpwm->Ts/(svpwm->t5 + svpwm-> t4);
107 svpwm->t5 = svpwm->Mod_Rate * svpwm->t5;
108 svpwm->t4 = svpwm->Mod_Rate * svpwm->t4;
109 }
110 svpwm->duty1 = 0.25f * (svpwm->Ts - svpwm->t4 - svpwm->t5);
111 svpwm->duty3 = svpwm->duty1 + 0.5f * svpwm->t5;
112 svpwm->duty2 = svpwm->duty3 + 0.5f * svpwm->t4;
113 break;
114
115 case 3: //Sector 1 100 110
116 svpwm->t4 = svpwm->u2 * svpwm->k;
117 svpwm->t6 = svpwm->u1 * svpwm->k;
118 if((svpwm->t4 + svpwm->t6) > svpwm->Ts)
119 {
120 svpwm->Mod_Rate = svpwm->Ts/(svpwm->t4 + svpwm->t6);
121 svpwm->t4 = svpwm->Mod_Rate * svpwm->t4;
122 svpwm->t6 = svpwm->Mod_Rate * svpwm->t6;
123 }
124 svpwm->duty1 = 0.25f *(svpwm->Ts - svpwm->t4 - svpwm->t6);
125 svpwm->duty2 = svpwm->duty1 + 0.5f * svpwm->t4;
126 svpwm->duty3 = svpwm->duty2 + 0.5f * svpwm->t6;
127 break;
128
129 case 4: //sector 4 011 001
130 svpwm->t3 = -svpwm->u1 * svpwm->k;
131 svpwm->t1 = -svpwm->u2 * svpwm->k;
132 if((svpwm->t3 + svpwm->t1) > svpwm->Ts)
133 {
134 svpwm->Mod_Rate = svpwm->Ts/(svpwm->t1 + svpwm->t3);
135 svpwm->t1 = svpwm->Mod_Rate * svpwm->t1;
136 svpwm->t3 = svpwm->Mod_Rate * svpwm->t3;
137 }
138 svpwm->duty3 = 0.25f * (svpwm->Ts - svpwm->t1 - svpwm->t3);
139 svpwm->duty2 = svpwm->duty3 + 0.5f * svpwm->t3;
140 svpwm->duty1 = svpwm->duty2 + 0.5f * svpwm->t1;
141 break;
142
143 case 5: //Sector 3 010 011
144 svpwm->t2 = svpwm->u1 * svpwm->k;
145 svpwm->t3 = svpwm->u3 * svpwm->k;
146 if((svpwm->t2 + svpwm->t3) > svpwm->Ts)
147 {
148 svpwm->Mod_Rate = svpwm->Ts/(svpwm->t2 + svpwm->t3);
149 svpwm->t2 = svpwm->Mod_Rate * svpwm->t2;
150 svpwm->t3 = svpwm->Mod_Rate * svpwm->t3;
151 }
152 svpwm->duty2 = 0.25f * (svpwm->Ts - svpwm->t2 - svpwm->t3);
153 svpwm->duty3 = svpwm->duty2 + 0.5f * svpwm->t2;
154 svpwm->duty1 = svpwm->duty3 + 0.5f * svpwm->t3;
155 break;
156
157 case 6: //Sector 5 001 101
158 svpwm->t1 = svpwm->u3 * svpwm->k;
159 svpwm->t5 = svpwm->u2 * svpwm->k;
160 if((svpwm->t5 + svpwm->t1) > svpwm->Ts)
161 {
162 svpwm->Mod_Rate = svpwm->Ts/(svpwm->t1 + svpwm->t5);
163 svpwm->t1 = svpwm->Mod_Rate * svpwm->t1;
164 svpwm->t5 = svpwm->Mod_Rate * svpwm->t5;
165 }
166 svpwm->duty3 = 0.25f * (svpwm->Ts - svpwm->t1 - svpwm->t5);
167 svpwm->duty1 = svpwm->duty3 + 0.5f * svpwm->t1;
168 svpwm->duty2 = svpwm->duty1 + 0.5f * svpwm->t5;
169 break;
170
171 default:
172 svpwm->duty1 = 0.5f;
173 svpwm->duty2 = 0.5f;
174 svpwm->duty3 = 0.5f;
175 break;
176 }
177
178 //update comp value for epwm1/epwm2/epwm3
179 SVPWM_Duty_Update(svpwm);
180}
