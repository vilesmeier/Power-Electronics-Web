2;;! \file source/fft/RFFT_adc_f32.asm
3;;!
4;;! \brief Real FFT with 12-bit ADC input
5;;! \author C2000
6;;
7;; DESCRIPTION:
8;;
9;; This function computes a real FFT with 12-bit fixed-point ADC input.
10;; The input buffer must be aligned to a multiple of the FFT size. If
11;; it is not aligned then the output buffer will yield invalid results.
12;; If you do not wish to align the input buffer then use the RFFT_adc_f32u
13;; function. Using this function will reduce cycle performance of the
14;; algorithm.
15;;
16;; FUNCTIONS:
17;;
18;; void RFFT_adc_f32 (RFFT_ADC_F32_STRUCT_Handle)
19;;
20;; where RFFT_ADC_F32_STRUCT_Handle is defined as:
21;; typedef RFFT_F32_STRUCT* RFFT_ADC_F32_STRUCT_Handle;
22;;
23;; and where RFFT_ADC_F32_STRUCT is a structure defined as:
24;;
25;; typedef struct {
26;; uint16_t *InBuf;
27;; void *Tail;
28;; } RFFT_ADC_F32_STRUCT;
29;;
30;; ASSUMPTIONS:
31;;
32;; * FFTSize must be a power of 2 (32, 64, 128, etc)
33;; * FFTSize must be greater or equal to 32
34;; * FFTStages must be log2(FFTSize)
35;; * InBuf, OutBuf, CosSinBuf are FFTSize in length
36;; * MagBuf and PhaseBuf are FFTSize/2 in length
37;; * MagBuf and PhaseBuf are not used by this function.
38;; They are only used by the magitude and phase calculation functions.
39;;
40;; ALGORITHUM:
41;;
42;; 1) Bit reverse input data and calculate stages 1, 2 & 3:
43;;
44;; In Buf (read in bit reverse order) Out Buf
45;; +----+ +----+
46;; | I1 | (((I1 + I2) + (I3 + I4)) + ((I5 + I6) + (I7 + I8)))/8 -> | I1'|
47;; | I2 | ((I1 - I2) + COS*((I5 - I6) + (I8 - I7)) )/8 -> | I2'|
48;; | I3 | ((I1 + I2) - (I3 + I4) )/8 -> | I3'|
49;; | I4 | ((I1 - I2) - COS*((I5 - I6) + (I8 - I7)) )/8 -> | I4'|
50;; | I5 | (((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)))/8 -> | I5'|
51;; | I6 | (COS*((I8 - I7) - (I5 - I6)) - (I4 - I3) )/8 -> | I6'|
52;; | I7 | ((I7 + I8) - (I5 + I6) )/8 -> | I7'|
53;; | I8 | (COS*((I8 - I7) - (I5 - I6)) + (I4 - I3) )/8 -> | I8'|
54;; .
55;; .
56;; \|/
57;; Repeat above FFTSize/8 (i.e. if FFTSize = 1024, Repeat = 128 times)
58;;
59;; Note: COS = COS( 1*2*PI/8) = SIN( 1*2*PI/8)
60;;
61;; 2) Calculate stages 4 and up:
62;;
63;;
64;; Out Buf 4 5 6 7 8 9 10 <- Stages
65;; +- +---------+ - - - - - - --
66;; | | Y1 | 0 0 0 0 0 0 0 <- Y1 + Y3
67;; | |---------|
68;; | | X(I1) | 1 1 1 1 1 1 1 <- X(I1) + [X(I3)*COS + X(I4)*SIN]
69;; | |---------|
70;; | | . |
71;; | | . | 3 7 15 31 63 127 255 <- Inner Loop Repeat Times
72;; | | \./ |
73;; | I |---------|
74;; | N | Y2 | 4 8 16 32 64 128 256 <- Y2
75;; | N |---------|
76;; | E | /.\ |
77;; | R | . |
78;; | | . |
79;; | L |---------|
80;; | O | X(I2) | 7 15 31 63 127 255 511 <- X(I1) - [X(I3)*COS + X(I4)*SIN]
81;; | O |---------|
82;; | P | Y3 | 8 16 32 64 128 256 512 <- Y1 - Y3
83;; | |---------|
84;; | | X(I3) | 9 17 33 65 129 257 513 <- [X(I4)*COS - X(I3)*SIN] - X(I2)
85;; | |---------|
86;; | | . |
87;; | | . |
88;; | | \./ |
89;; | |---------|
90;; | | Y4 |12 24 48 96 192 384 768 <- -Y4
91;; | |---------|
92;; | | /.\ |
93;; | | . |
94;; | | . |
95;; | |---------|
96;; | | X(I4) |15 31 63 127 255 511 1023 <- [X(I4)*COS - X(I3)*SIN] + X(I2)
97;; +- |---------|
98;; | |16 32 64 128 256 512 1024
99;; -- --- --- --- --- --- ----
100;; . 1 <- Outer Loop Repeat Times (16 FFT)
101;; . 2 1 <- Outer Loop Repeat Times (32 FFT)
102;; . 4 2 1 <- Outer Loop Repeat Times (64 FFT)
103;; . 8 4 2 1 <- Outer Loop Repeat Times (128 FFT)
104;; . 16 8 4 2 1 <- Outer Loop Repeat Times (256 FFT)
105;; . 32 16 8 4 2 1 <- Outer Loop Repeat Times (512 FFT)
106;; . 64 32 16 8 4 2 1 <- Outer Loop Repeat Times (1024 FFT)
107;; .
108;; .
109;; \|/
110;;
111;;#############################################################################

136;===========================================================================
137; Function: void rfft_adc_f32_Stages1and2and3andBitReverse(
138; RFFT_ADC_F32_STRUCT_Handle)
139;===========================================================================
140;---------------------------------------------------------------------------
141;
142; DESCRIPTION:
143;
144; This function bit reverses the input and computes stages 1, 2 and 3
145;
146; ON ENTRY:
147;
148; XAR4 = Starting address of the RFFT_F32_STRUCT structure
149;
150; REGISTER USAGE:
151;
152; AR0, XAR1, XAR2, XAR4, XAR5, AR3, AR6, XAR7, ACC,
153; R0H, R1H, R2H, R3H, R4H, R5H, R6H, R7H
154;
155; On Exit:
156;
157; RFFT_F32_STRUCT OutBuf contains the computed result
158;
159;---------------------------------------------------------------------------
160;offsets to the stack which holds the bit reversed elements
161;---------------------------------------------------------------------------
162
163I1 .set 06H
164I2 .set 08H
165I3 .set 0AH
166I4 .set 0CH
167I5 .set 0EH
168I6 .set 10H
169I7 .set 12H
170I8 .set 14H
171cossinBuf .set 04H
172;------------------------------------------------------------------------------
173;Offset to the stack which holds intermediate results of FFT computation
174;------------------------------------------------------------------------------
175 .text
176
177_rfft_adc_f32_Stages1and2and3andBitReverse:
178;----------------------------------------------------------------------
179; Save all save-on-entry registers used
180;----------------------------------------------------------------------
181 PUSH XAR1
182 PUSH XAR2
183 PUSH XAR3
184 MOV32 *SP++,R4H
185 MOV32 *SP++,R5H
186 MOV32 *SP++,R6H
187 MOV32 *SP++,R7H
188 ADDB SP,#14h
189
190 MOVL XAR2,*+XAR4[0] ; &Inbuf
191 MOVL XAR5,*+XAR4[2] ; &Outbuf
192 SUBB XAR5, #2
193 MOVL XAR4, XAR5
194 MOVL XAR5,*+XAR4[2] ; &Outbuf
195 MOVL XAR3,*+XAR4[2]
196 ADDB XAR3,#8 ; &Outbuf[4]
197 MOVL XAR7,*+XAR4[4] ; &CosSinbuf
198
199 MOV AR0,#0Ah
200 MOV AH,*+XAR4[AR0] ; FFT SIZE
201 LSR AH,1 ; FFT SIZE/2 - for 16-bit input data
202 MOV AR0,AH
203; LSR AH,3
204 LSR AH,2 ; for 16-bit input data
205 SUBB AH,#1 ; (Size / 8) - 1
206 MOVL XAR1,#0000h ; index if memory is not aligned
207
208 RPTB _rfft_32_Last, AH
209;------------------------------------------------------------------------------
210; Input buffer must be aligned for this code
211;------------------------------------------------------------------------------
212 NOP *,ARP2
213 MOVI32 R4H, 0x39800801 ; scale factor of 1/(2^12 - 1)
214 UI16TOF32 R0H, *BR0++ ;I1 load
215 UI16TOF32 R1H, *BR0++ ;I2 load
216 MPYF32 R0H, R0H, R4H ;I1 scale
217 UI16TOF32 R2H, *BR0++ ;I3 load
218 MPYF32 R1H, R1H, R4H ;I2 scale
219 || MOV32 *-SP[I1], R0H ;I1 save
220 UI16TOF32 R0H, *BR0++ ;I4 load
221 MPYF32 R2H, R2H, R4H ;I3 scale
222 || MOV32 *-SP[I2], R1H ;I2 save
223 UI16TOF32 R1H, *BR0++ ;I5 load
224 MPYF32 R0H, R0H, R4H ;I4 scale
225 || MOV32 *-SP[I3], R2H ;I3 save
226 UI16TOF32 R2H, *BR0++ ;I6 load
227 MPYF32 R1H, R1H, R4H ;I5 scale
228 || MOV32 *-SP[I4], R0H ;I4 save
229 UI16TOF32 R0H, *BR0++ ;I7 load
230 MPYF32 R2H, R2H, R4H ;I6 scale
231 || MOV32 *-SP[I5], R1H ;I5 save
232 UI16TOF32 R1H, *BR0++ ;I8 load
233 MPYF32 R0H, R0H, R4H ;I7 scale
234 || MOV32 *-SP[I6], R2H ;I6 save
235 MPYF32 R1H, R1H, R4H ;I8 scale
236 MOV32 *-SP[I7], R0H ;I7 save
237 MOV32 *-SP[I8], R1H ;I8 save
238
239
240;------------------------------------------------------------------------------
241; Computations for stages 1 2, and 3
242; OutBufIndex++ = (I1 + I2) + (I3 + I4) + (I5 + I6) + (I7 + I8); (A) <- XAR5
243; OutBufIndex++ = (I1 - I2) + COS x ((I5 - I6) + (I8 - I7)); (B)
244; OutBufIndex++ = (I1 + I2) - (I3 + I4); (C)
245; OutBufIndex++ = (I1 - I2) - COS x ((I5 - I6) + (I8 - I7)); (D)
246; OutBufIndex++ = ((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)); (E) <- XAR3
247; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) - (I4 - I3); (F)
248; OutBufIndex++ = (I7 + I8) - (I5 + I6); (G)
249; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) + (I4 - I3); (H)
250;------------------------------------------------------------------------------
251 MOV32 R0H, *-SP[I8] ; R0H = I8
252 MOV32 R1H, *-SP[I7] ; R1H = I7
253 SUBF32 R2H, R0H, R1H ; R2H = I8-I7
254 || MOV32 R3H, *-SP[I5] ; R3H = I5
255 ADDF32 R0H, R1H, R0H ; R0H = I7+I8
256 || MOV32 R4H, *-SP[I6] ; R4H = I6
257 SUBF32 R1H, R3H, R4H ; R1H = I5-I6
258 || MOV32 R5H, *-SP[I1] ; R5H = I1
259 ADDF32 R3H, R3H, R4H ; R3H = I5+I6
260 ADDF32 R4H, R1H, R2H ; R4H = (I5-I6) + (I8-I7)
261 || MOV32 R6H,*XAR7 ; R6H = COS
262 SUBF32 R7H, R0H, R3H ; R7H = (I7+I8) - (I5+I6) (G)
263 MPYF32 R4H, R6H, R4H ; R4H = COS x ((I5-I6) + (I8-I7))
264 || SUBF32 R2H, R2H, R1H ; R2H = (I8-I7) - (I5-I6)
265 ADDF32 R0H, R3H, R0H ; R0H = (I5+I6) + (I7+I8)
266 || MOV32 *+XAR3[4], R7H ; store G
267 MPYF32 R2H, R6H, R2H ; R2H = COS x ((I8-I7) - (I5-I6))
268 || MOV32 R7H, *-SP[I2] ; R7H = I2
269 SUBF32 R6H, R5H, R7H ; R6H = I1-I2
270 ADDF32 R5H, R5H, R7H ; R5H = I1+I2
271 || MOV32 R1H, *-SP[I4] ; R1H = I4
272 ADDF32 R3H, R6H, R4H ; R3H = (I1-I2) + COS x ((I5-I6) + (I8-I7)) (B)
273 || MOV32 R7H, *-SP[I3] ; R7H = I3
274 SUBF32 R6H, R6H, R4H ; R6H = (I1-I2) - COS x ((I5-I6) + (I8-I7)) (D)
275 SUBF32 R3H, R1H, R7H ; R3H = I4-I3
276 || MOV32 *+XAR5[2], R3H ; store B
277 ADDF32 R4H, R7H, R1H ; R4H = I3+I4
278 || MOV32 *+XAR5[6], R6H ; store D
279 SUBF32 R1H, R2H, R3H ; R1H = COS x ((I8-I7) - (I5-I6)) - (I4-I3) (F)
280 ADDF32 R2H, R2H, R3H ; R2H = COS x ((I8-I7) - (I5-I6)) + (I4-I3) (H)
281 ADDF32 R6H, R5H, R4H ; R6H = (I1+I2) + (I3+I4)
282 || MOV32 *+XAR3[2], R1H ; save F
283 SUBF32 R1H, R5H, R4H ; R1H = (I1+I2) - (I3+I4) (C)
284 || MOV32 *+XAR3[6], R2H ; save H
285 SUBF32 R3H, R6H, R0H ; R3H = (I1+I2) + (I3+I4) - ((I5+I6) + (I8+I7)) (E)
286 ADDF32 R0H, R6H, R0H ; R0H = (I1+I2) + (I3+I4) + (I5+I6) + (I8+I7) (A)
287 || MOV32 *+XAR5[4], R1H ; store C
288 MOV32 *+XAR3[0], R3H ; store E
289 MOV32 *+XAR5[0], R0H ; store A
290 ADDB XAR5, #16
291 ADDB XAR3, #16
292
293
294_rfft_32_Last:
295 SUBB SP,#14h
296 MOV32 R7H,*--SP
297 MOV32 R6H,*--SP
298 MOV32 R5H,*--SP
299 MOV32 R4H,*--SP
300 POP XAR3
301 POP XAR2
302 POP XAR1
303 LRETR

305;===========================================================================
306; Function: void rfft_f32_Stages4andUp(RFFT_ADC_F32_STRUCT_Handle)
307;===========================================================================
308;===========================================================================
309;
310; ON ENTRY:
311;
312; XAR4 = Starting address of the RFFT_F32_STRUCT structure
313;
314; REGISTER USAGE:
315;
316; AR0, XAR1, XAR2, XAR4, XAR5, AR3, AR6, XAR7, ACC,
317; R0H, R1H, R2H, R3H, R4H, R5H, R6H, R7H
318;
319; On Exit:
320;
321; RFFT_F32_STRUCT OutBuf contains the computed result
322;
323;---------------------------------------------------------------------------
324
325
326;----------------------------------------------------------------------
327; FFT_REAL structure offsets from XAR4
328;----------------------------------------------------------------------
329RFFT_InBuf .set (0*2)
330RFFT_OutBuf .set (1*2)
331RFFT_CosSinBuf .set (2*2)
332RFFT_MagBuf .set (3*2)
333RFFT_PhaseBuf .set (4*2)
334RFFT_FFTSize .set (5*2)
335RFFT_FFTStages .set (RFFT_FFTSize+1)
336;----------------------------------------------------------------------
337; Offsets for local variables stored on the stack
338;----------------------------------------------------------------------
339I1OFFSET .set 1 ; 16-bit
340I2I4OFFSET .set 2 ; 16-bit
341IDX_I .set 3 ; 16-bit
342IDX_K .set 4 ; 16-bit
343IDX_L .set 5 ; 16-bit
344IDX_SZBL .set 6 ; 16-bit
345IDX_M .set 8 ; 16-bit
346TEMPX1 .set (IDX_M+2*1) ; 32-bit
347OUTBUF .set 12
348COSSINBUF .set 14
349FRMSZ .set COSSINBUF
350
351 .global _rfft_adc_f32_Stages4andUp
352 .text
353
354_rfft_adc_f32_Stages4andUp:
355
356;----------------------------------------------------------------------
357; Save all save-on-entry registers used
358;----------------------------------------------------------------------
359 PUSH XAR1
360 PUSH XAR2
361 PUSH XAR3
362 MOV32 *SP++,R4H
363 MOV32 *SP++,R5H
364 MOV32 *SP++,R6H
365 MOV32 *SP++,R7H
366 ADDB SP,#FRMSZ
367;----------------------------------------------------------------------
368; Initialize loop counters and offsets
369; Note: Index k has changed from 3 to 2 due to loop unrolling.
370;----------------------------------------------------------------------
371 MOV *-SP[I1OFFSET],#26 ; I1 & I3 pointer offset
372 MOV *-SP[I2I4OFFSET],#38 ; I2 & I4 pointer offset
373 MOV *-SP[IDX_K],#(2-1) ; Inner loop count
374 MOV *-SP[IDX_L],#(16*2) ; 16 32-bit values calculated each
375 ; inner loop for the first stage
376;----------------------------------------------------------------------
377; Check the FFTStages and compute stage, outer loop count
378;----------------------------------------------------------------------
379 MOVB XAR1,#RFFT_FFTStages
380 MOV AL,*+XAR4[AR1]
381 SUB AL,#3 ; If (fft->FFTStages <= 3) exit
382 B _rfft_f32_End,LEQ
383 MOV *-SP[IDX_I],AL ; IDX_I = FFTStages - 3 = i for
384 ; stages loop control
385 MOVB XAR0,#RFFT_FFTSize
386 MOV AL,*+XAR4[AR0]
387 ASR AL,#4
388 MOV *-SP[IDX_SZBL],AL ; IDX_SZBL = FFTSize/16
389 DEC AL
390 MOV AR6, AL ; IDX_M = FFTSize/16-1 = m for
391 ; outer loop control
392;----------------------------------------------------------------------
393; Initialize pointers
394;----------------------------------------------------------------------
395 .asg AR0, Offset1
396 .asg AR1, Offset2
397 .asg XAR2, I1ptr
398 .asg AR3, InnerLoopCnt
399 .asg XAR4, I2ptr
400 .asg XAR5, I4ptr
401 .asg AR6, OuterLoopCnt
402 .asg XAR7, COSptr
403 .asg XAR7, SINptr
404
405
406 MOVL ACC, *+XAR4[RFFT_CosSinBuf]
407 MOVL *-SP[COSSINBUF], ACC
408 ADDB ACC, #8
409 MOVL *-SP[TEMPX1], ACC ; TEMPX1 = &CosSinBuf[4]
410
411
412 MOVL I1ptr, *+XAR4[RFFT_OutBuf] ; I1ptr = &OutBuf[0]
413 MOVL *-SP[OUTBUF], I1ptr
414 MOVL I2ptr, *+XAR4[RFFT_OutBuf]
415 MOVL I4ptr, *-SP[OUTBUF]
416
417 ADDB I2ptr, #16 ; I2ptr = &OutBuf[8]
418 MOV Offset1,#16 ; I1ptr[Offset1] = &OutBuf[8]
419 MOV Offset2,#18 ; I1ptr[Offset2] = &OutBuf[9]
420 ADDB I4ptr, #32 ; I4ptr = &OutBuf[16]
421 MOV InnerLoopCnt, #(2-1) ; Inner loop count for 4th stage
422;----------------------------------------------------------------------
423; Stage Loop: For each stage 4 and up:
424;----------------------------------------------------------------------
425_rfft_f32_StageLoop:
426;----------------------------------------------------------------------
427; Outer Loop: IDX_M
428; Stage 4: M = 1 to FFTSize/16
429; Stage 5: M = 1 to FFTSize/32
430; Stage 6: M = 1 to FFTSize/64 etc..
431;
432; Calculate the following M times:
433;
434; I1 = Y1 = *I1ptr;
435; I3 = *I3ptr;
436; *I1ptr++ = Y = (I1 + I3) = Y1 + Y3
437; *I3ptr++ = Y3 = (I1 - I3) = Y1 - Y3
438; for(j=1; j <= k; j ++)
439; {
440; I1 = *I1ptr;
441; I2 = *I2ptr;
442; I3 = *I3ptr;
443; I4 = *I4ptr;
444; COS = COSptr++;
445; SIN = COSptr++;
446; I1ptr++ = I1 + I3xCOS + I4xSIN; (A)
447; I2ptr-- = I1 - (I3xCOS + I4xSIN); (B)
448; I3ptr++ = I4xCOS - I3xSIN - I2; (C)
449; I4ptr-- = I4xCOS - I3xSIN + I2; (D)
450; }
451; Y4 = -Y4:
452;----------------------------------------------------------------------
453_rfft_f32_OuterLoop:
454 MOVL COSptr, *-SP[TEMPX1]
455 MOV32 R1H, *I1ptr ; R1H = Y1
456 MOV32 R3H, *I2ptr ; R3H = Y3
457 ADDF32 R2H, R1H, R3H ; R2H = Y1 + Y3
458 || MOV32 R0H, *COSptr++ ; R0H = COS (1)
459 SUBF32 R4H, R1H, R3H ; R4H = (Y1 - Y3)
460 || MOV32 R3H, *+I1ptr[Offset2] ; R3H = I3 (1)
461 MOV32 *I1ptr++, R2H ; I1 = (Y1 + Y3)
462 MPYF32 R1H, R3H, R0H ; R1H = I3*COS (1)
463 || MOV32 R6H, *--I4ptr ; R6H = I4 (1)
464 MPYF32 R2H, R6H, R0H ; R2H = I4*COS (1)
465 || MOV32 R7H, *SINptr++ ; R7H = SIN (1)
466 MPYF32 R3H, R3H, R7H ; R3H = I3*SIN (1)
467 || MOV32 *I2ptr,R4H ; Y3 = Y1 - Y3
468
469 RPTB _rfft_f32_InnerLoop, InnerLoopCnt
470 MPYF32 R4H, R6H, R7H ; R4H = I4*SIN (1)
471 || MOV32 R6H, *--I2ptr ; R6H = I2 (1)
472 SUBF32 R7H, R2H, R3H ; R7H = I4*COS - I3*SIN (1)
473 || MOV32 R0H, *COSptr++ ; R0H = COS (2)
474 ADDF32 R3H, R1H, R4H ; R3H = I3*COS + I4*SIN (1)
475 || MOV32 R5H, *+I1ptr[Offset2] ; R5H = I3 (2)
476 SUBF32 R2H, R7H, R6H ; R2H = I4*COS - I3*SIN - I2 (C1)
477 || MOV32 R4H, *I1ptr ; R4H = I1 (1)
478 ADDF32 R7H, R7H, R6H ; R7H = I4*COS - I3*SIN + I2 (D1)
479 || MOV32 R6H, *--I4ptr ; R6H = I4 (2)
480 ADDF32 R2H, R4H, R3H ; R2H = I1 + (I3*COS + I4*SIN) (A1)
481 || MOV32 *+I1ptr[Offset1], R2H ; I3 = I4*COS - I3*SIN - I2 (save C1)
482 SUBF32 R4H, R4H, R3H ; R4H = I1 - (I3*COS + I4*SIN) (B1)
483 || MOV32 *+I4ptr[2], R7H ; I4 = I4*COS - I3*SIN + I2 (save D1)
484 MPYF32 R1H, R5H, R0H ; R1H = I3*COS (2)
485 || MOV32 *I1ptr++, R2H ; I1 = I1 + (I3*COS + I4*SIN) (save A1)
486 MPYF32 R2H, R6H, R0H ; R2H = I4*COS (2)
487 || MOV32 R7H, *SINptr++ ; R7H = SIN (2)
488 MPYF32 R3H, R5H, R7H ; R3H = I3*SIN (2)
