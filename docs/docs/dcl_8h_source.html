117static inline void DCL_PI_Run(DCL_PI *p, float32_t rk, float32_t yk)
118{
119 float32_t v1, v5, v7, v8;
120 uint16_t l11, l12, l14, l17, l18, l19;
121
122 v1 = rk - yk;
123 v5 = (v1 * p->Ki * p->i6) + p->i10;
124 p->i10 = v5;
125 v7 = (v1 * p->Kp) + v5;
126 v8 = (v7 > p->Umax) ? p->Umax : v7;
127 v8 = (v8 < p->Umin) ? p->Umin : v8;
128 l17 = ((v7 - v8) == 0) ? 1U : 0U;
129 l11 = (v5 >= p->Imax) ? 1U : 0U;
130 l12 = (v5 <= p->Imin) ? 1U : 0U;
131 l19 = (v5 > 0) ? 1U : 0U;
132 l14 = (v1 > 0) ? 1U : 0U;
133 l18 = l17 & (!(l11 | l12) | (l19 ^ l14));
134 p->i6 = (l18 == 0U) ? 0.0f : 1.0f;
135 p->Output = v8;
136}

248static inline void DCL_PMR_Run(DCL_PMR *p, float32_t rk, float32_t yk, float32_t fk)
249{
250 // 计算偏差
251 float32_t err = rk-yk;
252
253 // P
254 p->Output = p->Kp * err;
255
256 // MR
257 uint16_t i = 0;
258
259 for ( i = 0; i < DCL_HARMONIC_NUM; i++)
260 {
261 p->Output = p->Output + DCL_DF22_Run(&(p->pr[i]), err);
262 }
263
264 // 前馈解耦
265 p->Output = p->Output + fk;
266}
