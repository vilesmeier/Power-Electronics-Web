67void POWER_3ph_Run(POWER_3PH *power)
68{
69 // getting current sample number
70 power->sample_num = power->sample_num + 1;
71
72 // rms for volate and current
73 power->va_sum = power->Va * power->Va + power->va_sum;
74 power->vb_sum = power->Vb * power->Vb + power->vb_sum;
75 power->vc_sum = power->Vc * power->Vc + power->vc_sum;
76 power->ia_sum = power->Ia * power->Ia + power->ia_sum;
77 power->ib_sum = power->Ib * power->Ib + power->ib_sum;
78 power->ic_sum = power->Ic * power->Ic + power->ic_sum;
79 power->p_sum = power->Va * power->Ia + power->Vb * power->Ib + power->Vc * power->Ic + power->p_sum;
80
81 // when sample == nsample, indecating get a sine period
82 // update output
83 if (power->nsample == power->sample_num)
84 {
85 // voltage rms
86 power->Varms = sqrtf(power->va_sum * power->nsample_inv);
87 power->Vbrms = sqrtf(power->vb_sum * power->nsample_inv);
88 power->Vcrms = sqrtf(power->vc_sum * power->nsample_inv);
89
90 // current rms
91 power->Iarms = sqrtf(power->ia_sum * power->nsample_inv);
92 power->Ibrms = sqrtf(power->ib_sum * power->nsample_inv);
93 power->Icrms = sqrtf(power->ic_sum * power->nsample_inv);
94
95 // power
96 power->P = power->nsample_inv * power->p_sum;
97 power->S = power->Varms * power->Iarms + power->Vbrms * power->Ibrms + power->Vcrms * power->Icrms;
98
99 power->PF = (power->S > 1e-6f) ? (power->P / power->S) : 0.0f;
100
101 float32_t temp = power->S * power->S - power->P * power->P;
102 power->Q = (temp > 0.0f) ? sqrtf(temp) : 0.0f;
103
104 // reset integrator
105 power->va_sum = 0;
106 power->vb_sum = 0;
107 power->vc_sum = 0;
108 power->ia_sum = 0;
109 power->ib_sum = 0;
110 power->ic_sum = 0;
111 power->p_sum = 0;
112
113 // reset sample counter
114 power->sample_num = 0;
115 }
116}

118void POWER_3ph_RMS_Run(POWER_3PH *power, float32_t Varms, float32_t Vbrms, float32_t Vcrms, float32_t Iarms, float32_t Ibrms, float32_t Icrms, float32_t PF)
119{
120 power->Varms = Varms;
121 power->Vbrms = Vbrms;
122 power->Vcrms = Vcrms;
123 power->Iarms = Iarms;
124 power->Ibrms = Ibrms;
125 power->Icrms = Icrms;
126
127 power->S = power->Varms * power->Iarms + power->Vbrms * power->Ibrms + power->Vcrms * power->Icrms;
128 power->P = power->PF * power->S;
129 power->Q = power->S * power->S - power->P *power->P;
130 if (power->Q < 0)
131 {
132 power->Q = 0;
133 }
134 power->Q = sqrtf(power->Q);
