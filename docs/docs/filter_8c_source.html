17void Filter_Set(FILTER *fil, FilterType type, float32_t wc, float32_t damp, float32_t isr_freq)
18{
19 float32_t delta_t = (1.0f) / isr_freq;
20 float32_t A0;
21
22 // 在线修改滤波器参数时, 清零中间变量
23 fil->filter.x1 = 0.0f;
24 fil->filter.x2 = 0.0f;
25
26 // 计算滤波器参数, 使用Tustin双线性离散化
27 switch (type)
28 {
29 case FIL_NULL: // 未指定滤波器类型或者滤波器类型错误, 配置为直通
30 fil->filter.a1 = 0.0f;
31 fil->filter.a2 = 0.0f;
32 fil->filter.b0 = 1.0f;
33 fil->filter.b1 = 0.0f;
34 fil->filter.b2 = 0.0f;
35 break;
36
37 case LPF_1_ORDER: //config as 1 order LPF， 一阶滤波器使用 DCL_DF22 结构体实现
38 A0 = wc * delta_t + 2.0f;
39 fil->filter.a1 = (wc * delta_t - 2.0f) / A0;
40 fil->filter.a2 = 0.0f;
41 fil->filter.b0 = wc * delta_t / A0;
42 fil->filter.b1 = fil->filter.b0;
43 fil->filter.b2 = 0.0f;
44 break;
45
46 case HPF_1_ORDER: //config as 1 order HPF
47 A0 = 2.0f + wc * delta_t;
48 fil->filter.a1 = (wc * delta_t - 2.0f) / A0;
49 fil->filter.a2 = 0.0f;
50 fil->filter.b0 = (2.0f) / A0;
51 fil->filter.b1 = -fil->filter.b0;
52 fil->filter.b2 = 0.0f;
53 break;
54
55 case LPF_2_ORDER: //config as 2 order LPF
56 A0 = 4.0f + 4.0f * wc * delta_t * damp + wc * wc * delta_t * delta_t;
57 fil->filter.a1 = (2 * wc * wc * delta_t * delta_t - 8.0f) / A0;
58 fil->filter.a2 = (4.0f - 4.0f * wc * delta_t * damp + wc * wc * delta_t * delta_t) / A0;
59 fil->filter.b0 = (wc * wc * delta_t * delta_t) / A0;
60 fil->filter.b1 = 2.0f * fil->filter.b0;
61 fil->filter.b2 = fil->filter.b0;
62 break;
63
64 case HPF_2_ORDER: //config as 2 order HPF
65 A0 = 4.0f + 4.0f * damp * wc * delta_t + wc * wc * delta_t * delta_t;
66 fil->filter.a1 = (-8.0f + 2.0f * wc * wc * delta_t * delta_t) / A0;
67 fil->filter.a2 = ( 4.0f - 4.0f * damp * wc * delta_t + wc * wc * delta_t * delta_t) / A0;
68 fil->filter.b0 = 4.0f / A0;
69 fil->filter.b1 = -2.0f * fil->filter.b0;
70 fil->filter.b2 = fil->filter.b0;
71 break;
72
73
74 case BPF_2_ORDER: //config as 2 order BPF
75 A0 = 4.0f + 4.0f * delta_t * damp * wc + wc * wc * delta_t * delta_t;
76 fil->filter.a1 = (-8.0f + 2.0f * wc * wc * delta_t * delta_t) / A0;
77 fil->filter.a2 = ( 4.0f - 4.0f * delta_t * damp * wc + wc * wc * delta_t * delta_t) / A0;
78 fil->filter.b0 = 4.0f * damp * wc * delta_t / A0;
79 fil->filter.b1 = 0.0f;
80 fil->filter.b2 = -fil->filter.b0;
81 break;
82
83 case BSF_2_ORDER: //config as 2 order BSF
84 A0 = 4.0f + 4.0f * delta_t * damp * wc + wc * wc * delta_t * delta_t;
85 fil->filter.a1 = (-8.0f + wc * wc * delta_t * delta_t * 2) / A0 ;
86 fil->filter.a2 = ( 4.0f - 4.0f * damp * wc * delta_t + wc * wc * delta_t * delta_t) / A0 ;
87 fil->filter.b0 = ( 4.0f + wc * wc * delta_t * delta_t) / A0 ;
88 fil->filter.b1 = (-8.0f + wc * wc * delta_t * delta_t * 2) / A0 ;
89 fil->filter.b2 = fil->filter.b0 ;
90 break;
91
92 default: // 未指定滤波器类型或者滤波器类型错误, 配置为直通
93 fil->filter.a1 = 0.0f;
94 fil->filter.a2 = 0.0f;
95 fil->filter.b0 = 1.0f;
96 fil->filter.b1 = 0.0f;
97 fil->filter.b2 = 0.0f;
98 break;
99 }
100}
