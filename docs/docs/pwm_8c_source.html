204void EPWM_Init(void)
205{
206
207 EALLOW;
208 SysCtrlRegs.PCLKCR0.bit.TBCLKSYNC = 0;
209 EDIS;
210
211 //init GPIO for ePWM1-6
212 InitEPwmGpio();
213 //init GPIO for Tz1-6
214 InitTzGpio();
215
216 //init system interrupt
217 DINT;
218 EALLOW;
219 PieVectTable.EPWM1_INT = &epwm1_isr;
220 IER |= M_INT3;
221 PieCtrlRegs.PIEIER3.bit.INTx1 = 1;
222 EDIS;
223 EINT;
224 ERTM;
225
226 //init ePWM
227 EPWM1_Init();
228 EPWM2_Init();
229 EPWM3_Init();
230 EPWM4_Init();
231 EPWM5_Init();
232 EPWM6_Init();
233
234 EALLOW;
235 SysCtrlRegs.PCLKCR0.bit.TBCLKSYNC = 1;
236 EDIS;
237}
