31;; ASSUMPTIONS:
32;;
33;; * FFTSize must be a power of 2 (32, 64, 128, etc)
34;; * FFTSize must be greater or equal to 32
35;; * FFTStages must be ln(FFTSize)/ln(2)
36;; * InBuf, OutBuf, CosSinBuf are FFTSize in length

46;; +----------------------------+

58;; . .
59;; . .
60;; . .

62;; | I(3) |
63;; | I(2) |
64;; | I(1) | FFTSize - 1
65;; +----------------------------+

140_RFFT_f32_phase:
141
142 PUSH XAR1 ; Save XAR1 (save-on-entry register)
143 PUSH XAR2 ; Save XAR2 (save-on-entry register)
144 PUSH XAR3 ; Save XAR3 (save-on-entry register)
145
146 MOVL XAR2, *+XAR4[2] ; XAR1 = Rptr = &OutBuf[0]
147 ADDB XAR4, #8 ; Move the structure pointer up
148 MOVL XAR1, *+XAR4[0] ; XAR2 = &Phasebuf
149
150;--- Calculate pointer to imaginary part results
151 MOV ACC, *+XAR4[2] << 1 ; ACC = 2*FFTSize
152 ADDL ACC, XAR2 ; Add OutBuff address
153 MOVL XAR3, ACC ; XAR3 = Iptr = &OutBuf[last value]
154
155;--- Calculate main loop count
156 MOV AL, *+XAR4[2] ; AL = FFTSize
157 LSR AL, #1 ; AL = FFTSize/2
158 SUBB AL, #2 ; phase[0] handled separately (subtract 1)
159 ; loop counter is "N-1" (subtract 1)
160 MOV AR4, AL ; AR4 = calc_phase counter
161
162;---------------------------------------------------------------------------
163; Begin phase calculations
164;---------------------------------------------------------------------------
165
166;--- phase[0] is handled separately since no Im part
167 MOVIZF32 R0H, #0 ; R0H = 0.0
168 MOV32 *XAR1++, R0H ; Store phase[0]
169 ADDB XAR2, #2 ; Increment Rptr to next value
170
171_rfft_f32_calc_phase:
172 MOV32 R1H, *XAR2++ ; R1H = Re[i]
173 MOV32 R0H, *--XAR3 ; R0H = Im[i]
174 PUSH XAR4 ; Save AR4 (save-on-call register)
175 LCR _atan2f ; R0H = phase[i] = atan2f(Im[i], Re[i]);
176 POP XAR4 ; Restore AR4
177 MOV32 *XAR1++,R0H ; Store phase[i]
178 BANZ _rfft_f32_calc_phase,AR4--
179
180;--- Finish up
181 POP XAR3 ; Restore XAR3
182 POP XAR2 ; Restore XAR2
183 POP XAR1 ; Restore XAR1
184 LRETR
