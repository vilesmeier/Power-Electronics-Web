56struct MODE_BITS { // bits description
57 Uint16 PERINTSEL:5; // 4:0 Peripheral Interrupt and Sync Select Bits (R/W):
58 // 0 no interrupt
59 // 1 SEQ1INT & ADCSYNC
60 // 2 SEQ2INT
61 // 3 XINT1
62 // 4 XINT2
63 // 5 XINT3
64 // 6 XINT4
65 // 7 XINT5
66 // 8 XINT6
67 // 9 XINT7
68 // 10 XINT13
69 // 11 TINT0
70 // 12 TINT1
71 // 13 TINT2
72 // 14 MXEVTA & MXSYNCA
73 // 15 MREVTA & MRSYNCA
74 // 16 MXEVTB & MXSYNCB
75 // 17 MREVTB & MRSYNCB
76 // 18 ePWM1SOCA
77 // 19 ePWM1SOCB
78 // 20 ePWM2SOCA
79 // 21 ePWM2SOCB
80 // 22 ePWM3SOCA
81 // 23 ePWM3SOCB
82 // 24 ePWM4SOCA
83 // 25 ePWM4SOCB
84 // 26 ePWM5SOCA
85 // 27 ePWM5SOCB
86 // 28 ePWM6SOCA
87 // 29 ePWM6SOCB
88 // 30:31 no interrupt
89 Uint16 rsvd1:2; // 6:5 (R=0:0)
90 Uint16 OVRINTE:1; // 7 Overflow Interrupt Enable (R/W):
91 // 0 overflow interrupt disabled
92 // 1 overflow interrupt enabled
93 Uint16 PERINTE:1; // 8 Peripheral Interrupt Enable Bit (R/W):
94 // 0 peripheral interrupt disabled
95 // 1 peripheral interrupt enabled
96 Uint16 CHINTMODE:1; // 9 Channel Interrupt Mode Bit (R/W):
97 // 0 generate interrupt at beginning of new
98 // transfer
99 // 1 generate interrupt at end of transfer
100 Uint16 ONESHOT:1; // 10 One Shot Mode Bit (R/W):
101 // 0 only interrupt event triggers single
102 // burst transfer
103 // 1 first interrupt triggers burst,
104 // continue until transfer count is zero
105 Uint16 CONTINUOUS:1;// 11 Continous Mode Bit (R/W):
106 // 0 stop when transfer count is zero
107 // 1 re-initialize when transfer count is
108 // zero
109 Uint16 SYNCE:1; // 12 Sync Enable Bit (R/W):
110 // 0 ignore selected interrupt sync signal
111 // 1 enable selected interrupt sync signal
112 Uint16 SYNCSEL:1; // 13 Sync Select Bit (R/W):
113 // 0 sync signal controls source wrap
114 // counter
115 // 1 sync signal controls destination wrap
116 // counter
117 Uint16 DATASIZE:1; // 14 Data Size Mode Bit (R/W):
118 // 0 16-bit data transfer size
119 // 1 32-bit data transfer size
120 Uint16 CHINTE:1; // 15 Channel Interrupt Enable Bit (R/W):
121 // 0 channel interrupt disabled
122 // 1 channel interrupt enabled
123};

130//
131// Channel CONTROL register bit definitions
132//
133struct CONTROL_BITS { // bits description
134 Uint16 RUN:1; // 0 Run Bit (R=0/W=1)
135 Uint16 HALT:1; // 1 Halt Bit (R=0/W=1)
136 Uint16 SOFTRESET:1; // 2 Soft Reset Bit (R=0/W=1)
137 Uint16 PERINTFRC:1; // 3 Interrupt Force Bit (R=0/W=1)
138 Uint16 PERINTCLR:1; // 4 Interrupt Clear Bit (R=0/W=1)
139 Uint16 SYNCFRC:1; // 5 Sync Force Bit (R=0/W=1)
140 Uint16 SYNCCLR:1; // 6 Sync Clear Bit (R=0/W=1)
141 Uint16 ERRCLR:1; // 7 Error Clear Bit (R=0/W=1)
142 Uint16 PERINTFLG:1; // 8 Interrupt Flag Bit (R):
143 // 0 no interrupt pending
144 // 1 interrupt pending
145 Uint16 SYNCFLG:1; // 9 Sync Flag Bit (R):
146 // 0 no sync pending
147 // 1 sync pending
148 Uint16 SYNCERR:1; // 10 Sync Error Flag Bit (R):
149 // 0 no sync error
150 // 1 sync error detected
151 Uint16 TRANSFERSTS:1; // 11 Transfer Status Bit (R):
152 // 0 no transfer in progress or pending
153 // 1 transfer in progress or pending
154 Uint16 BURSTSTS:1; // 12 Burst Status Bit (R):
155 // 0 no burst in progress or pending
156 // 1 burst in progress or pending
157 Uint16 RUNSTS:1; // 13 Run Status Bit (R):
158 // 0 channel not running or halted
159 // 1 channel running
160 Uint16 OVRFLG:1; // 14 Overflow Flag Bit(R)
161 // 0 no overflow event
162 // 1 overflow event
163 Uint16 rsvd1:1; // 15 (R=0)
164};

218struct PRIORITYSTAT_BITS { // bits description
219 Uint16 ACTIVESTS:3; // 2:0 Active Channel Status Bits (R):
220 // 0,0,0 no channel active
221 // 0,0,1 Ch1 channel active
222 // 0,1,0 Ch2 channel active
223 // 0,1,1 Ch3 channel active
224 // 1,0,0 Ch4 channel active
225 // 1,0,1 Ch5 channel active
226 // 1,1,0 Ch6 channel active
227 Uint16 rsvd1:1; // 3 (R=0)
228 Uint16 ACTIVESTS_SHADOW:3; // 6:4 Active Channel Status Shadow Bits (R):
229 // 0,0,0 no channel active & interrupted by Ch1
230 // 0,0,1 cannot occur
231 // 0,1,0 Ch2 was active and interrupted by Ch1
232 // 0,1,1 Ch3 was active and interrupted by Ch1
233 // 1,0,0 Ch4 was active and interrupted by Ch1
234 // 1,0,1 Ch5 was active and interrupted by Ch1
235 // 1,1,0 Ch6 was active and interrupted by Ch1
236 Uint16 rsvd2:9; // 15:7 (R=0:0)
