4;;! \brief Real FFT (Unaligned)
5;;! \author C2000
6;;!
7;; DESCRIPTION:
8;;
9;; This function calculates a real FFT. The input buffer does not have to
10;; be aligned. If alignment can be done, then use the RFFT_f32 function to
11;; improve performance.
12;;
13;; FUNCTIONS:
14;;
15;; void RFFT_f32u(RFFT_F32_STRUCT_Handle)
16;;
17;; where RFFT_F32_STRUCT_Handle is defined as:
18;; typedef RFFT_F32_STRUCT* RFFT_F32_STRUCT_Handle;
19;;
20;; and where RFFT_F32_STRUCT is a structure defined as:
21;;
22;; typedef struct {
23;; float *InBuf;
24;; float *OutBuf;
25;; float *CosSinBuf;
26;; float *MagBuf;
27;; float *PhaseBuf;
28;; uint16_t FFTSize;
29;; uint16_t FFTStages;
30;; } RFFT_F32_STRUCT;
31;;
32;; ASSUMPTIONS:
33;;
34;; * FFTSize must be a power of 2 (32, 64, 128, etc)
35;; * FFTSize must be greater or equal to 32
36;; * FFTStages must be log2(FFTSize)
37;; * InBuf, OutBuf, CosSinBuf are FFTSize in length
38;; * MagBuf and PhaseBuf are FFTSize/2 in length
39;; * MagBuf and PhaseBuf are not used by this function.
40;; They are only used by the magitude and phase calculation functions.
41;;
42;; ALGORITHUM:
43;;
44;; 1) Bit reverse input data and calculate stages 1, 2 & 3:
45;;
46;; In Buf (read in bit reverse order) Out Buf
47;; +----+ +----+
48;; | I1 | (((I1 + I2) + (I3 + I4)) + ((I5 + I6) + (I7 + I8)))/8 -> | I1'|
49;; | I2 | ((I1 - I2) + COS*((I5 - I6) + (I8 - I7)) )/8 -> | I2'|
50;; | I3 | ((I1 + I2) - (I3 + I4) )/8 -> | I3'|
51;; | I4 | ((I1 - I2) - COS*((I5 - I6) + (I8 - I7)) )/8 -> | I4'|
52;; | I5 | (((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)))/8 -> | I5'|
53;; | I6 | (COS*((I8 - I7) - (I5 - I6)) - (I4 - I3) )/8 -> | I6'|
54;; | I7 | ((I7 + I8) - (I5 + I6) )/8 -> | I7'|
55;; | I8 | (COS*((I8 - I7) - (I5 - I6)) + (I4 - I3) )/8 -> | I8'|
56;; .
57;; .
58;; \|/
59;; Repeat above FFTSize/8 (i.e. if FFTSize = 1024, Repeat = 128 times)
60;;
61;; Note: COS = COS( 1*2*PI/8) = SIN( 1*2*PI/8)
62;;
63;; 2) Calculate stages 4 and up:
64;;
65;;
66;; Out Buf 4 5 6 7 8 9 10 <- Stages
67;; +- +---------+ - - - - - - --
68;; | | Y1 | 0 0 0 0 0 0 0 <- Y1 + Y3
69;; | |---------|
70;; | | X(I1) | 1 1 1 1 1 1 1 <- X(I1) + [X(I3)*COS + X(I4)*SIN]
71;; | |---------|
72;; | | . |
73;; | | . | 3 7 15 31 63 127 255 <- Inner Loop Repeat Times
74;; | | \./ |
75;; | I |---------|
76;; | N | Y2 | 4 8 16 32 64 128 256 <- Y2
77;; | N |---------|
78;; | E | /.\ |
79;; | R | . |
80;; | | . |
81;; | L |---------|
82;; | O | X(I2) | 7 15 31 63 127 255 511 <- X(I1) - [X(I3)*COS + X(I4)*SIN]
83;; | O |---------|
84;; | P | Y3 | 8 16 32 64 128 256 512 <- Y1 - Y3
85;; | |---------|
86;; | | X(I3) | 9 17 33 65 129 257 513 <- [X(I4)*COS - X(I3)*SIN] - X(I2)
87;; | |---------|
88;; | | . |
89;; | | . |
90;; | | \./ |
91;; | |---------|
92;; | | Y4 |12 24 48 96 192 384 768 <- -Y4
93;; | |---------|
94;; | | /.\ |
95;; | | . |
96;; | | . |
97;; | |---------|
98;; | | X(I4) |15 31 63 127 255 511 1023 <- [X(I4)*COS - X(I3)*SIN] + X(I2)
99;; +- |---------|
100;; | |16 32 64 128 256 512 1024
101;; -- --- --- --- --- --- ----
102;; . 1 <- Outer Loop Repeat Times (16 FFT)
103;; . 2 1 <- Outer Loop Repeat Times (32 FFT)
104;; . 4 2 1 <- Outer Loop Repeat Times (64 FFT)
105;; . 8 4 2 1 <- Outer Loop Repeat Times (128 FFT)
106;; . 16 8 4 2 1 <- Outer Loop Repeat Times (256 FFT)
107;; . 32 16 8 4 2 1 <- Outer Loop Repeat Times (512 FFT)
108;; . 64 32 16 8 4 2 1 <- Outer Loop Repeat Times (1024 FFT)
109;; .
110;; .
111;; \|/
112;;

142;
143; This function bit reverses the input and computes stages 1, 2 and 3
144; This function does not have any alignment requirements for the input
145; buffer.

147; ON ENTRY:
148;
149; XAR4 = Starting address of the RFFT_F32_STRUCT structure
150;
151; REGISTER USAGE:

153; AR0, XAR1, XAR2, XAR4, XAR5, AR3, AR6, XAR7, ACC,
154; R0H, R1H, R2H, R3H, R4H, R5H, R6H, R7H
155;
156; On Exit:
157;
158; RFFT_F32_STRUCT OutBuf contains the computed result
159;
160;---------------------------------------------------------------------------

162;---------------------------------------------------------------------------

164I1 .set 06H
165I2 .set 08H
166I3 .set 0AH
167I4 .set 0CH
168I5 .set 0EH
169I6 .set 10H

172cossinBuf .set 04H

179_rfft_f32u_Stages1and2and3andBitReverse:
180;----------------------------------------------------------------------
181; Save all save-on-entry registers used
182;----------------------------------------------------------------------
183 PUSH XAR1
184 PUSH XAR2
185 PUSH XAR3
186 MOV32 *SP++,R4H
187 MOV32 *SP++,R5H
188 MOV32 *SP++,R6H
189 MOV32 *SP++,R7H
190 ADDB SP,#14h
191
192 MOVL XAR2,*+XAR4[0] ; &Inbuf
193 MOVL XAR5,*+XAR4[2] ; &Outbuf
194 MOVL XAR3,*+XAR4[2]
195 ADDB XAR3,#8 ; &Outbuf[4]
196 MOVL XAR7,*+XAR4[4] ; &CosSinbuf
197
198 MOV AR0,#0Ah
199 MOV AH,*+XAR4[AR0] ; FFT SIZE
200 MOV AR0,AH
201 LSR AH,3
202 SUBB AH,#1 ; (Size / 8) - 1
203 MOVL XAR1,#0000h ; index if memory is not aligned
204
205 RPTB _rfft_32_Last, AH
206;------------------------------------------------------------------------------
207; Code assumes no aligment
208;------------------------------------------------------------------------------
209 NOP *,ARP1
210 MOVL ACC,*+XAR2[AR1]
211 MOVL *-SP[I1],ACC ;I1
212 NOP *,ARP1
213 NOP *BR0++
214 MOVL ACC,*+XAR2[AR1]

216

245 MOVL *-SP[I8],ACC ;I8

249;-------------------------------------------------------------------------------
250; Computations for stages 1 2, and 3
251; OutBufIndex++ = (I1 + I2) + (I3 + I4) + (I5 + I6) + (I7 + I8); (A) <- XAR5
252; OutBufIndex++ = (I1 - I2) + COS x ((I5 - I6) + (I8 - I7)); (B)
253; OutBufIndex++ = (I1 + I2) - (I3 + I4); (C)
254; OutBufIndex++ = (I1 - I2) - COS x ((I5 - I6) + (I8 - I7)); (D)
255; OutBufIndex++ = ((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)); (E) <- XAR3
256; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) - (I4 - I3); (F)
257; OutBufIndex++ = (I7 + I8) - (I5 + I6); (G)
258; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) + (I4 - I3); (H)
259;-------------------------------------------------------------------------------
260 MOV32 R0H, *-SP[I8] ; R0H = I8
261 MOV32 R1H, *-SP[I7] ; R1H = I7
262 SUBF32 R2H, R0H, R1H ; R2H = I8-I7
263 || MOV32 R3H, *-SP[I5] ; R3H = I5
264 ADDF32 R0H, R1H, R0H ; R0H = I7+I8
265 || MOV32 R4H, *-SP[I6] ; R4H = I6
266 SUBF32 R1H, R3H, R4H ; R1H = I5-I6
267 || MOV32 R5H, *-SP[I1] ; R5H = I1
268 ADDF32 R3H, R3H, R4H ; R3H = I5+I6
269 ADDF32 R4H, R1H, R2H ; R4H = (I5-I6) + (I8-I7)
270 || MOV32 R6H,*XAR7 ; R6H = COS
271 SUBF32 R7H, R0H, R3H ; R7H = (I7+I8) - (I5+I6) (G)
272 MPYF32 R4H, R6H, R4H ; R4H = COS x ((I5-I6) + (I8-I7))
273 || SUBF32 R2H, R2H, R1H ; R2H = (I8-I7) - (I5-I6)
274 ADDF32 R0H, R3H, R0H ; R0H = (I5+I6) + (I7+I8)
275 || MOV32 *+XAR3[4], R7H ; store G
276 MPYF32 R2H, R6H, R2H ; R2H = COS x ((I8-I7) - (I5-I6))
277 || MOV32 R7H, *-SP[I2] ; R7H = I2
278 SUBF32 R6H, R5H, R7H ; R6H = I1-I2
279 ADDF32 R5H, R5H, R7H ; R5H = I1+I2
280 || MOV32 R1H, *-SP[I4] ; R1H = I4
281 ADDF32 R3H, R6H, R4H ; R3H = (I1-I2) + COS x ((I5-I6) + (I8-I7)) (B)
282 || MOV32 R7H, *-SP[I3] ; R7H = I3
283 SUBF32 R6H, R6H, R4H ; R6H = (I1-I2) - COS x ((I5-I6) + (I8-I7)) (D)
284 SUBF32 R3H, R1H, R7H ; R3H = I4-I3
285 || MOV32 *+XAR5[2], R3H ; store B
286 ADDF32 R4H, R7H, R1H ; R4H = I3+I4
287 || MOV32 *+XAR5[6], R6H ; store D
288 SUBF32 R1H, R2H, R3H ; R1H = COS x ((I8-I7) - (I5-I6)) - (I4-I3) (F)
289 ADDF32 R2H, R2H, R3H ; R2H = COS x ((I8-I7) - (I5-I6)) + (I4-I3) (H)
290 ADDF32 R6H, R5H, R4H ; R6H = (I1+I2) + (I3+I4)
291 || MOV32 *+XAR3[2], R1H ; save F
292 SUBF32 R1H, R5H, R4H ; R1H = (I1+I2) - (I3+I4) (C)
293 || MOV32 *+XAR3[6], R2H ; save H
294 SUBF32 R3H, R6H, R0H ; R3H = (I1+I2) + (I3+I4) - ((I5+I6) + (I8+I7)) (E)
295 ADDF32 R0H, R6H, R0H ; R0H = (I1+I2) + (I3+I4) + (I5+I6) + (I8+I7) (A)
296 || MOV32 *+XAR5[4], R1H ; store C
297 MOV32 *+XAR3[0], R3H ; store E
298 MOV32 *+XAR5[0], R0H ; store A
299 ADDB XAR5, #16
300 ADDB XAR3, #16
