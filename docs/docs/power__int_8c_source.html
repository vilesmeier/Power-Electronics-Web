70void POWER_1ph_Run(POWER_1PH *v, float32_t vol, float32_t cur)
71{
72 v->v = vol;
73 v->i = cur;
74 v->vNorm = fabsf(v->v);
75 v->iNorm = fabsf(v->i);
76 v->currSign = ( v->v > v->threshold) ? 1 : 0;
77 v->nSamples++;
78 v->vSum = v->vSum+v->vNorm;
79 v->vSqrSum = v->vSqrSum+(v->vNorm*v->vNorm);
80 v->vEma = v->vEma+(v->emaFilterMultiplier*(v->vNorm - v->vEma));
81 v->iSqrSum = v->iSqrSum+(v->iNorm*v->iNorm);
82 v->pSum = v->pSum+(v->i*v->v);
83 v->zcd=0;
84
85 // when positive zero pass is detected, indecating a full sine wave period
86 if((v->prevSign != v->currSign) && (v->currSign == 1))
87 {
88 //
89 // check if the nSamples are in the ball park of a real frequency
90 // that can be on the grid, this is done by comparing the nSamples
91 // with the max value and min value it can be for the
92 // AC Grid Connection these Max and Min are initialized by the
93 // user in the code
94 //
95 if(v->nSamplesMin < v->nSamples )
96 {
97 v->zcd=1;
98 v->inverse_nSamples = (1.0f)/(v->nSamples);
99 v->sqrt_inverse_nSamples = sqrtf(v->inverse_nSamples);
100 v->vAvg = (v->vSum*v->inverse_nSamples);
101 v->vRms = sqrtf(v->vSqrSum)*v->sqrt_inverse_nSamples;
102 v->iRms = sqrtf(v->iSqrSum)*v->sqrt_inverse_nSamples;
103 v->pRms = v->pSum*v->inverse_nSamples;
104 v->vaRms = v->vRms*v->iRms;
105 v->varRms = v->vaRms * v->vaRms - v->pRms * v->pRms;
106 if(v->varRms <= 0 )
107 {
108 v->varRms = 0;
109 }
110 else
111 {
112 v->varRms = sqrtf(v->varRms);
113 }
114 v->acFreq = (v->sampleFreq*v->inverse_nSamples);
115 v->powerFactor=v->pRms/v->vaRms;
116
117 v->jitterCount=0;
118 v->nSamples=0;
119 v->vSum=0;
120 v->vSqrSum=0;
121 v->iSqrSum=0;
122 v->pSum =0;
123 }
124 else
125 {
126 //
127 // otherwise it may be jitter ignore this reading
128 // but count the number of jitters you are getting
129 // but do not count to infinity as then when the grid comes back
130 // it will take too much time to wind down the jitter count
131 //
132 if(v->jitterCount<30)
133 {
134 v->jitterCount++;
135 }
136 v->nSamples=0;
137 }
138 }
139
140 if(v->nSamples>v->nSamplesMax || v->jitterCount>20)
141 {
142 //
143 // most certainly the AC voltage is not present
144 //
145 v->vRms = 0;
146 v->vAvg = 0;
147 v->vEma = 0;
148 v->acFreq=0;
149 v->iRms = 0;
150 v->pRms = 0;
151 v->vaRms =0;
152 v->powerFactor=0;
153
154 v->zcd=0;
155 v->vSum=0;
156 v->vSqrSum=0;
157 v->iSqrSum=0;
158 v->pSum=0;
159 v->nSamples=0;
160 v->jitterCount=0;
161 }
162 v->prevSign = v->currSign;
163}

166{
167 v->vRms = vrms;
168 v->iRms = irms;
169 v->powerFactor = pf;

232void POWER_3ph_Run(POWER_3PH *power)
233{
234 // getting current sample number
235 power->sample_num = power->sample_num + 1;
236
237 // rms for volate and current
238 power->va_sum = power->Va * power->Va + power->va_sum;
239 power->vb_sum = power->Vb * power->Vb + power->vb_sum;
240 power->vc_sum = power->Vc * power->Vc + power->vc_sum;
241 power->ia_sum = power->Ia * power->Ia + power->ia_sum;
242 power->ib_sum = power->Ib * power->Ib + power->ib_sum;
243 power->ic_sum = power->Ic * power->Ic + power->ic_sum;
244 power->p_sum = power->Va * power->Ia + power->Vb * power->Ib + power->Vc * power->Ic + power->p_sum;
245
246 // when sample == nsample, indecating get a sine period
247 // update output
248 if (power->nsample == power->sample_num)
249 {
250 // voltage rms
251 power->Varms = sqrtf(power->va_sum * power->nsample_inv);
252 power->Vbrms = sqrtf(power->vb_sum * power->nsample_inv);
253 power->Vcrms = sqrtf(power->vc_sum * power->nsample_inv);
254
255 // current rms
256 power->Iarms = sqrtf(power->ia_sum * power->nsample_inv);
257 power->Ibrms = sqrtf(power->ib_sum * power->nsample_inv);
258 power->Icrms = sqrtf(power->ic_sum * power->nsample_inv);
259
260 // power
261 power->P = power->nsample_inv * power->p_sum;
262 power->S = power->Varms * power->Iarms + power->Vbrms * power->Ibrms + power->Vcrms * power->Icrms;
263
264 power->PF = (power->S > 1e-6f) ? (power->P / power->S) : 0.0f;
265
266 float32_t temp = power->S * power->S - power->P * power->P;
267 power->Q = (temp > 0.0f) ? sqrtf(temp) : 0.0f;
268
269 // reset integrator
270 power->va_sum = 0;
271 power->vb_sum = 0;
272 power->vc_sum = 0;
273 power->ia_sum = 0;
274 power->ib_sum = 0;
275 power->ic_sum = 0;
276 power->p_sum = 0;
277
278 // reset sample counter
279 power->sample_num = 0;
280 }
281}

283void POWER_3ph_RMS_Run(POWER_3PH *power, float32_t Varms, float32_t Vbrms, float32_t Vcrms, float32_t Iarms, float32_t Ibrms, float32_t Icrms, float32_t PF)
284{
285 power->Varms = Varms;
286 power->Vbrms = Vbrms;
287 power->Vcrms = Vcrms;
288 power->Iarms = Iarms;
289 power->Ibrms = Ibrms;
290 power->Icrms = Icrms;
291
292 power->S = power->Varms * power->Iarms + power->Vbrms * power->Ibrms + power->Vcrms * power->Icrms;
293 power->P = power->PF * power->S;
294 power->Q = power->S * power->S - power->P *power->P;
295 if (power->Q < 0)
296 {
297 power->Q = 0;
298 }
299 power->Q = sqrtf(power->Q);
