2;;! \file source/fft/RFFT_f32_mag.asm
3;;!
4;;! \brief Real FFT magnitude calculations
5;;! \author David M. Alter
6;;! \date 11/20/14
7;;
8;; DESCRIPTION:
9;;
10;; Calculates the magnitude for the 32-bit floating point real FFT.
11;;
12;; HISTORY:
13;; 11/20/14 - optimized v1.40.00.00 function. (D. Alter)
14;;
15;; FUNCTIONS:
16;;
17;; void RFFT_f32_mag (RFFT_F32_STRUCT_Handle)
18;;
19;; where RFFT_F32_STRUCT_Handle is defined as:
20;; typedef RFFT_F32_STRUCT* RFFT_F32_STRUCT_Handle;
21;;
22;; and where RFFT_F32_STRUCT is a structure defined as:
23;;
24;; typedef struct {
25;; float *InBuf;
26;; float *OutBuf;
27;; float *CosSinBuf;
28;; float *MagBuf;
29;; float *PhaseBuf;
30;; uint16_t FFTSize;
31;; uint16_t FFTStages;
32;; } RFFT_F32_STRUCT;
33;;
34;; ASSUMPTIONS:
35;;
36;; * FFTSize must be a power of 2 (32, 64, 128, etc)
37;; * FFTSize must be greater or equal to 32
38;; * FFTStages must be ln(FFTSize)/ln(2)
39;; * InBuf, OutBuf, CosSinBuf are FFTSize in length
40;; * MagBuf is (FFTSize/2 + 1) in length
41;; * PhaseBuf is FFTSize/2 in length
42;; * RFFT_f32() or RFFT_f32u() has been called prior to this function
43;;
44;;
45;; ALGORITHM:
46;;
47;; Calculate the real FFT magnitude. The data is organized as follows in
48;; the output buffer after calculating FFT:
49;;
50;; +----------------------------+
51;; | R(0) = DC | 0
52;; | R(1) |
53;; | R(2) | Real Component
54;; | R(3) |
55;; . .
56;; . .
57;; . .
58;; . .
59;; | R(FFTSize/2) = fs/2 |
60;; | I(FFTSize/2 - 1) |
61;; . .
62;; . .
63;; . .
64;; . .
65;; | I(4) | Imaginary Component
66;; | I(3) |
67;; | I(2) |
68;; | I(1) | FFTSize - 1
69;; +----------------------------+
70;;
71;; CODE OPTIMIZATION:
72;;
73;; The loop was temporally unrolled, and then spatially unrolled once to
74;; eliminate FPU delay slots.
75;;
76;; BENCHMARK: 18 cycles/complex value + overhead
77;;
78;; On Entry: XAR4 = &RFFT_F32_STRUCT
79;;
80;;#############################################################################

90;RFFT_PhaseBuf .set (4*2)
91;RFFT_FFTSize .set (5*2)
92;RFFT_FFTStages .set (RFFT_FFTSize+1)
93;----------------------------------------------------------------------

101_RFFT_f32_mag:
102
103 MOV32 *SP++, R4H ; Save R4H (save-on-entry register)
104 MOV32 *SP++, R5H ; Save R5H (save-on-entry register)
105 MOV32 *SP++, R6H ; Save R6H (save-on-entry register)
106 MOV32 *SP++, R7H ; Save R7H (save-on-entry register)
107
108 MOVL XAR6, *+XAR4[2] ; XAR6 = Rptr = &OutBuf[0]
109 MOVL XAR5, *+XAR4[6] ; XAR5 = &MagBuf
110
111 ADDB XAR4, #10 ; Move the structure pointer up
112
113;--- Calculate pointer to imaginary part results
114 MOV ACC, *+XAR4[0] << 1 ; ACC = 2*FFTSize
115 ADDL ACC, XAR6 ; Add OutBuff address
116 MOVL XAR7, ACC ; XAR7 = Iptr = &OutBuf[last value]
117
118;--- Calculate main loop count
119 MOV AL, *+XAR4[0] ; AL = FFTSize
120 LSR AL, #2 ; AL = FFTSize/4
121 SUBB AL, #2 ; mag[0:1] handled seperately, and "N-1"
122 ; loop count so subtract 2
123;----------------------------------------------------------------------
124; Begin magnitude calculations
125;----------------------------------------------------------------------
126
127;--- Mag[0] is handled special since no Im part
128 MOV32 R0H, *XAR6++ ; R0H = Re[0]
129 ABSF32 R0H, R0H ; R0H = abs(Re[0])
130 MOV32 *XAR5++, R0H ; Store mag[0]
131
132;--- Must also compute Mag[1] seperately to keep an even number of mag
133; computations in the main loop
134
135 ;--- Compute magnitude squared
136 MOV32 R0H, *XAR6++ ; R0H = Re[1]
137 MPYF32 R0H, R0H, R0H ; R0H = Re[1]*Re[1]
138 || MOV32 R1H, *--XAR7 ; R1H = Im[1]
139 MPYF32 R1H, R1H, R1H ; R1H = Im[1]*Im[1]
140 NOP
141 ADDF32 R0H, R0H, R1H ; R0H = (Re[1]*Re[1]) + (Im[1]*Im[1])
142 NOP
143
144 ;--- Do the sqrt
145 EISQRTF32 R1H, R0H ; R1H = Ye[1] = Estimate(1/sqrt(X[1]))
146 MPYF32 R2H, R0H, #0.5 ; R2H = X[1]*0.5
147 MPYF32 R3H, R1H, R1H ; R3H = Ye[1]*Ye[1]
148 NOP
149 MPYF32 R3H, R3H, R2H ; R3H = Ye[1]*Ye[1]*X[1]*0.5
150 NOP
151 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[1]*Ye[1]*X[1]*0.5
152 NOP
153 MPYF32 R1H, R1H, R3H ; R2H = Ye[1] = Ye[1]*(1.5 - Ye[1]*Ye[1]*X[1]*0.5)
154 NOP
155 MPYF32 R3H, R1H, R2H ; R3H = Ye[1]*X[1]*0.5
156 NOP
157 MPYF32 R3H, R1H, R3H ; R3H = Ye[1]*Ye[1]*X[1]*0.5
158 NOP
159 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[1]*Ye[1]*X[1]*0.5
160 CMPF32 R0H, #0.0 ; Check if X[1] == 0
161 MPYF32 R1H, R1H, R3H ; R2H = Ye[1] = Ye[1]*(1.5 - Ye[1]*Ye[1]*X[1]*0.5)
162 NOP
163 MOV32 R1H, R0H, EQ ; If X is zero, change the Ye[1] estimate to 0
164 MPYF32 R0H, R0H, R1H ; R0H = Y[1] = X[1]*Ye[1] = sqrt(X[1])
165 NOP
166
167 ;--- Store the result
168 MOV32 *XAR5++,R0H ; Store mag[1]
169
170;--- Main loop
171
172_rfft_f32_calc_mag:
173
174 RPTB _rfft_f32_mag_end, @AL
175
176 ;--- Compute magnitude squared
177
178 MOV32 R0H, *XAR6++ ; R0H = Re[i]
179 MPYF32 R0H, R0H, R0H ; R0H = Re[i]*Re[i]
180 || MOV32 R1H, *--XAR7 ; R1H = Im[i]
181
182 MOV32 R4H, *XAR6++ ; R4H = Re[i+1]
183 MPYF32 R4H, R4H, R4H ; R4H = Re[i+1]*Re[i+1]
184 || MOV32 R5H, *--XAR7 ; R5H = Im[i+1]
185
186 MPYF32 R1H, R1H, R1H ; R1H = Im[i]*Im[i]
187 MPYF32 R5H, R5H, R5H ; R5H = Im[i+1]*Im[i+1]
188
189 ADDF32 R0H, R1H, R0H ; R0H = (Re[i]*Re[i]) + (Im[i]*Im[i])
190 ADDF32 R4H, R5H, R4H ; R4H = (Re[i+1]*Re[i+1]) + (Im[i+1]*Im[i+1])
191
192 ;--- Do the sqrt
193
194 EISQRTF32 R1H, R0H ; R1H = Ye[i] = Estimate(1/sqrt(X[i]))
195 EISQRTF32 R5H, R4H ; R5H = Ye[i+1] = Estimate(1/sqrt(X[i+1]))
196
197 MPYF32 R2H, R0H, #0.5 ; R2H = X[i]*0.5
198 MPYF32 R6H, R4H, #0.5 ; R6H = X[i+1]*0.5
199
200 MPYF32 R3H, R1H, R1H ; R3H = Ye[i]*Ye[i]
201 MPYF32 R7H, R5H, R5H ; R7H = Ye[i+1]*Ye[i+1]
202
203 MPYF32 R3H, R3H, R2H ; R3H = Ye[i]*Ye[i]*X*0.5
204 MPYF32 R7H, R7H, R6H ; R7H = Ye[i+1]*Ye[i+1]*X[i+1]*0.5
205
206 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[i]*Ye[i]*X[i]*0.5
207 SUBF32 R7H, #1.5, R7H ; R7H = 1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5
208
209 MPYF32 R1H, R1H, R3H ; R1H = Ye[i] = Ye[i]*(1.5 - Ye[i]*Ye[i]*X[i]*0.5)
210 MPYF32 R5H, R5H, R7H ; R6H = Ye[i+1] = Ye[i+1]*(1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5)
211
212 MPYF32 R3H, R1H, R2H ; R3H = Ye[i]*X[i]*0.5
213 MPYF32 R7H, R5H, R6H ; R7H = Ye[i+1]*X[i+1]*0.5
214
215 MPYF32 R3H, R1H, R3H ; R3H = Ye[i]*Ye[i]*X[i]*0.5
216 MPYF32 R7H, R5H, R7H ; R7H = Ye[i+1]*Ye[i+1]*X[i+1]*0.5
217
218 SUBF32 R3H, #1.5, R3H ; R3H = 1.5 - Ye[i]*Ye[i]*X*0.5
219 CMPF32 R0H, #0.0 ; Check if X[i] == 0
220 MPYF32 R1H, R1H, R3H ; R1H = Ye[i] = Ye[i]*(1.5 - Ye[i]*Ye[i]*X[i]*0.5)
221
222 SUBF32 R7H, #1.5, R7H ; R7H = 1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5
223 MOV32 R1H, R0H, EQ ; If X[i] is zero, change the Ye[i] estimate to 0
224 CMPF32 R4H, #0.0 ; Check if X[i+1] == 0
225 MPYF32 R5H, R5H, R7H ; R5H = Ye[i+1] = Ye[i+1]*(1.5 - Ye[i+1]*Ye[i+1]*X[i+1]*0.5)
226
227 MPYF32 R0H, R0H, R1H ; R0H = Y[i] = X[i]*Ye[i] = sqrt(X[i])
228 MOV32 R5H, R4H, EQ ; If X[i+1] is zero, change the Ye[i+1] estimate to 0
229 MPYF32 R4H, R4H, R5H ; R4H = Y[i+1] = X[i+1]*Ye[i+1] = sqrt(X[i+1])
230
231 ;--- Store the result
232 MOV32 *XAR5++,R0H ; Store mag[i]
233 MOV32 *XAR5++,R4H ; Store mag[i+1]
234
235_rfft_f32_mag_end:
236
237;--- Mag[N/2+1] is handled special since no Im part
238 MOV32 R0H, *XAR6++ ; R0H = Re[N/2+1]
239 ABSF32 R0H, R0H ; R0H = abs(Re[N/2+1])
240 MOV32 *XAR5++, R0H ; Store mag[N/2+1]
241
242;--- Finish up
243 MOV32 R7H, *--SP ; Restore R7H
244 MOV32 R6H, *--SP ; Restore R6H
245 MOV32 R5H, *--SP ; Restore R5H
246 MOV32 R4H, *--SP ; Restore R4H
247 LRETR
