2;;! \file source/fft/RFFT_f32.asm
3;;!
4;;! \brief Real FFT
5;;! \author C2000
6;;
7;; DESCRIPTION:
8;;
9;; This function computes a real FFT. The input buffer must be aligned to
10;; a multiple of the FFT size. If it is not aligned then the output buffer
11;; will yield invalid results. If you do not wish to align the input buffer
12;; then use the RFFT_f32u function. Using this function will reduce cycle
13;; performance of the algorithm.
14;;
15;; FUNCTIONS:
16;;
17;; void RFFT_f32(RFFT_F32_STRUCT_Handle)
18;;
19;; where RFFT_F32_STRUCT_Handle is defined as:
20;; typedef RFFT_F32_STRUCT* RFFT_F32_STRUCT_Handle;
21;;
22;; and where RFFT_F32_STRUCT is a structure defined as:
23;;
24;; typedef struct {
25;; float *InBuf;
26;; float *OutBuf;
27;; float *CosSinBuf;
28;; float *MagBuf;
29;; float *PhaseBuf;
30;; uint16_t FFTSize;
31;; uint16_t FFTStages;
32;; } RFFT_F32_STRUCT;
33;;
34;; ASSUMPTIONS:
35;;
36;; * FFTSize must be a power of 2 (32, 64, 128, etc)
37;; * FFTSize must be greater or equal to 32
38;; * FFTStages must be log2(FFTSize)
39;; * InBuf, OutBuf, CosSinBuf are FFTSize in length
40;; * MagBuf and PhaseBuf are FFTSize/2 in length
41;; * MagBuf and PhaseBuf are not used by this function.
42;; They are only used by the magitude and phase calculation functions.
43;;
44;; ALGORITHM:
45;;
46;; 1) Bit reverse input data and calculate stages 1, 2 & 3:
47;;
48;; In Buf (read in bit reverse order) Out Buf
49;; +----+ +----+
50;; | I1 | (((I1 + I2) + (I3 + I4)) + ((I5 + I6) + (I7 + I8)))/8 -> | I1'|
51;; | I2 | ((I1 - I2) + COS*((I5 - I6) + (I8 - I7)) )/8 -> | I2'|
52;; | I3 | ((I1 + I2) - (I3 + I4) )/8 -> | I3'|
53;; | I4 | ((I1 - I2) - COS*((I5 - I6) + (I8 - I7)) )/8 -> | I4'|
54;; | I5 | (((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)))/8 -> | I5'|
55;; | I6 | (COS*((I8 - I7) - (I5 - I6)) - (I4 - I3) )/8 -> | I6'|
56;; | I7 | ((I7 + I8) - (I5 + I6) )/8 -> | I7'|
57;; | I8 | (COS*((I8 - I7) - (I5 - I6)) + (I4 - I3) )/8 -> | I8'|
58;; .
59;; .
60;; \|/
61;; Repeat above FFTSize/8 (i.e. if FFTSize = 1024, Repeat = 128 times)
62;;
63;; Note: COS = COS( 1*2*PI/8) = SIN( 1*2*PI/8)
64;;
65;; 2) Calculate stages 4 and up:
66;;
67;;
68;; Out Buf 4 5 6 7 8 9 10 <- Stages
69;; +- +---------+ - - - - - - --
70;; | | Y1 | 0 0 0 0 0 0 0 <- Y1 + Y3
71;; | |---------|
72;; | | X(I1) | 1 1 1 1 1 1 1 <- X(I1) + [X(I3)*COS + X(I4)*SIN]
73;; | |---------|
74;; | | . |
75;; | | . | 3 7 15 31 63 127 255 <- Inner Loop Repeat Times
76;; | | \./ |
77;; | I |---------|
78;; | N | Y2 | 4 8 16 32 64 128 256 <- Y2
79;; | N |---------|
80;; | E | /.\ |
81;; | R | . |
82;; | | . |
83;; | L |---------|
84;; | O | X(I2) | 7 15 31 63 127 255 511 <- X(I1) - [X(I3)*COS + X(I4)*SIN]
85;; | O |---------|
86;; | P | Y3 | 8 16 32 64 128 256 512 <- Y1 - Y3
87;; | |---------|
88;; | | X(I3) | 9 17 33 65 129 257 513 <- [X(I4)*COS - X(I3)*SIN] - X(I2)
89;; | |---------|
90;; | | . |
91;; | | . |
92;; | | \./ |
93;; | |---------|
94;; | | Y4 |12 24 48 96 192 384 768 <- -Y4
95;; | |---------|
96;; | | /.\ |
97;; | | . |
98;; | | . |
99;; | |---------|
100;; | | X(I4) |15 31 63 127 255 511 1023 <- [X(I4)*COS - X(I3)*SIN] + X(I2)
101;; +- |---------|
102;; | |16 32 64 128 256 512 1024
103;; -- --- --- --- --- --- ----
104;; . 1 <- Outer Loop Repeat Times (16 FFT)
105;; . 2 1 <- Outer Loop Repeat Times (32 FFT)
106;; . 4 2 1 <- Outer Loop Repeat Times (64 FFT)
107;; . 8 4 2 1 <- Outer Loop Repeat Times (128 FFT)
108;; . 16 8 4 2 1 <- Outer Loop Repeat Times (256 FFT)
109;; . 32 16 8 4 2 1 <- Outer Loop Repeat Times (512 FFT)
110;; . 64 32 16 8 4 2 1 <- Outer Loop Repeat Times (1024 FFT)
111;; .
112;; .
113;; \|/

184;===========================================================================
185; Function: void rfft_f32_Stages1and2and3andBitReverse(RFFT_F32_STRUCT_Handle)
186;===========================================================================
187;---------------------------------------------------------------------------
188;
189; DESCRIPTION:
190;
191; This function bit reverses the input and computes stages 1, 2 and 3
192;
193; ON ENTRY:
194;
195; XAR4 = Starting address of the RFFT_F32_STRUCT structure
196;
197; REGISTER USAGE:
198;
199; AR0, XAR1, XAR2, XAR4, XAR5, AR3, AR6, XAR7, ACC,
200; R0H, R1H, R2H, R3H, R4H, R5H, R6H, R7H
201;
202; On Exit:
203;
204; RFFT_F32_STRUCT OutBuf contains the computed result
205;
206;---------------------------------------------------------------------------
207;offsets to the stack which holds the bit reversed elements
208;---------------------------------------------------------------------------
209
210I1 .set 06H
211I2 .set 08H
212I3 .set 0AH
213I4 .set 0CH
214I5 .set 0EH
215I6 .set 10H
216I7 .set 12H
217I8 .set 14H
218cossinBuf .set 04H
219;------------------------------------------------------------------------------
220;Offset to the stack which holds intermediate results of FFT computation
221;------------------------------------------------------------------------------
222 .global _rfft_f32_Stages1and2and3andBitReverse
223 .text
224
225_rfft_f32_Stages1and2and3andBitReverse:
226;----------------------------------------------------------------------
227; Save all save-on-entry registers used
228;----------------------------------------------------------------------
229 PUSH XAR1
230 PUSH XAR2
231 PUSH XAR3
232 MOV32 *SP++,R4H
233 MOV32 *SP++,R5H
234 MOV32 *SP++,R6H
235 MOV32 *SP++,R7H
236 ADDB SP,#14h
237
238 MOVL XAR2,*+XAR4[0] ; &Inbuf
239 MOVL XAR5,*+XAR4[2] ; &Outbuf
240 MOVL XAR3,*+XAR4[2]
241 ADDB XAR3,#8 ; &Outbuf[4]
242 MOVL XAR7,*+XAR4[4] ; &CosSinbuf
243
244 MOV AR0,#0Ah
245 MOV AH,*+XAR4[AR0] ; FFT SIZE
246 MOV AR0,AH
247 LSR AH,3
248 SUBB AH,#1 ; (Size / 8) - 1
249 MOVL XAR1,#0000h ; index if memory is not aligned
250
251 RPTB _rfft_32_Last, AH
252;------------------------------------------------------------------------------
253; Input buffer must be aligned for this code
254;------------------------------------------------------------------------------
255 NOP *,ARP2
256
257 MOVL ACC,*BR0++
258 MOVL *-SP[I1],ACC ;I1
259
260 MOVL ACC,*BR0++
261 MOVL *-SP[I2],ACC ;I2
262
263 MOVL ACC,*BR0++
264 MOVL *-SP[I3],ACC ;I3
265
266 MOVL ACC,*BR0++
267 MOVL *-SP[I4],ACC ;I4
268
269 MOVL ACC,*BR0++
270 MOVL *-SP[I5],ACC ;I5
271
272 MOVL ACC,*BR0++
273 MOVL *-SP[I6],ACC ;I6
274
275 MOVL ACC,*BR0++
276 MOVL *-SP[I7],ACC ;I7
277
278 MOVL ACC,*BR0++
279 MOVL *-SP[I8],ACC ;I8
280
281;------------------------------------------------------------------------------
282; Computations for stages 1 2, and 3
283; OutBufIndex++ = (I1 + I2) + (I3 + I4) + (I5 + I6) + (I7 + I8); (A) <- XAR5
284; OutBufIndex++ = (I1 - I2) + COS x ((I5 - I6) + (I8 - I7)); (B)
285; OutBufIndex++ = (I1 + I2) - (I3 + I4); (C)
286; OutBufIndex++ = (I1 - I2) - COS x ((I5 - I6) + (I8 - I7)); (D)
287; OutBufIndex++ = ((I1 + I2) + (I3 + I4)) - ((I5 + I6) + (I7 + I8)); (E) <- XAR3
288; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) - (I4 - I3); (F)
289; OutBufIndex++ = (I7 + I8) - (I5 + I6); (G)
290; OutBufIndex++ = COS x ((I8 - I7) - (I5 - I6)) + (I4 - I3); (H)
291;------------------------------------------------------------------------------
292 MOV32 R0H, *-SP[I8] ; R0H = I8
293 MOV32 R1H, *-SP[I7] ; R1H = I7
294 SUBF32 R2H, R0H, R1H ; R2H = I8-I7
295 || MOV32 R3H, *-SP[I5] ; R3H = I5
296 ADDF32 R0H, R1H, R0H ; R0H = I7+I8
297 || MOV32 R4H, *-SP[I6] ; R4H = I6
298 SUBF32 R1H, R3H, R4H ; R1H = I5-I6
299 || MOV32 R5H, *-SP[I1] ; R5H = I1
300 ADDF32 R3H, R3H, R4H ; R3H = I5+I6
301 ADDF32 R4H, R1H, R2H ; R4H = (I5-I6) + (I8-I7)
302 || MOV32 R6H,*XAR7 ; R6H = COS
303 SUBF32 R7H, R0H, R3H ; R7H = (I7+I8) - (I5+I6) (G)
304 MPYF32 R4H, R6H, R4H ; R4H = COS x ((I5-I6) + (I8-I7))
305 || SUBF32 R2H, R2H, R1H ; R2H = (I8-I7) - (I5-I6)
306 ADDF32 R0H, R3H, R0H ; R0H = (I5+I6) + (I7+I8)
307 || MOV32 *+XAR3[4], R7H ; store G
308 MPYF32 R2H, R6H, R2H ; R2H = COS x ((I8-I7) - (I5-I6))
309 || MOV32 R7H, *-SP[I2] ; R7H = I2
310 SUBF32 R6H, R5H, R7H ; R6H = I1-I2
311 ADDF32 R5H, R5H, R7H ; R5H = I1+I2
312 || MOV32 R1H, *-SP[I4] ; R1H = I4
313 ADDF32 R3H, R6H, R4H ; R3H = (I1-I2) + COS x ((I5-I6) + (I8-I7)) (B)
314 || MOV32 R7H, *-SP[I3] ; R7H = I3
315 SUBF32 R6H, R6H, R4H ; R6H = (I1-I2) - COS x ((I5-I6) + (I8-I7)) (D)
316 SUBF32 R3H, R1H, R7H ; R3H = I4-I3
317 || MOV32 *+XAR5[2], R3H ; store B
318 ADDF32 R4H, R7H, R1H ; R4H = I3+I4
319 || MOV32 *+XAR5[6], R6H ; store D
320 SUBF32 R1H, R2H, R3H ; R1H = COS x ((I8-I7) - (I5-I6)) - (I4-I3) (F)
321 ADDF32 R2H, R2H, R3H ; R2H = COS x ((I8-I7) - (I5-I6)) + (I4-I3) (H)
322 ADDF32 R6H, R5H, R4H ; R6H = (I1+I2) + (I3+I4)
323 || MOV32 *+XAR3[2], R1H ; save F
324 SUBF32 R1H, R5H, R4H ; R1H = (I1+I2) - (I3+I4) (C)
325 || MOV32 *+XAR3[6], R2H ; save H
326 SUBF32 R3H, R6H, R0H ; R3H = (I1+I2) + (I3+I4) - ((I5+I6) + (I8+I7)) (E)
327 ADDF32 R0H, R6H, R0H ; R0H = (I1+I2) + (I3+I4) + (I5+I6) + (I8+I7) (A)
328 || MOV32 *+XAR5[4], R1H ; store C
329 MOV32 *+XAR3[0], R3H ; store E
330 MOV32 *+XAR5[0], R0H ; store A
331 ADDB XAR5, #16
332 ADDB XAR3, #16
333
334
335_rfft_32_Last:
336 SUBB SP,#14h
337 MOV32 R7H,*--SP
338 MOV32 R6H,*--SP
339 MOV32 R5H,*--SP
340 MOV32 R4H,*--SP
341 POP XAR3
342 POP XAR2
343 POP XAR1
344 LRETR

346;===========================================================================
347; Function: void rfft_f32_Stages4andUp(RFFT_F32_STRUCT_Handle)
348;===========================================================================
349;===========================================================================
350;
351; ON ENTRY:
352;
353; XAR4 = Starting address of the RFFT_F32_STRUCT structure
354;
355; REGISTER USAGE:
356;
357; AR0, XAR1, XAR2, XAR4, XAR5, AR3, AR6, XAR7, ACC,
358; R0H, R1H, R2H, R3H, R4H, R5H, R6H, R7H
359;
360; On Exit:
361;
362; RFFT_F32_STRUCT OutBuf contains the computed result
363;
364;---------------------------------------------------------------------------
365
366
367;----------------------------------------------------------------------
368; FFT_REAL structure offsets from XAR4
369;----------------------------------------------------------------------
370RFFT_InBuf .set (0*2)
371RFFT_OutBuf .set (1*2)
372RFFT_CosSinBuf .set (2*2)
373RFFT_MagBuf .set (3*2)
374RFFT_PhaseBuf .set (4*2)
375RFFT_FFTSize .set (5*2)
376RFFT_FFTStages .set (RFFT_FFTSize+1)
377;----------------------------------------------------------------------
378; Offsets for local variables stored on the stack
379;----------------------------------------------------------------------
380I1OFFSET .set 1 ; 16-bit
381I2I4OFFSET .set 2 ; 16-bit
382IDX_I .set 3 ; 16-bit
383IDX_K .set 4 ; 16-bit
384IDX_L .set 5 ; 16-bit
385IDX_SZBL .set 6 ; 16-bit
386IDX_M .set 8 ; 16-bit
387TEMPX1 .set (IDX_M+2*1) ; 32-bit
388OUTBUF .set 12
389COSSINBUF .set 14
390FRMSZ .set COSSINBUF
391
392 .global _rfft_f32_Stages4andUp
393 .text
394
395_rfft_f32_Stages4andUp:
396
397;----------------------------------------------------------------------
398; Save all save-on-entry registers used
399;----------------------------------------------------------------------
400 PUSH XAR1
401 PUSH XAR2
402 PUSH XAR3
403 MOV32 *SP++,R4H
404 MOV32 *SP++,R5H
405 MOV32 *SP++,R6H
406 MOV32 *SP++,R7H
407 ADDB SP,#FRMSZ
408;----------------------------------------------------------------------
409; Initialize loop counters and offsets
410; Note: Index k has changed from 3 to 2 due to loop unrolling.
411;----------------------------------------------------------------------
412 MOV *-SP[I1OFFSET],#26 ; I1 & I3 pointer offset
413 MOV *-SP[I2I4OFFSET],#38 ; I2 & I4 pointer offset
414 MOV *-SP[IDX_K],#(2-1) ; Inner loop count
415 MOV *-SP[IDX_L],#(16*2) ; 16 32-bit values calculated each
416 ; inner loop for the first stage
417;----------------------------------------------------------------------
418; Check the FFTStages and compute stage, outer loop count
419;----------------------------------------------------------------------
420 MOVB XAR1,#RFFT_FFTStages
421 MOV AL,*+XAR4[AR1]
422 SUB AL,#3 ; If (fft->FFTStages <= 3) exit
423 B _rfft_f32_End,LEQ
424 MOV *-SP[IDX_I],AL ; IDX_I = FFTStages - 3 = i for
425 ; stages loop control
426 MOVB XAR0,#RFFT_FFTSize
427 MOV AL,*+XAR4[AR0]
428 ASR AL,#4
429 MOV *-SP[IDX_SZBL],AL ; IDX_SZBL = FFTSize/16
430 DEC AL
431 MOV AR6, AL ; IDX_M = FFTSize/16-1 = m for
432 ; outer loop control
433;----------------------------------------------------------------------
434; Initialize pointers
435;----------------------------------------------------------------------
436 .asg AR0, Offset1
437 .asg AR1, Offset2
438 .asg XAR2, I1ptr
439 .asg AR3, InnerLoopCnt
440 .asg XAR4, I2ptr
441 .asg XAR5, I4ptr
442 .asg AR6, OuterLoopCnt
443 .asg XAR7, COSptr
444 .asg XAR7, SINptr
445
446
447 MOVL ACC, *+XAR4[RFFT_CosSinBuf]
448 MOVL *-SP[COSSINBUF], ACC
449 ADDB ACC, #8
450 MOVL *-SP[TEMPX1], ACC ; TEMPX1 = &CosSinBuf[4]
451
452
453 MOVL I1ptr, *+XAR4[RFFT_OutBuf] ; I1ptr = &OutBuf[0]
454 MOVL *-SP[OUTBUF], I1ptr
455 MOVL I2ptr, *+XAR4[RFFT_OutBuf]
456 MOVL I4ptr, *-SP[OUTBUF]
457
458 ADDB I2ptr, #16 ; I2ptr = &OutBuf[8]
459 MOV Offset1,#16 ; I1ptr[Offset1] = &OutBuf[8]
460 MOV Offset2,#18 ; I1ptr[Offset2] = &OutBuf[9]
461 ADDB I4ptr, #32 ; I4ptr = &OutBuf[16]
462 MOV InnerLoopCnt, #(2-1) ; Inner loop count for 4th stage
463;----------------------------------------------------------------------
464; Stage Loop: For each stage 4 and up:
465;----------------------------------------------------------------------
466_rfft_f32_StageLoop:
467;----------------------------------------------------------------------
468; Outer Loop: IDX_M
469; Stage 4: M = 1 to FFTSize/16
470; Stage 5: M = 1 to FFTSize/32
471; Stage 6: M = 1 to FFTSize/64 etc..
472;
473; Calculate the following M times:
474;
475; I1 = Y1 = *I1ptr;
476; I3 = *I3ptr;
477; *I1ptr++ = Y = (I1 + I3) = Y1 + Y3
478; *I3ptr++ = Y3 = (I1 - I3) = Y1 - Y3
479; for(j=1; j <= k; j ++)
480; {
481; I1 = *I1ptr;
482; I2 = *I2ptr;
483; I3 = *I3ptr;
484; I4 = *I4ptr;
485; COS = COSptr++;
486; SIN = COSptr++;
487; I1ptr++ = I1 + I3xCOS + I4xSIN; (A)
488; I2ptr-- = I1 - (I3xCOS + I4xSIN); (B)
489; I3ptr++ = I4xCOS - I3xSIN - I2; (C)
490; I4ptr-- = I4xCOS - I3xSIN + I2; (D)
491; }
492; Y4 = -Y4:
493;----------------------------------------------------------------------
494_rfft_f32_OuterLoop:
495 MOVL COSptr, *-SP[TEMPX1]
496 MOV32 R1H, *I1ptr ; R1H = Y1
497 MOV32 R3H, *I2ptr ; R3H = Y3
498 ADDF32 R2H, R1H, R3H ; R2H = Y1 + Y3
499 || MOV32 R0H, *COSptr++ ; R0H = COS (1)
500 SUBF32 R4H, R1H, R3H ; R4H = (Y1 - Y3)
501 || MOV32 R3H, *+I1ptr[Offset2] ; R3H = I3 (1)
502 MOV32 *I1ptr++, R2H ; I1 = (Y1 + Y3)
503 MPYF32 R1H, R3H, R0H ; R1H = I3*COS (1)
504 || MOV32 R6H, *--I4ptr ; R6H = I4 (1)
505 MPYF32 R2H, R6H, R0H ; R2H = I4*COS (1)
506 || MOV32 R7H, *SINptr++ ; R7H = SIN (1)
507 MPYF32 R3H, R3H, R7H ; R3H = I3*SIN (1)
508 || MOV32 *I2ptr,R4H ; Y3 = Y1 - Y3
509
510 RPTB _rfft_f32_InnerLoop, InnerLoopCnt
511 MPYF32 R4H, R6H, R7H ; R4H = I4*SIN (1)
512 || MOV32 R6H, *--I2ptr ; R6H = I2 (1)
513 SUBF32 R7H, R2H, R3H ; R7H = I4*COS - I3*SIN (1)
514 || MOV32 R0H, *COSptr++ ; R0H = COS (2)
515 ADDF32 R3H, R1H, R4H ; R3H = I3*COS + I4*SIN (1)
516 || MOV32 R5H, *+I1ptr[Offset2] ; R5H = I3 (2)
517 SUBF32 R2H, R7H, R6H ; R2H = I4*COS - I3*SIN - I2 (C1)
518 || MOV32 R4H, *I1ptr ; R4H = I1 (1)
519 ADDF32 R7H, R7H, R6H ; R7H = I4*COS - I3*SIN + I2 (D1)
520 || MOV32 R6H, *--I4ptr ; R6H = I4 (2)
521 ADDF32 R2H, R4H, R3H ; R2H = I1 + (I3*COS + I4*SIN) (A1)
