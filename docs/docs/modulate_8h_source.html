115static inline void Modulate_SVPWM_Run(MOD* mod, float32_t ua, float32_t ub, float32_t uc)
116{
117 float32_t max, min;
118
119 ua = (ua+ 1.0f) * 0.5f;
120 ub = (ub + 1.0f) * 0.5f;
121 uc = (uc + 1.0f) * 0.5f;
122
123 // 计算三相最大值、最小值
124 max = (ua >ub) ? ua :ub;
125 max = (max > uc) ? max : uc;
126 min = (ua <ub) ? ua :ub;
127 min = (min < uc) ? min : uc;
128
129 mod->comp1 = (ua + (1.0f - max - min) * 0.5f) * mod->tbprd;
130 mod->comp2 = (ub + (1.0f - max - min) * 0.5f) * mod->tbprd;
131 mod->comp3 = (uc + (1.0f - max - min) * 0.5f) * mod->tbprd;

143static inline void Modulate_DPWM1_Run(MOD* mod, float32_t ua, float32_t ub, float32_t uc)
144{
145 float32_t max, min, u0p, u0n, u0;
146
147 // 计算三相最大值、最小值
148 max = (ua > ub) ? ua : ub;
149 max = (max > uc) ? max : uc;
150 min = (ua < ub) ? ua : ub;
151 min = (min < uc) ? min : uc;
152
153 // 产生零序分量
154 // 最大值钳位至 +1
155 u0p = 1.0f - max;
156 // 最小值钳位至 -1
157 u0n = -1.0f - min;
158
159 // 二者取较小的一项作为零序分量
160 u0 = (fabsf(u0p) <= fabsf(u0n)) ? u0p : u0n;
161
162 // 零序分量注入
163 ua = ua + u0;
164 ub = ub + u0;
165 uc = uc + u0;
166
167 // 调制
168 mod->comp1 = (ua + 1.0f) * mod->half_tbprd;
169 mod->comp2 = (ub + 1.0f) * mod->half_tbprd;
170 mod->comp3 = (uc + 1.0f) * mod->half_tbprd;
171}

246static inline void Modulate_GDPWM_Run(MOD* mod, float32_t ua, float32_t ub, float32_t uc, float32_t pf_angle)
247{
248 float32_t u0, uaf, ubf, ucf, ua_abs, ub_abs, uc_abs;
249
250 // abc坐标系旋转, 转过角度为功率因数角
251 uaf = ua * cosf(pf_angle) - (uc - ub) * 0.5773502692f * sinf(pf_angle);
252 ubf = ub * cosf(pf_angle) + ((uc - ub) * 0.2886751346f - 0.8660254038f * ua) * sinf(pf_angle);
253 ucf = -uaf - ubf;
254 ua_abs = fabsf(uaf);
255 ub_abs = fabsf(ubf);
256 uc_abs = fabsf(ucf);
257
258 // 判断钳位相, 产生零序分量
259 if ((ua_abs >= ub_abs) && (ua_abs >= uc_abs))
260 {
261 if (ua >= 0.0f)
262 {
263 u0 = 1.0f - ua;
264 }
265 else
266 {
267 u0 = -1.0f - ua;
268 }
269 }
270
271 else if ((ub_abs >= ua_abs) && (ub_abs >= uc_abs))
272 {
273 if (ub >= 0.0f)
274 {
275 u0 = 1.0f - ub;
276 }
277 else
278 {
279 u0 = -1.0f - ub;
280 }
281 }
282
283 else if ((uc_abs >= ua_abs) && (uc_abs >= ub_abs))
284 {
285 if (uc >= 0.0f)
286 {
287 u0 = 1.0f - uc;
288 }
289 else
290 {
291 u0 = -1.0f - uc;
292 }
293 }
294
295 // 零序分量注入
296 ua = ua + u0;
297 ub = ub + u0;
298 uc = uc + u0;
299
300 // 计算COMP寄存器值
301 mod->comp1 = (ua + 1.0f) * mod->half_tbprd;
302 mod->comp2 = (ub + 1.0f) * mod->half_tbprd;
303 mod->comp3 = (uc + 1.0f) * mod->half_tbprd;
304}
