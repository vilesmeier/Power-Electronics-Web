112static inline void Modulate_SVPWM_Run(MOD* mod, float32_t ua, float32_t ub, float32_t uc)
113{
114 float32_t max, min;
115
116 ua = (ua+ 1.0f) * 0.5f;
117 ub = (ub + 1.0f) * 0.5f;
118 uc = (uc + 1.0f) * 0.5f;
119
120 // 计算三相最大值、最小值
121 max = (ua >ub) ? ua :ub;
122 max = (max > uc) ? max : uc;
123 min = (ua <ub) ? ua :ub;
124 min = (min < uc) ? min : uc;
125
126 mod->comp1 = (ua + (1.0f - max - min) * 0.5f) * mod->tbprd;
127 mod->comp2 = (ub + (1.0f - max - min) * 0.5f) * mod->tbprd;
128 mod->comp3 = (uc + (1.0f - max - min) * 0.5f) * mod->tbprd;

140static inline void Modulate_DPWM1_Run(MOD* mod, float32_t ua, float32_t ub, float32_t uc)
141{
142 float32_t max, min, u0p, u0n, u0;
143
144 // 计算三相最大值、最小值
145 max = (ua > ub) ? ua : ub;
146 max = (max > uc) ? max : uc;
147 min = (ua < ub) ? ua : ub;
148 min = (min < uc) ? min : uc;
149
150 // 产生零序分量
151 // 最大值钳位至 +1
152 u0p = 1.0f - max;
153 // 最小值钳位至 -1
154 u0n = -1.0f - min;
155
156 // 二者取较小的一项作为零序分量
157 u0 = (fabsf(u0p) <= fabsf(u0n)) ? u0p : u0n;
158
159 // 零序分量注入
160 ua = ua + u0;
161 ub = ub + u0;
162 uc = uc + u0;
163
164 // 调制
165 mod->comp1 = (ua + 1.0f) * mod->half_tbprd;
166 mod->comp2 = (ub + 1.0f) * mod->half_tbprd;
167 mod->comp3 = (uc + 1.0f) * mod->half_tbprd;
168}

243static inline void Modulate_GDPWM_Run(MOD* mod, float32_t ua, float32_t ub, float32_t uc, float32_t pf_angle)
244{
245 float32_t u0, uaf, ubf, ucf, ua_abs, ub_abs, uc_abs;
246
247 // abc坐标系旋转, 转过角度为功率因数角
248 uaf = ua * cosf(pf_angle) - (uc - ub) * 0.5773502692f * sinf(pf_angle);
249 ubf = ub * cosf(pf_angle) + ((uc - ub) * 0.2886751346f - 0.8660254038f * ua) * sinf(pf_angle);
250 ucf = -uaf - ubf;
251 ua_abs = fabsf(uaf);
252 ub_abs = fabsf(ubf);
253 uc_abs = fabsf(ucf);
254
255 // 判断钳位相, 产生零序分量
256 if ((ua_abs >= ub_abs) && (ua_abs >= uc_abs))
257 {
258 if (ua >= 0.0f)
259 {
260 u0 = 1.0f - ua;
261 }
262 else
263 {
264 u0 = -1.0f - ua;
265 }
266 }
267
268 else if ((ub_abs >= ua_abs) && (ub_abs >= uc_abs))
269 {
270 if (ub >= 0.0f)
271 {
272 u0 = 1.0f - ub;
273 }
274 else
275 {
276 u0 = -1.0f - ub;
277 }
278 }
279
280 else if ((uc_abs >= ua_abs) && (uc_abs >= ub_abs))
281 {
282 if (uc >= 0.0f)
283 {
284 u0 = 1.0f - uc;
285 }
286 else
287 {
288 u0 = -1.0f - uc;
289 }
290 }
291
292 // 零序分量注入
293 ua = ua + u0;
294 ub = ub + u0;
295 uc = uc + u0;
296
297 // 计算COMP寄存器值
298 mod->comp1 = (ua + 1.0f) * mod->half_tbprd;
299 mod->comp2 = (ub + 1.0f) * mod->half_tbprd;
300 mod->comp3 = (uc + 1.0f) * mod->half_tbprd;
301}
