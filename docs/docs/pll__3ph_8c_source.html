44void SRF_3PH_Run(float32_t v_q, SRF_3PH *spll_obj)
45{
46 //
47 // Update the spll_obj->v_q[0] with the grid value
48 //
49 spll_obj->v_q[0] = v_q;
50
51 //
52 // Loop Filter
53 //
54 spll_obj->ylf[0] = spll_obj->ylf[1]
55 + (spll_obj->lpf_coeff.b0 * spll_obj->v_q[0])
56 + (spll_obj->lpf_coeff.b1 * spll_obj->v_q[1]);
57 spll_obj->ylf[1] = spll_obj->ylf[0];
58 spll_obj->v_q[1] = spll_obj->v_q[0];
59
60 spll_obj->ylf[0] = (spll_obj->ylf[0] > (float32_t)(200.0)) ?
61 (float32_t)(200.0) : spll_obj->ylf[0];
62
63 //
64 // VCO
65 //
66 spll_obj->fo = spll_obj->fn + spll_obj->ylf[0];
67
68 spll_obj->theta[0] = spll_obj->theta[1] +
69 ((spll_obj->fo * spll_obj->delta_t) *
70 (float32_t)(2.0 * 3.1415926));
71 if(spll_obj->theta[0] > (float32_t)(2.0 * 3.1415926))
72 {
73 spll_obj->theta[0] = spll_obj->theta[0] - (float32_t)(2.0 * 3.1415926);
74 }
75
76 spll_obj->theta[1] = spll_obj->theta[0];
77}

163void DDSRF_3PH_Run(DDSRF_3PH *spll_obj,
164 float32_t d_p, float32_t d_n,
165 float32_t q_p, float32_t q_n)
166{
167 //
168 // before calling this routine run the ABC_DQ0_Pos & Neg run routines
169 // pass updated values for d_p,d_n,q_p,q_n
170 // and update the cos_2theta and sin_2theta values with the prev angle
171 //
172
173 //
174 // Decoupling Network
175 //
176 spll_obj->d_p_decoupl = d_p
177 - (spll_obj->d_n_decoupl_lpf * spll_obj->cos_2theta)
178 - (spll_obj->q_n_decoupl * spll_obj->sin_2theta);
179 spll_obj->q_p_decoupl = q_p
180 + (spll_obj->d_n_decoupl_lpf * spll_obj->sin_2theta)
181 - (spll_obj->q_n_decoupl * spll_obj->cos_2theta);
182
183 spll_obj->d_n_decoupl = d_n
184 - (spll_obj->d_p_decoupl_lpf * spll_obj->cos_2theta)
185 + (spll_obj->q_p_decoupl * spll_obj->sin_2theta);
186 spll_obj->q_n_decoupl = q_n
187 - (spll_obj->d_p_decoupl_lpf * spll_obj->sin_2theta)
188 - (spll_obj->q_p_decoupl * spll_obj->cos_2theta);
189
190 //
191 // Low pass filter
192 //
193
194 spll_obj->y[1] = (spll_obj->d_p_decoupl * spll_obj->k1)
195 - (spll_obj->y[0] * spll_obj->k2);
196 spll_obj->d_p_decoupl_lpf = spll_obj->y[1] + spll_obj->y[0];
197 spll_obj->y[0] = spll_obj->y[1];
198
199 spll_obj->x[1] = (spll_obj->q_p_decoupl * spll_obj->k1)
200 - (spll_obj->x[0] * spll_obj->k2);
201 spll_obj->q_p_decoupl_lpf = spll_obj->x[1] + spll_obj->x[0];
202 spll_obj->x[0] = spll_obj->x[1];
203
204 spll_obj->w[1] = (spll_obj->d_n_decoupl * spll_obj->k1)
205 - (spll_obj->w[0] * spll_obj->k2);
206 spll_obj->d_n_decoupl_lpf = spll_obj->w[1] + spll_obj->w[0];
207 spll_obj->w[0] = spll_obj->w[1];
208
209 spll_obj->z[1] = (spll_obj->q_n_decoupl * spll_obj->k1)
210 - (spll_obj->z[0] * spll_obj->k2);
211 spll_obj->q_n_decoupl_lpf = spll_obj->z[1] + spll_obj->z[0];
212 spll_obj->z[0] = spll_obj->z[1];
213
214 spll_obj->v_q[0] = spll_obj->q_p_decoupl;
215
216 //
217 // Loop Filter
218 //
219 spll_obj->ylf[0] = spll_obj->ylf[1]
220 + (spll_obj->lpf_coeff.b0 * spll_obj->v_q[0])
221 + (spll_obj->lpf_coeff.b1 * spll_obj->v_q[1]);
222 spll_obj->ylf[1] = spll_obj->ylf[0];
223 spll_obj->v_q[1] = spll_obj->v_q[0];
224
225 //
226 // VCO
227 //
228 spll_obj->fo = spll_obj->fn + spll_obj->ylf[0];
229
230 spll_obj->theta[0] = spll_obj->theta[1] +
231 ((spll_obj->fo * spll_obj->delta_t)
232 * (float32_t)(2.0f * 3.1415926f));
233
234 if(spll_obj->theta[0] > (float32_t)(2.0f * 3.1415926f))
235 {
236 spll_obj->theta[0] = spll_obj->theta[0] -
237 (float32_t)(2.0f * 3.1415926f);
238 }
239
240
241 spll_obj->theta[1] = spll_obj->theta[0];
242
243 spll_obj->cos_2theta = cosf(spll_obj->theta[1] * 2.0f);
244 spll_obj->sin_2theta = sinf(spll_obj->theta[1] * 2.0f);
245}
