278//*****************************************************************************
279//
280// Defines related to clock configuration
281//
282//*****************************************************************************
283//
284// 20MHz XTAL on controlCARD and Launchpad. For use with SysCtl_getClock().
285//
286#define DEVICE_OSCSRC_FREQ 20000000U
287
288//
289// Define to pass to SysCtl_setClock(). Will configure the clock as follows:
290// PLLSYSCLK = 20MHz (XTAL_OSC) * 10 (IMULT) * 1 (FMULT) / 2 (PLLCLK_BY_2)
291//
292#define DEVICE_SETCLOCK_CFG (SYSCTL_OSCSRC_XTAL | SYSCTL_IMULT(10) | \
293 SYSCTL_FMULT_NONE | SYSCTL_SYSDIV(2) | \
294 SYSCTL_PLL_ENABLE)
295
296//
297// 100MHz SYSCLK frequency based on the above DEVICE_SETCLOCK_CFG. Update the
298// code below if a different clock configuration is used!
299//
300#define DEVICE_SYSCLK_FREQ ((DEVICE_OSCSRC_FREQ * 10 * 1) / 2)
301
302//
303// 25MHz LSPCLK frequency based on the above DEVICE_SYSCLK_FREQ and a default
304// low speed peripheral clock divider of 4. Update the code below if a
305// different LSPCLK divider is used!
306//
307#define DEVICE_LSPCLK_FREQ (DEVICE_SYSCLK_FREQ / 4)
