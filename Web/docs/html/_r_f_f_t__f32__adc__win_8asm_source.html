22;;! \fn RFFT_adc_f32_win
23;;! \brief Real FFT (ADC input) Windowing Function
24;;! \param pBuffer (XAR4), pointer to the ADC buffer that needs to be windowed
25;;! \param pWindow (XAR5), pointer to the windowing table
26;;! \param size (AL), size of the buffer (size of the RFFT)
27;;
28;; This function windows a 2N-point real valued data sequence stored as an
29;; N point complex sequence in the location pointed to by "pBuffer". It is to
30;; be used with the real FFT module
31;;
32;; A 2N-pt real FFT is carried out by first doing an N point complex FFT
33;; followed by post processing (unwrapping) to get the correct answer. The
34;; windows are of size N (they are symmetrical).
35;;
36;; This routine will apply the window in two steps
37;;
38;; Step 1: Forward walk through win
39;; Loop for i = 0 to N - 1 {
40;; p -> pBuffer, q -> win
41;; a = *p (u16) , b = *q (u16)
42;; c = *(p+1) (u16), d = *(q+1)(u16)
43;; temp1 = (A x B)>>16 (u16)
44;; temp2 = (C x D)>>16 (u16)
45;; *p = temp1 , *(p+1) = temp2
46;; p = p + 1
47;; q = q + 1
48;; }
49;;
50;; Step 2: Backward walk through win
51;; Loop for i = N to 2N - 1 {
52;; p -> pBuffer(from the last step), q -> win(N/2)
53;; a = *p (u16) , b = *q (u16)
54;; c = *(p+1) (u16), d = *(q+1)(u16)
55;; temp1 = (A x B)>>16 (u16)
56;; temp2 = (C x D)>>16 (u16)
57;; *p = temp1 , *(p+1) = temp2
58;; p = p + 1
59;; q = q - 1
60;; }
61;;
62;; Register Usage
63;; XAR4 -> pBuffer
64;; XAR5 -> pWindow
65;; AL = size
66;; AR7 = loop counter 1/2
67;; R0H = window element
68;; R1H = buffer element
69;; R2H = window element
70;; R3H = buffer element
71;; Registers Save on Entry = 0
72;;=============================================================================
73;;

75_RFFT_adc_f32_win:
76 .asmfunc
77 CLRC SXM ; Turn off sign extension
78 SPM 0 ; No shift of the product
79 ; size -> size of RFFT i.e. 2N-point
80 LSR AL, #2 ; AL = (size/4)
81 MOVZ AR7, AL ; AR7 = (size/2)
82 SUBB XAR7,#1 ; AR7 = (size/2)-1
83
84 .align 2
85 RPTB _RFFT_adc_f32_win_loop1, AR7
86 MOV T, *XAR4 ; T = d[i].real
87 MOV PL, *XAR5++ ; P = win[i++]
88 MPYU ACC, T, PL ; ACC = d[i].real*win[i++]
89 SFR ACC, #16 ; ACC = d[i].real*win[i++] >> 16
90 MOV *XAR4++, ACC ;
91 MOV T, *XAR4 ; T = d[i].imag
92 MOV PL, *XAR5++ ; P = win[i++]
93 MPYU ACC, T, PL ; ACC = d[i].imag*win[i++]
94 SFR ACC, #16 ; ACC = d[i].imag*win[i++] >> 16
95 MOV *XAR4++, ACC ;
96_RFFT_adc_f32_win_loop1:
97 ; At the end of this loop, XAR5 points to the
98 ; N/2 th element of the real input
99 .align 2
100 RPTB _RFFT_adc_f32_win_loop2, AR7
101 MOV T, *XAR4 ; T = d[i].real
102 MOV PL, *--XAR5 ; P = win[i], i--
103 MPYU ACC, T, PL ; ACC = d[i].real*win[i]
104 SFR ACC, #16 ; ACC = d[i].real*win[i] >> 16
105 MOV *XAR4++, ACC ;
106 MOV T, *XAR4 ; T = d[i].imag
107 MOV PL, *--XAR5 ; P = win[i], i--
108 MPYU ACC, T, PL ; ACC = d[i].imag*win[i]
109 SFR ACC, #16 ; ACC = d[i].imag*win[i] >> 16
110 MOV *XAR4++, ACC ;
111_RFFT_adc_f32_win_loop2:
112 LRETR
