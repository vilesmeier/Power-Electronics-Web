76static inline void Power_fil_1ph_Run(POWER_FIL_1PH_T *power, float32_t v, float32_t i)
77{
78 // rms 滤波器, 更新结果
79 power->v_rms = Power_Filter_Run(power->v_fil, v*v);
80 power->i_rms = Power_Filter_Run(power->i_fil, i*i);
81 power->v_rms = sqrtf(power->v_rms);
82 power->i_rms = sqrtf(power->i_rms);
83
84 // 有功 无功滤波器, 更新结果
85 power->p = Power_Filter_Run(power->p_fil, v*i);
86
87 // 其他变量计算
88 power->s = power->v_rms * power->i_rms;
89 power->q = sqrtf(power->s*power->s-power->p*power->p);
90 power->pf = power->p / power->s;
91}
