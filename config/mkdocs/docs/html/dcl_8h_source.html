120static inline void DCL_PI_Run(DCL_PI *p, float32_t rk, float32_t yk)
121{
122 float32_t v1, v5, v7, v8;
123 uint16_t l11, l12, l14, l17, l18, l19;
124
125 v1 = rk - yk;
126 v5 = (v1 * p->Ki * p->i6) + p->i10;
127 p->i10 = v5;
128 v7 = (v1 * p->Kp) + v5;
129 v8 = (v7 > p->Umax) ? p->Umax : v7;
130 v8 = (v8 < p->Umin) ? p->Umin : v8;
131 l17 = ((v7 - v8) == 0) ? 1U : 0U;
132 l11 = (v5 >= p->Imax) ? 1U : 0U;
133 l12 = (v5 <= p->Imin) ? 1U : 0U;
134 l19 = (v5 > 0) ? 1U : 0U;
135 l14 = (v1 > 0) ? 1U : 0U;
136 l18 = l17 & (!(l11 | l12) | (l19 ^ l14));
137 p->i6 = (l18 == 0U) ? 0.0f : 1.0f;
138 p->Output = v8;
139}

251static inline void DCL_PMR_Run(DCL_PMR *p, float32_t rk, float32_t yk, float32_t fk)
252{
253 // 计算偏差
254 float32_t err = rk-yk;
255
256 // P
257 p->Output = p->Kp * err;
258
259 // MR
260 uint16_t i = 0;
261
262 for ( i = 0; i < DCL_HARMONIC_NUM; i++)
263 {
264 p->Output = p->Output + DCL_DF22_Run(&(p->pr[i]), err);
265 }
266
267 // 前馈解耦
268 p->Output = p->Output + fk;
269}
