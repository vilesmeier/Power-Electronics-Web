12void Power_Filter_Set(DCL_DF22_T *fil, float32_t isr_freq, float32_t wc)
13{
14 // 更新结构体指针
15 fil->df22.css = &(fil->css);
16 fil->df22.sps = &(fil->sps);
17 (fil+1U)->df22.css = &((fil+1U)->css);
18 (fil+1U)->df22.sps = &((fil+1U)->sps);
19
20 // 更新滤波器参数
21 float32_t dt = 1.0f / isr_freq;
22 float32_t A0;
23 A0 = 4.0f + 4.0f * wc * dt * 0.9239f + wc * wc * dt * dt;
24 fil->sps.a1 = (2 * wc * wc * dt * dt - 8.0f) / A0;
25 fil->sps.a2 = (4.0f - 4.0f * wc * dt * 0.9239f + wc * wc * dt * dt) / A0;
26 fil->sps.b0 = (wc * wc * dt * dt) / A0;
27 fil->sps.b1 = 2.0f * fil->sps.b0;
28 fil->sps.b2 = fil->sps.b0;
29
30 A0 = 4.0f + 4.0f * wc * dt * 0.3827f + wc * wc * dt * dt;
31 (fil+1U)->sps.a1 = (2 * wc * wc * dt * dt - 8.0f) / A0;
32 (fil+1U)->sps.a2 = (4.0f - 4.0f * wc * dt * 0.3827f + wc * wc * dt * dt) / A0;
33 (fil+1U)->sps.b0 = (wc * wc * dt * dt) / A0;
34 (fil+1U)->sps.b1 = 2.0f * (fil+1U)->sps.b0;
35 (fil+1U)->sps.b2 = (fil+1U)->sps.b0;
36
37 // config interrupt period in DCL CSS
38 DCL_SET_CONTROLLER_PERIOD(&(fil->df22),dt);
39 DCL_SET_CONTROLLER_PERIOD(&((fil+1U)->df22),dt);
40
41 // request for DCL parameter load
42 DCL_REQUEST_UPDATE(&(fil->df22));
43 DCL_REQUEST_UPDATE(&((fil+1U)->df22));
44
45 // acknowledge update request
46 DCL_fupdateDF22(&(fil->df22));
47 DCL_fupdateDF22(&((fil+1U)->df22));
48}
