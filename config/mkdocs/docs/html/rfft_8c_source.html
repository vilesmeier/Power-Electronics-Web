18void RFFT_Set(RFFT_T *rfft, RFFT_DATA_T *rfft_data,float32_t fs, float32_t fgrid)
19{
20 // 配置结构体地址关联
21 rfft->rfft_struct_pointer = &(rfft->rfft_struct);
22 RFFT_f32_setFFTSize(rfft->rfft_struct_pointer, FFT_SIZE);
23 RFFT_f32_setInputPtr(rfft->rfft_struct_pointer, rfft_data->input);
24 RFFT_f32_setOutputPtr(rfft->rfft_struct_pointer, rfft_data->output);
25 RFFT_f32_setTwiddlesPtr(rfft->rfft_struct_pointer, rfft_data->twiddle);
26 RFFT_f32_setMagnitudePtr(rfft->rfft_struct_pointer, rfft_data->magnitude);
27 RFFT_f32_setStages(rfft->rfft_struct_pointer, FFT_STAGE);
28
29 // 计算预畸变因子表
30 RFFT_f32_sincostable(rfft->rfft_struct_pointer);
31
32 // 内部变量
33 rfft->size_inv = 1.0f / FFT_SIZE;
34
35 // 判定FFT频域分析模式
36 rfft->fs = fs;
37 float32_t delta_f;
38 delta_f = fs * rfft->size_inv;
39 if (fabsf(delta_f - fgrid) <= 0.1f)
40 {
41 rfft->mode = FFT_GRID_LOCK;
42 }
43 else
44 {
45 rfft->mode = FFT_PEAK_FIND;
46 }
47}

49void RFFT_Run(RFFT_T *rfft)
50{
51 // RFFT加窗, 若不加窗则跳过
52 #ifdef FFT_WINDOW_ENABLE
53 RFFT_f32_win(rfft->rfft_struct.InBuf, FFT_window, FFT_SIZE);
54 #endif
55
56 // RFFT算法
57 RFFT_f32(rfft->rfft_struct_pointer);
58
59 // 计算幅值, 若使用TMU则调用TMU版本
60 #ifdef __TMS320C28XX_TMU__
61 RFFT_f32_mag_TMU0(rfft->rfft_struct_pointer);
62 #else
63 RFFT_f32_mag(rfft->rfft_struct_pointer);
64 #endif
65
66 // FFT频域分析
67 switch (rfft->mode)
68 {
69 case FFT_GRID_LOCK:
70 RFFT_Grid_Lock(rfft);
71 break;
72 case FFT_PEAK_FIND:
73 RFFT_Peak_Find(rfft);
74 break;
75 case FFT_MODE_NULL:
76 default:
77 break;
78 }
79
80 // DC分量
81 rfft->dc_amp = rfft->rfft_struct.MagBuf[0] * rfft->size_inv * FFT_WINDOW_GAIN;
82
83 // THD
84 float32_t sum = 0;
85 // 默认rfft->amp[0]为基波, 其余为谐波
86 uint16_t i = 0;
87 for (i = 1; i<rfft->peak_num; i++)
88 {
89 sum = sum + rfft->amp[i] * rfft->amp[i];
90 }
91 rfft->thd = sqrtf(sum) / rfft->amp[0];
92}

94void RFFT_Grid_Lock(RFFT_T *rfft)
95{
96 uint16_t i = 0;
97
98 // 幅值最大值搜索
99 float32_t magnitude_max = 0.0f;
100 for (i = 1; i < (FFT_SIZE >> 1) + 1; i++)
101 {
102 if (rfft->rfft_struct.MagBuf[i] > magnitude_max)
103 {
104 magnitude_max = rfft->rfft_struct.MagBuf[i];
105 }
106 }
107
108 // 部分变量初始化清空
109 rfft->peak_num = 0;
110 uint16_t peak_index[FFT_PEAK_MAX] = {0};
111
112 // 以最大幅值*阈限为最小值,大于该值视为一个谐波峰
113 for (i = 1; i < (FFT_SIZE >> 1) + 1; i++)
114 {
115 if (rfft->rfft_struct.MagBuf[i] > FFT_TOLERANCE * magnitude_max)
116 {
117 peak_index[rfft->peak_num] = i;
118 rfft->peak_num = rfft->peak_num + 1U;
119
120 if (rfft->peak_num >= FFT_PEAK_MAX)
121 {
122 break;
123 }
124 }
125 }
126
127 // 换算至实际频率与幅值
128 for (i = 0; i < rfft->peak_num; i++)
129 {
130 rfft->freq[i] = peak_index[i] * rfft->fs * rfft->size_inv;
131 rfft->amp[i] = rfft->rfft_struct.MagBuf[peak_index[i]] * 2.0f * rfft->size_inv * FFT_WINDOW_GAIN;
132 }
133}

135void RFFT_Peak_Find(RFFT_T *rfft)
136{
137 // 幅值归一化
138 uint16_t i = 0;
139
140 // 幅值最大值搜索
141 float32_t magnitude_max = 0.0f;
142 for (i = 1; i < (FFT_SIZE >> 1) + 1; i++)
143 {
144 if (rfft->rfft_struct.MagBuf[i] > magnitude_max)
145 {
146 magnitude_max = rfft->rfft_struct.MagBuf[i];
147 }
148 }
149
150 // 部分变量初始化清空
151 rfft->peak_num = 0;
152 uint16_t peak_index[FFT_PEAK_MAX] = {0};
153
154 // 幅值峰值搜索
155 for (i = 1; i < (FFT_SIZE >> 1); i++)
156 {
157 if ((rfft->rfft_struct.MagBuf[i] >= rfft->rfft_struct.MagBuf[i - 1]) && (rfft->rfft_struct.MagBuf[i] >= rfft->rfft_struct.MagBuf[i + 1]) && (rfft->rfft_struct.MagBuf[i] >= magnitude_max * FFT_TOLERANCE))
158 {
159 peak_index[rfft->peak_num] = i;
160 rfft->peak_num = rfft->peak_num + 1U;
161
162 if (rfft->peak_num >= FFT_PEAK_MAX)
163 {
164 break;
165 }
166 }
167 }
168
169 i = FFT_SIZE >> 1;
170 if ((rfft->rfft_struct.MagBuf[i] >= rfft->rfft_struct.MagBuf[i - 1]) && (rfft->rfft_struct.MagBuf[i] >= magnitude_max * FFT_TOLERANCE) && (rfft->peak_num < FFT_PEAK_MAX))
171 {
172 peak_index[rfft->peak_num++] = i;
173 }
174
175 // 换算至实际频率与幅值
176 for (i = 0; i < rfft->peak_num; i++)
177 {
178 rfft->freq[i] = peak_index[i] * rfft->fs * rfft->size_inv;
179 rfft->amp[i] = rfft->rfft_struct.MagBuf[peak_index[i]] * 2.0f * rfft->size_inv * FFT_WINDOW_GAIN;
180 }
181}
