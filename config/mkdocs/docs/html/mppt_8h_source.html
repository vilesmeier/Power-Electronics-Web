67static inline void MPPT_INCC_Run(MPPT_INCC_T *v, float32_t Vpv, float32_t Ipv)
68{
69 v->Vpv = Vpv;
70 v->Ipv = Ipv;
71 if (v->mppt_enable==1)
72 { /*MPPT mode enable*/
73 if (v->mppt_first == 1)
74 {
75 v->VmppOut= v->Vpv - v->StepFirst;
76 v->VpvOld=v->Vpv;
77 v->IpvOld=v->Ipv;
78 v->mppt_first=0;
79 }
80 else
81 {
82 v->DeltaV = v->Vpv-v->VpvOld ; /* PV voltage change */
83 v->DeltaI = v->Ipv-v->IpvOld ; /* PV current change */
84 if(v->DeltaV == 0) /* DeltaV=0 */
85 {
86 if(v->DeltaI !=0 ) /* dI=0 operating at MPP */
87 {
88 if ( v->DeltaI > 0 ) /* climbing towards MPP */
89 {
90 v->VmppOut=v->Vpv+v->Stepsize; /* Increment Vref */
91 }
92 else
93 {
94 v->VmppOut=v->Vpv-v->Stepsize; /* Decrement Vref */
95 }
96 }
97 }
98 else
99 {
100 v->Cond = v->Ipv*__einvf32(v->Vpv); /*Cond =-(v->Ipv<<10)/v->Vpv;*/
101 v->IncCond = v->DeltaI*__einvf32(v->DeltaV); /*IncCond =(DeltaI<<10)/DeltaV;*/
102 if (v->IncCond != v->Cond) /* Operating at MPP */
103 {
104 if (v->IncCond > (-v->Cond)) /* left of MPP */
105 {
106 v->VmppOut=v->Vpv+v->Stepsize; /* change Vref */
107 }
108 else /* Right of MPP */
109 {
110 v->VmppOut=v->Vpv-v->Stepsize; /* change Vref */
111 }
112 }
113 }
114 if(v->VmppOut < v->MinVolt)
115 {
116 v->VmppOut = v->MinVolt;
117 }
118 else if(v->VmppOut > v->MaxVolt)
119 {
120 v->VmppOut = v->MaxVolt;
121 }
122 else
123 {
124 v->VmppOut = v->VmppOut;
125 }
126 v->VpvOld = v->Vpv;
127 v->IpvOld = v->Ipv;
128 }
129 }
130}

290{

292 v->Ipv = Ipv;
293 if (v->mppt_enable==1)
294 {

296 {
297 v->VmppOut= v->Vpv - (0.02);
298 v->mppt_first=0;
299 v->PanelPower_Prev=v->PanelPower;
300 }
301 else
302 {
303 v->PanelPower=(v->Vpv*v->Ipv);
304 v->DeltaP=v->PanelPower-v->PanelPower_Prev;

306 {
307 v->VmppOut=v->Vpv+v->Stepsize;
308 }
309 else
310 {
311 if (v->DeltaP < -v->DeltaPmin)
312 {
313 v->Stepsize=-v->Stepsize;
314 v->VmppOut=v->Vpv+v->Stepsize;
315 }
316 }
317 v->PanelPower_Prev = v->PanelPower;
318 }
319 if(v->VmppOut < v->MinVolt)
320 {
321 v->VmppOut = v->MinVolt;
322 }
323 else if(v->VmppOut > v->MaxVolt)
324 {
325 v->VmppOut = v->MaxVolt;
