35void DCL_PR_Set(DCL_PR_T *p, float32_t Kp, float32_t Kr, float32_t wc, float32_t w0, float32_t isr_freq)
36{
37 // 更新结构体指针
38 p->pr.df22.css = &(p->pr.css);
39 p->pr.df22.sps = &(p->pr.sps);
40
41 // 更新控制周期
42 DCL_SET_CONTROLLER_PERIOD(&(p->pr.df22), 1.0f / isr_freq);
43
44 // 更新 Shadow parameter
45 float32_t delta_t = (1.0f) / isr_freq;
46 float32_t a0 = 4.0f + 4.0f * delta_t * wc + w0 * w0 * delta_t * delta_t;
47 p->pr.sps.a1 = (2.0f * w0 * w0 * delta_t * delta_t - 8.0f) / a0;
48 p->pr.sps.a2 = (4.0f - 4.0f * delta_t * wc + w0 * w0 * delta_t * delta_t) / a0;
49 p->pr.sps.b0 = (4.0f * Kp + 4.0f * delta_t * (Kp + Kr) * wc + Kp * w0 * w0 * delta_t * delta_t) / a0;
50 p->pr.sps.b1 = (Kp * (2.0f * w0 * w0 * delta_t * delta_t - 8.0f)) / a0;
51 p->pr.sps.b2 = (4.0f * Kp - 4.0f * delta_t * (Kp + Kr) *wc + Kp * w0 *w0 * delta_t * delta_t) / a0;
52
53 // 请求装载
54 DCL_REQUEST_UPDATE(&(p->pr.df22));
55 DCL_fupdateDF22(&(p->pr.df22));
56}

58void DCL_PIR_Set(DCL_PIR_T *p, float32_t kp, float32_t ki, float32_t kr, float32_t Umax, float32_t Umin, float32_t Imax,float32_t wc, float32_t w0, float32_t isr_freq)
59{
60 // 配置结构体指针
61 p->pi.pi.css = &(p->pi.css);
62 p->pi.pi.sps = &(p->pi.sps);
63 p->pr.df22.css = &(p->pr.css);
64 p->pr.df22.sps = &(p->pr.sps);
65
66 // 计算中间变量
67 float32_t delta_t = (1.0f) / isr_freq;
68 float32_t a0 = 4.0f + 4.0f * delta_t * wc + w0 * w0 * delta_t * delta_t;
69
70 // 配置 R 控制器
71 p->pr.sps.a1 = (2.0f * w0 * w0 * delta_t * delta_t - 8.0f) / a0;
72 p->pr.sps.a2 = (4.0f - 4.0f * delta_t * wc + w0 * w0 * delta_t * delta_t) / a0;
73 p->pr.sps.b0 = (4.0f * delta_t * kr * wc) / a0;
74 p->pr.sps.b1 = 0.0f;
75 p->pr.sps.b2 = ( -4.0f * delta_t * kr * wc) / a0;
76
77 // 配置PI控制器
78 p->pi.sps.Umax = Umax;
79 p->pi.sps.Umin = Umin;
80 p->pi.sps.Kp = kp;
81 p->pi.sps.Ki = ki / isr_freq;
82 p->pi.sps.Imax = Imax;
83 p->pi.sps.Imin = -Imax;
84
85 // 更新控制周期
86 DCL_SET_CONTROLLER_PERIOD(&(p->pi.pi), delta_t);
87 DCL_SET_CONTROLLER_PERIOD(&(p->pr.df22), delta_t);
88
89 // 装载系数
90 DCL_REQUEST_UPDATE(&(p->pi.pi));
91 DCL_REQUEST_UPDATE(&(p->pr.df22));
92 DCL_fupdatePI(&(p->pi.pi));
93 DCL_fupdateDF22(&(p->pr.df22));
94}

96void DCL_PMR_Set(DCL_PMR_T *p, float32_t Kp, float32_t *Kr, float32_t wc, float32_t w0, uint16_t *harmonic_set, float32_t isr_freq)
97{
98 p->kp = Kp;
99 float32_t delta_t = (1.0f) / isr_freq;
100
101 uint16_t i = 0;
102 for ( i = 0; i < DCL_HARMONIC_NUM; i++)
103 {
104 // 更新控制周期
105 DCL_SET_CONTROLLER_PERIOD(&(p->pr[i].df22), delta_t);
106
107 // 更新系数
108 float32_t a0 = 4.0f + 4.0f * delta_t * wc + w0 * w0 * delta_t * delta_t * harmonic_set[i] * harmonic_set[i];
109 p->pr[i].sps.a1 = (2.0f * w0 * w0 * delta_t * delta_t * harmonic_set[i] * harmonic_set[i] - 8.0f) / a0;
110 p->pr[i].sps.a2 = (4.0f - 4.0f * delta_t * wc + w0 * w0 * delta_t * delta_t * harmonic_set[i] * harmonic_set[i]) / a0;
111 p->pr[i].sps.b0 = (4.0f * delta_t * Kr[i] * wc) / a0;
112 p->pr[i].sps.b1 = 0.0f;
113 p->pr[i].sps.b2 = ( -4.0f * delta_t * Kr[i] * wc) / a0;
114
115 // 更新系数
116 DCL_REQUEST_UPDATE(&(p->pr[i].df22));
117 DCL_fupdateDF22(&(p->pr[i].df22));
118 }
