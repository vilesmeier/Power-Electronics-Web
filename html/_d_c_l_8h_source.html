111static inline float32_t DCL_runPI_C5(DCL_PI *p, float32_t rk, float32_t yk)
112{
113 float32_t v1, v5, v7, v8;
114 uint16_t l11, l12, l14, l17, l18, l19;
115
116 v1 = rk - yk;
117 v5 = (v1 * p->Ki * p->i6) + p->i10;
118 p->i10 = v5;
119 v7 = (v1 * p->Kp) + v5;
120 v8 = (v7 > p->Umax) ? p->Umax : v7;
121 v8 = (v8 < p->Umin) ? p->Umin : v8;
122 l17 = ((v7 - v8) == 0) ? 1U : 0U;
123 l11 = (v5 >= p->Imax) ? 1U : 0U;
124 l12 = (v5 <= p->Imin) ? 1U : 0U;
125 l19 = (v5 > 0) ? 1U : 0U;
126 l14 = (v1 > 0) ? 1U : 0U;
127 l18 = l17 & (!(l11 | l12) | (l19 ^ l14));
128 p->i6 = (l18 == 0U) ? 0.0f : 1.0f;
129
130#ifdef DCL_TESTPOINTS_ENABLED
131 p->css->tpt = v7;
132#endif
133
134 return(v8);
135}

293static inline void DCL_loadDF22asZPK(DCL_DF22 *p, DCL_ZPK3 *q)
294{
295#ifdef DCL_ERROR_HANDLING_ENABLED
296 p->css->err |= (!F32_IS_VALUE((cimagf(q->z1) + cimagf(q->z2)), 0.0f)) ? ERR_PARAM_INVALID : ERR_NONE;
297 p->css->err |= (!F32_IS_VALUE((cimagf(q->p1) + cimagf(q->p2)), 0.0f)) ? ERR_PARAM_INVALID : ERR_NONE;
298 if (p->css->err)
299 {
300 DCL_GET_ERROR_LOC(p->css);
301 DCL_RUN_ERROR_HANDLER(p->css);
302 }
303#endif
304
305 float32_t beta1 = -(float32_t) crealf(q->z1 + q->z2);
306 float32_t beta0 = (float32_t) crealf(q->z1 * q->z2);
307 float32_t alpha1 = -(float32_t) crealf(q->p1 + q->p2);
308 float32_t alpha0 = (float32_t) crealf(q->p1 * q->p2);
309
310 float32_t T = p->css->T;
311 float32_t a0p = 4.0f + (alpha1 * 2.0f * T) + (alpha0 * T * T);
312
313 p->sps->b0 = q->K * (4.0f + (beta1 * 2.0f * T) + (beta0 * T * T)) / a0p;
314 p->sps->b1 = q->K * (-8.0f + (2.0f * beta0 * T * T)) / a0p;
315 p->sps->b2 = q->K * (4.0f - (beta1 * 2.0f * T) + (beta0 * T * T)) / a0p;
316 p->sps->a1 = (-8.0f + (2.0f * alpha0 * T * T)) / a0p;
317 p->sps->a2 = (4.0f - (alpha1 * 2.0f * T) + (alpha0 * T * T)) / a0p;
318}

328static inline void DCL_loadDF22asZwn(DCL_DF22 *p, float32_t z, float32_t wn)
329{
330#ifdef DCL_ERROR_HANDLING_ENABLED
331 p->css->err |= (z < 0.0f) ? ERR_PARAM_INVALID : ERR_NONE;
332 p->css->err |= (wn < 0.0f) ? ERR_PARAM_INVALID : ERR_NONE;
333 if (p->css->err)
334 {
335 DCL_GET_ERROR_LOC(p->css);
336 DCL_RUN_ERROR_HANDLER(p->css);
337 }
338#endif
339
340 float32_t T = p->css->T;
341 float32_t v1 = wn*wn*T*T;
342 float32_t a2p = 1.0f / (4.0f + 4.0f*z*wn*T + v1);
343 p->sps->b0 = v1 * a2p;
344 p->sps->b1 = 2.0f * p->sps->b0;
345 p->sps->b2 = p->sps->b0;
346 p->sps->a1 = (2.0f*v1 - 8.0f) * a2p;
347 p->sps->a2 = (4.0f - 4.0f*z*wn*T + v1) * a2p;
348}

441static inline void DCL_loadDF23asZPK(DCL_DF23 *p, DCL_ZPK3 *q)
442{
443#ifdef DCL_ERROR_HANDLING_ENABLED
444 p->css->err |= (!F32_IS_VALUE((cimagf(q->z1) + cimagf(q->z2) + cimagf(q->z3)), 0.0f)) ? ERR_PARAM_INVALID : ERR_NONE;
445 p->css->err |= (!F32_IS_VALUE((cimagf(q->p1) + cimagf(q->p2) + cimagf(q->p3)), 0.0f)) ? ERR_PARAM_INVALID : ERR_NONE;
446 if (p->css->err)
447 {
448 DCL_GET_ERROR_LOC(p->css);
449 DCL_RUN_ERROR_HANDLER(p->css);
450 }
451#endif
452
453 float32_t beta2 = -(float32_t) crealf(q->z1 + q->z2 + q->z3);
454 float32_t beta1 = (float32_t) crealf((q->z1 * q->z2) + (q->z2 * q->z3) + (q->z1 * q->z3));
455 float32_t beta0 = -(float32_t) crealf(q->z1 * q->z2 * q->z3);
456
457 float32_t alpha2 = -(float32_t) crealf(q->p1 + q->p2 + q->p3);
458 float32_t alpha1 = (float32_t) crealf((q->p1 * q->p2) + (q->p2 * q->p3) + (q->p1 * q->p3));
459 float32_t alpha0 = -(float32_t) crealf(q->p1 * q->p2 * q->p3);
460
461 float32_t T = p->css->T;
462 float32_t a0p = 8.0f + (alpha2 * 4.0f * T) + (alpha1 * 2.0f * T * T) + (alpha0 * T * T * T);
463
464 p->sps->b0 = q->K * (8.0f + (beta2 * 4.0f * T) + (beta1 * 2.0f * T * T) + (beta0 * T * T * T)) / a0p;
465 p->sps->b1 = q->K * (-24.0f - (beta2 * 4.0f * T) + (beta1 * 2.0f * T * T) + (3.0f * beta0 * T * T * T)) / a0p;
466 p->sps->b2 = q->K * (24.0f - (beta2 * 4.0f * T) - (beta1 * 2.0f * T * T) + (3.0f * beta0 * T * T * T)) / a0p;
467 p->sps->b3 = q->K * (-8.0f + (beta2 * 4.0f * T) - (beta1 * 2.0f * T * T) + (beta0 * T * T * T)) / a0p;
468
469 p->sps->a1 = (-24.0f - (alpha2 * 4.0f * T) + (alpha1 * 2.0f * T * T) + (3.0f * alpha0 * T * T * T)) / a0p;
470 p->sps->a2 = (24.0f - (alpha2 * 4.0f * T) - (alpha1 * 2.0f * T * T) + (3.0f * alpha0 * T * T * T)) / a0p;
471 p->sps->a3 = (-8.0f + (alpha2 * 4.0f * T) - (alpha1 * 2.0f * T * T) + (alpha0 * T * T * T)) / a0p;
472}
