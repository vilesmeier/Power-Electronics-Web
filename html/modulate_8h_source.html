97static inline void Modulate_SVPWM_Run(MOD_T *mod, float32_t ua, float32_t ub, float32_t uc)
98{
99 float32_t max, min;
100
101 ua = (ua+ 1.0f) * 0.5f;
102 ub = (ub + 1.0f) * 0.5f;
103 uc = (uc + 1.0f) * 0.5f;
104
105 // 计算三相最大值、最小值
106 max = (ua >ub) ? ua :ub;
107 max = (max > uc) ? max : uc;
108 min = (ua <ub) ? ua :ub;
109 min = (min < uc) ? min : uc;
110
111 mod->comp1 = (ua + (1.0f - max - min) * 0.5f) * mod->tbprd;
112 mod->comp2 = (ub + (1.0f - max - min) * 0.5f) * mod->tbprd;
113 mod->comp3 = (uc + (1.0f - max - min) * 0.5f) * mod->tbprd;

123static inline void Modulate_DPWM1_Run(MOD_T *mod, float32_t ua, float32_t ub, float32_t uc)
124{
125 float32_t max, min, u0p, u0n, u0;
126
127 // 计算三相最大值、最小值
128 max = (ua > ub) ? ua : ub;
129 max = (max > uc) ? max : uc;
130 min = (ua < ub) ? ua : ub;
131 min = (min < uc) ? min : uc;
132
133 // 产生零序分量
134 // 最大值钳位至 +1
135 u0p = 1.0f - max;
136 // 最小值钳位至 -1
137 u0n = -1.0f - min;
138
139 // 二者取较小的一项作为零序分量
140 u0 = (fabsf(u0p) <= fabsf(u0n)) ? u0p : u0n;
141
142 // 零序分量注入
143 ua = ua + u0;
144 ub = ub + u0;
145 uc = uc + u0;
146
147 // 调制
148 mod->comp1 = (ua + 1.0f) * mod->half_tbprd;
149 mod->comp2 = (ub + 1.0f) * mod->half_tbprd;
150 mod->comp3 = (uc + 1.0f) * mod->half_tbprd;
151}

221static inline void Modulate_GDPWM_Run(MOD_T *mod, float32_t ua, float32_t ub, float32_t uc, float32_t pf_angle)
222{
223 float32_t u0, uaf, ubf, ucf, ua_abs, ub_abs, uc_abs;
224
225 // abc坐标系旋转, 转过角度为功率因数角
226 uaf = ua * cosf(pf_angle) - (uc - ub) * 0.5773502692f * sinf(pf_angle);
227 ubf = ub * cosf(pf_angle) + ((uc - ub) * 0.2886751346f - 0.8660254038f * ua) * sinf(pf_angle);
228 ucf = -uaf - ubf;
229 ua_abs = fabsf(uaf);
230 ub_abs = fabsf(ubf);
231 uc_abs = fabsf(ucf);
232
233 // 判断钳位相, 产生零序分量
234 if ((ua_abs >= ub_abs) && (ua_abs >= uc_abs))
235 {
236 if (ua >= 0.0f)
237 {
238 u0 = 1.0f - ua;
239 }
240 else
241 {
242 u0 = -1.0f - ua;
243 }
244 }
245
246 else if ((ub_abs >= ua_abs) && (ub_abs >= uc_abs))
247 {
248 if (ub >= 0.0f)
249 {
250 u0 = 1.0f - ub;
251 }
252 else
253 {
254 u0 = -1.0f - ub;
255 }
256 }
257
258 else if ((uc_abs >= ua_abs) && (uc_abs >= ub_abs))
259 {
260 if (uc >= 0.0f)
261 {
262 u0 = 1.0f - uc;
263 }
264 else
265 {
266 u0 = -1.0f - uc;
267 }
268 }
269
270 // 零序分量注入
271 ua = ua + u0;
272 ub = ub + u0;
273 uc = uc + u0;
274
275 // 计算COMP寄存器值
276 mod->comp1 = (ua + 1.0f) * mod->half_tbprd;
277 mod->comp2 = (ub + 1.0f) * mod->half_tbprd;
278 mod->comp3 = (uc + 1.0f) * mod->half_tbprd;
279}
