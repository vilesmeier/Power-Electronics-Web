70void Power_int_1ph_Run(POWER_INT_1PH_T *v, float32_t vol, float32_t cur)
71{
72 v->v = vol;
73 v->i = cur;
74 v->vNorm = fabsf(v->v);
75 v->iNorm = fabsf(v->i);
76 v->currSign = ( v->v > v->threshold) ? 1 : 0;
77 v->nSamples++;
78 v->vSum = v->vSum+v->vNorm;
79 v->vSqrSum = v->vSqrSum+(v->vNorm*v->vNorm);
80 v->vEma = v->vEma+(v->emaFilterMultiplier*(v->vNorm - v->vEma));
81 v->iSqrSum = v->iSqrSum+(v->iNorm*v->iNorm);
82 v->pSum = v->pSum+(v->i*v->v);
83 v->zcd=0;
84
85 // when positive zero pass is detected, indecating a full sine wave period
86 if((v->prevSign != v->currSign) && (v->currSign == 1))
87 {
88 //
89 // check if the nSamples are in the ball park of a real frequency
90 // that can be on the grid, this is done by comparing the nSamples
91 // with the max value and min value it can be for the
92 // AC Grid Connection these Max and Min are initialized by the
93 // user in the code
94 //
95 if(v->nSamplesMin < v->nSamples )
96 {
97 v->zcd=1;
98 v->inverse_nSamples = (1.0f)/(v->nSamples);
99 v->sqrt_inverse_nSamples = sqrtf(v->inverse_nSamples);
100 v->vAvg = (v->vSum*v->inverse_nSamples);
101 v->vRms = sqrtf(v->vSqrSum)*v->sqrt_inverse_nSamples;
102 v->iRms = sqrtf(v->iSqrSum)*v->sqrt_inverse_nSamples;
103 v->pRms = v->pSum*v->inverse_nSamples;
104 v->vaRms = v->vRms*v->iRms;
105 v->varRms = v->vaRms * v->vaRms - v->pRms * v->pRms;
106 if(v->varRms <= 0 )
107 {
108 v->varRms = 0;
109 }
110 else
111 {
112 v->varRms = sqrtf(v->varRms);
113 }
114 v->acFreq = (v->sampleFreq*v->inverse_nSamples);
115 v->powerFactor=v->pRms/v->vaRms;
116
117 v->jitterCount=0;
118 v->nSamples=0;
119 v->vSum=0;
120 v->vSqrSum=0;
121 v->iSqrSum=0;
122 v->pSum =0;
123 }
124 else
125 {
126 //
127 // otherwise it may be jitter ignore this reading
128 // but count the number of jitters you are getting
129 // but do not count to infinity as then when the grid comes back
130 // it will take too much time to wind down the jitter count
131 //
132 if(v->jitterCount<30)
133 {
134 v->jitterCount++;
135 }
136 v->nSamples=0;
137 }
138 }
139
140 if(v->nSamples>v->nSamplesMax || v->jitterCount>20)
141 {
142 //
143 // most certainly the AC voltage is not present
144 //
145 v->vRms = 0;
146 v->vAvg = 0;
147 v->vEma = 0;
148 v->acFreq=0;
149 v->iRms = 0;
150 v->pRms = 0;
151 v->vaRms =0;
152 v->powerFactor=0;
153
154 v->zcd=0;
155 v->vSum=0;
156 v->vSqrSum=0;
157 v->iSqrSum=0;
158 v->pSum=0;
159 v->nSamples=0;
160 v->jitterCount=0;
161 }
162 v->prevSign = v->currSign;
163}

208void Power_int_3ph_Run(POWER_INT_3PH_T *power, float32_t Va, float32_t Vb, float32_t Vc, float32_t Ia, float32_t Ib, float32_t Ic)
209{
210 power->Va = Va;
211 power->Vb = Vb;
212 power->Vc = Vc;
213
214 power->Ia = Ia;
215 power->Ib = Ib;
216 power->Ic = Ic;
217
218 // getting current sample number
219 power->sample_num = power->sample_num + 1;
220
221 // rms for volate and current
222 power->va_sum = power->Va * power->Va + power->va_sum;
223 power->vb_sum = power->Vb * power->Vb + power->vb_sum;
224 power->vc_sum = power->Vc * power->Vc + power->vc_sum;
225 power->ia_sum = power->Ia * power->Ia + power->ia_sum;
226 power->ib_sum = power->Ib * power->Ib + power->ib_sum;
227 power->ic_sum = power->Ic * power->Ic + power->ic_sum;
228 power->p_sum = power->Va * power->Ia + power->Vb * power->Ib + power->Vc * power->Ic + power->p_sum;
229
230 // when sample == nsample, indecating get a sine period
231 // update output
232 if (power->nsample == power->sample_num)
233 {
234 // voltage rms
235 power->Varms = sqrtf(power->va_sum * power->nsample_inv);
236 power->Vbrms = sqrtf(power->vb_sum * power->nsample_inv);
237 power->Vcrms = sqrtf(power->vc_sum * power->nsample_inv);
238
239 // current rms
240 power->Iarms = sqrtf(power->ia_sum * power->nsample_inv);
241 power->Ibrms = sqrtf(power->ib_sum * power->nsample_inv);
242 power->Icrms = sqrtf(power->ic_sum * power->nsample_inv);
243
244 // power
245 power->P = power->nsample_inv * power->p_sum;
246 power->S = power->Varms * power->Iarms + power->Vbrms * power->Ibrms + power->Vcrms * power->Icrms;
247
248 power->PF = (power->S > 1e-6f) ? (power->P / power->S) : 0.0f;
249
250 power->Q = sqrtf(power->S * power->S - power->P * power->P);
251
252 // reset integrator
253 power->va_sum = 0;
254 power->vb_sum = 0;
255 power->vc_sum = 0;
256 power->ia_sum = 0;
257 power->ib_sum = 0;
258 power->ic_sum = 0;
259 power->p_sum = 0;
260
261 // reset sample counter
262 power->sample_num = 0;
263 }
