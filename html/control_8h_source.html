158static inline void DCL_PMR_Run(DCL_PMR_T *p, float32_t ref, float32_t feedback, float32_t feedforward)
159{
160 // 计算偏差
161 float32_t err = ref - feedback;
162
163 // P
164 p->output = p->kp * err;
165
166 // MR
167 uint16_t i = 0;
168
169 for ( i = 0; i < DCL_HARMONIC_NUM; i++)
170 {
171 p->output = p->output + DCL_runDF22_C1(&(p->pr[i]), err);
172 }
173
174 // 前馈解耦
175 p->output = p->output + feedforward;
176}
